75void LoadMcStas::exec() {
76 std::string filename = getPropertyValue("Filename");
77 H5::H5File file(filename, H5F_ACC_RDONLY, Nexus::H5Util::defaultFileAcc());
78
79 Nexus::NexusDescriptor const descriptor(filename);
80 auto const &allEntries = descriptor.getAllEntries();
81
82 auto const iterSDS = allEntries.find(Nexus::SCIENTIFIC_DATA_SET);
83 if (iterSDS == allEntries.cend()) {
84 throw std::runtime_error("Could not find any entries.");
85 }
86 auto const &entries = iterSDS->second;
87
88 const char *attributeName = "long_name";
89 std::vector<std::string> eventEntries;
90 std::map<std::string, std::vector<std::string>> histogramEntries;
91 for (auto &entry : entries) {
92 if (entry.find("/entry1/data") == std::string::npos) {
93 continue;
94 }
95
96 Nexus::NexusAddress const address(entry);
97 Nexus::NexusAddress const groupAddress = address.parent_path();
98 std::string const datasetName = address.stem();
99
100 if (groupAddress.stem() == "content_nxs")
101 continue;
102
103 H5::DataSet const dataset = file.openDataSet(address);
104
105 if (!H5Util::hasAttribute(dataset, attributeName)) {
106 continue;
107 }
108
109 std::string nameAttrValue;
110 H5Util::readStringAttribute(dataset, attributeName, nameAttrValue);
111 if (nameAttrValue.find("Neutron_ID") != std::string::npos) {
112 eventEntries.emplace_back(groupAddress);
113 } else if (histogramEntries.find(groupAddress) == histogramEntries.cend()) {
114 histogramEntries[groupAddress] = {datasetName};
115 } else {
116 histogramEntries[groupAddress].emplace_back(datasetName);
117 }
118 }
119
120 std::vector<std::string> scatteringWSNames;
121 if (!eventEntries.empty()) {
122 scatteringWSNames = readEventData(eventEntries, file);
123 }
124 const auto histoWSNames = readHistogramData(histogramEntries, file);
125 file.close();
126
127 // join two vectors together
128 scatteringWSNames.insert(scatteringWSNames.end(), histoWSNames.cbegin(), histoWSNames.cend());
129
130 setProperty("OutputWorkspace", groupWorkspaces(scatteringWSNames));
131}

155std::vector<std::string> LoadMcStas::readEventData(const std::vector<std::string> &eventEntries,
156 const H5::H5File &file) {
157
158 // vector to store output workspaces
159 std::vector<std::string> scatteringWSNames;
160
161 std::string filename = getPropertyValue("Filename");
162 const bool errorBarsSetTo1 = getProperty("ErrorBarsSetTo1");
163
164 Geometry::Instrument_sptr instrument;
165
166 // Initialize progress reporting
167 int reports = 2;
168 const double progressFractionInitial = 0.1;
169 Progress progInitial(this, 0.0, progressFractionInitial, reports);
170
171 std::string instrumentXML;
172 progInitial.report("Loading instrument");
173 try {
174 const H5::Group group = file.openGroup("/entry1/instrument/instrument_xml");
175 const H5::DataSet dataset = group.openDataSet("data");
176 instrumentXML = H5Util::readString(dataset);
177 } catch (...) {
178 g_log.warning() << "\nCould not find the instrument description in the Nexus file:" << filename
179 << " Ignore eventdata from the Nexus file\n";
180 return scatteringWSNames;
181 ;
182 }
183
184 try {
185 std::string instrumentName = "McStas";
186 Geometry::InstrumentDefinitionParser parser(filename, instrumentName, instrumentXML);
187 std::string instrumentNameMangled = parser.getMangledName();
188
189 // Check whether the instrument is already in the InstrumentDataService
190 if (InstrumentDataService::Instance().doesExist(instrumentNameMangled)) {
191 // If it does, just use the one from the one stored there
192 instrument = InstrumentDataService::Instance().retrieve(instrumentNameMangled);
193 } else {
194 // Really create the instrument
195 instrument = parser.parseXML(nullptr);
196 // Add to data service for later retrieval
197 InstrumentDataService::Instance().add(instrumentNameMangled, instrument);
198 }
199 } catch (Kernel::Exception::InstrumentDefinitionError &e) {
200 g_log.warning() << "When trying to read the instrument description in the Nexus file: " << filename
201 << " the following error is reported: " << e.what() << " Ignore eventdata from the Nexus file\n";
202 return scatteringWSNames;
203 ;
204 } catch (...) {
205 g_log.warning() << "Could not parse instrument description in the Nexus file: " << filename
206 << " Ignore eventdata from the Nexus file\n";
207 return scatteringWSNames;
208 ;
209 }
210
211 // create and prepare an event workspace ready to receive the mcstas events
212 progInitial.report("Set up EventWorkspace");
213 EventWorkspace_sptr eventWS(new EventWorkspace());
214 // initialize, where create up front number of eventlists = number of
215 // detectors
216 eventWS->initialize(instrument->getNumberDetectors(), 1, 1);
217 // Set the units
218 eventWS->getAxis(0)->unit() = UnitFactory::Instance().create("TOF");
219 eventWS->setYUnit("Counts");
220 // set the instrument
221 eventWS->setInstrument(instrument);
222 // assign detector ID to eventlists
223
224 std::vector<detid_t> detIDs = instrument->getDetectorIDs();
225
226 for (size_t i = 0; i < instrument->getNumberDetectors(); i++) {
227 eventWS->getSpectrum(i).addDetectorID(detIDs[i]);
228 // spectrum number are treated as equal to detector IDs for McStas data
229 eventWS->getSpectrum(i).setSpectrumNo(detIDs[i]);
230 }
231 // the one is here for the moment for backward compatibility
232 eventWS->rebuildSpectraMapping(true);
233 const auto detIDtoWSIndex = eventWS->getDetectorIDToWorkspaceIndexMap(true);
234
235 bool isAnyNeutrons = false;
236 // to store shortest and longest recorded TOF
237 double shortestTOF(0.0);
238 double longestTOF(0.0);
239
240 // create vector container all the event output workspaces needed
241 const size_t numEventEntries = eventEntries.size();
242 std::string nameOfGroupWS = getProperty("OutputWorkspace");
243 const auto eventDataTotalName = "EventData_" + nameOfGroupWS;
244 std::vector<std::pair<EventWorkspace_sptr, std::string>> allEventWS = {{eventWS, eventDataTotalName}};
245 // if numEventEntries > 1 also create separate event workspaces
246 const bool onlySummedEventWorkspace = getProperty("OutputOnlySummedEventWorkspace");
247 if (!onlySummedEventWorkspace && numEventEntries > 1) {
248 for (const auto &eventEntry : eventEntries) {
249 NexusAddress address(eventEntry);
250 std::string const groupName = address.stem();
251 // create container to hold partial event data
252 // plus the name users will see for it
253 const auto ws_name = groupName + "_" + nameOfGroupWS;
254 allEventWS.emplace_back(eventWS->clone(), ws_name);
255 }
256 }
257
258 Progress progEntries(this, progressFractionInitial, 1.0, numEventEntries * 2);
259
260 // Refer to entry in allEventWS. The first non-summed workspace index is 1
261 auto eventWSIndex = 1u;
262 // Loop over McStas event data components
263 for (const auto &groupAddress : eventEntries) {
264 const H5::Group group = file.openGroup(groupAddress);
265 const H5::DataSet dataset = group.openDataSet("events");
266
267 // open second level entry
268 std::vector<double> data;
269 progEntries.report("read event data from nexus");
270
271 // Need to take into account that the nexus readData method reads a
272 // multi-column data entry
273 // into a vector
274 // The number of data column for each neutron is here hardcoded to (p, x,
275 // y, n, id, t)
276 // Thus we have
277 // column 0 : p neutron wight
278 // column 1 : x x coordinate
279 // column 2 : y y coordinate
280 // column 3 : n accumulated number of neutrons
281 // column 4 : id pixel id
282 // column 5 : t time
283
284 // get info about event data
285 const H5::DataSpace dataspace = dataset.getSpace();
286 const auto rank = dataspace.getSimpleExtentNdims();
287
288 std::vector<hsize_t> dims(rank);
289 dataspace.getSimpleExtentDims(dims.data());
290 if (dims.size() != 2) {
291 g_log.error() << "Event data in McStas nexus file not loaded. Expected "
292 "event data block to be two dimensional\n";
293 return scatteringWSNames;
294 }
295 hsize_t nNeutrons = dims[0];
296 hsize_t numberOfDataColumn = dims[1];
297 if (nNeutrons && numberOfDataColumn != 6) {
298 g_log.error() << "Event data in McStas nexus file expecting 6 columns\n";
299 return scatteringWSNames;
300 ;
301 }
302 if (!isAnyNeutrons && nNeutrons > 0)
303 isAnyNeutrons = true;
304
305 hsize_t start[2];
306 hsize_t step[2];
307
308 // read the event data in blocks. 1 million event is 1000000*6*8 doubles
309 // about 50Mb
310 hsize_t nNeutronsInBlock = 1000000;
311 hsize_t nOfFullBlocks = nNeutrons / nNeutronsInBlock;
312 hsize_t nRemainingNeutrons = nNeutrons - nOfFullBlocks * nNeutronsInBlock;
313 // sum over number of blocks + 1 to cover the remainder
314 for (hsize_t iBlock = 0; iBlock < nOfFullBlocks + 1; iBlock++) {
315 if (iBlock == nOfFullBlocks) {
316 // read remaining neutrons
317 start[0] = nOfFullBlocks * nNeutronsInBlock;
318 start[1] = 0;
319 step[0] = nRemainingNeutrons;
320 step[1] = numberOfDataColumn;
321 } else {
322 // read neutrons in a full block
323 start[0] = iBlock * nNeutronsInBlock;
324 start[1] = 0;
325 step[0] = nNeutronsInBlock;
326 step[1] = numberOfDataColumn;
327 }
328 const hsize_t nNeutronsForthisBlock = step[0]; // number of neutrons read for this block
329 data.resize(nNeutronsForthisBlock * numberOfDataColumn);
330
331 // Check that the type is what it is supposed to be
332 const H5::DataType datatype = dataset.getDataType();
333 if (datatype.getClass() != H5T_FLOAT) {
334 g_log.warning() << "Entry event field is not H5T_FLOAT! It will be skipped.\n";
335 continue;
336 }
337
338 H5::DataSpace memspace(rank, step);
339 dataspace.selectHyperslab(H5S_SELECT_SET, step, start);
340
341 dataset.read(data.data(), H5::PredType::NATIVE_DOUBLE, memspace, dataspace);
342
343 // populate workspace with McStas events
344 progEntries.report("read event data into workspace");
345 for (hsize_t in = 0; in < nNeutronsForthisBlock; in++) {
346 const auto detectorID = static_cast<int>(data[4 + numberOfDataColumn * in]);
347 const double detector_time = data[5 + numberOfDataColumn * in] * 1.0e6; // convert to microseconds
348 if (in == 0 && iBlock == 0) {
349 shortestTOF = detector_time;
350 longestTOF = detector_time;
351 } else {
352 if (detector_time < shortestTOF)
353 shortestTOF = detector_time;
354 if (detector_time > longestTOF)
355 longestTOF = detector_time;
356 }
357
358 const size_t workspaceIndex = detIDtoWSIndex.find(detectorID)->second;
359
360 int64_t pulse_time = 0;
361 WeightedEvent weightedEvent;
362 if (errorBarsSetTo1) {
363 weightedEvent = WeightedEvent(detector_time, pulse_time, data[numberOfDataColumn * in], 1.0);
364 } else {
365 weightedEvent = WeightedEvent(detector_time, pulse_time, data[numberOfDataColumn * in],
366 data[numberOfDataColumn * in] * data[numberOfDataColumn * in]);
367 }
368 allEventWS[0].first->getSpectrum(workspaceIndex) += weightedEvent;
369 if (!onlySummedEventWorkspace && numEventEntries > 1) {
370 allEventWS[eventWSIndex].first->getSpectrum(workspaceIndex) += weightedEvent;
371 }
372 }
373 eventWSIndex++;
374 } // end reading over number of blocks of an event dataset
375 } // end reading over number of event datasets
376
377 // Create a default TOF-vector for histogramming, for now just 2 bins
378 // 2 bins is the standard. However for McStas simulation data it may make
379 // sense to
380 // increase this number for better initial visual effect
381
382 auto axis = HistogramData::BinEdges{shortestTOF - 1, longestTOF + 1};
383
384 // ensure that specified name is given to workspace (eventWS) when added to
385 // outputGroup
386 for (const auto &wsAndName : allEventWS) {
387 const auto ws = wsAndName.first;
388 ws->setAllX(axis);
389 AnalysisDataService::Instance().addOrReplace(wsAndName.second, ws);
390 scatteringWSNames.emplace_back(wsAndName.second);
391 }
392 return scatteringWSNames;
393}

401std::vector<std::string>
402LoadMcStas::readHistogramData(const std::map<std::string, std::vector<std::string>> &histogramEntries,
403 const H5::H5File &file) {
404
405 std::string nameAttrValueYLABEL;
406 std::vector<std::string> histoWSNames;
407
408 for (const auto &entry : histogramEntries) {
409 const auto groupAddress = entry.first;
410 const H5::Group group = file.openGroup(groupAddress);
411
412 std::string nameAttrValueTITLE;
413 H5Util::readStringAttribute(group, "filename", nameAttrValueTITLE);
414
415 if (H5Util::hasAttribute(group, "ylabel")) {
416 H5Util::readStringAttribute(group, "ylabel", nameAttrValueYLABEL);
417 }
418
419 // Find the axis names
420 std::string axis1Name, axis2Name;
421 for (const auto &datasetName : entry.second) {
422 if (datasetName == "ncount")
423 continue;
424 H5::DataSet dataset = group.openDataSet(datasetName);
425
426 if (H5Util::hasAttribute(dataset, "axis")) {
427 const auto axisNo = H5Util::readNumAttributeCoerce<int>(dataset, "axis");
428 if (axisNo == 1)
429 axis1Name = datasetName;
430 else if (axisNo == 2)
431 axis2Name = datasetName;
432 else
433 throw std::invalid_argument("Unknown axis number");
434 }
435 }
436
437 std::vector<double> axis1Values;
438 H5Util::readArray1DCoerce(group, axis1Name, axis1Values);
439 std::vector<double> axis2Values;
440
441 if (axis2Name.length() == 0) {
442 axis2Name = nameAttrValueYLABEL;
443 axis2Values.emplace_back(0.0);
444 } else {
445 H5Util::readArray1DCoerce(group, axis2Name, axis2Values);
446 }
447
448 const size_t axis1Length = axis1Values.size();
449 const size_t axis2Length = axis2Values.size();
450 g_log.debug() << "Axis lengths=" << axis1Length << " " << axis2Length << '\n';
451
452 // Require "data" field
453 std::vector<double> data;
454 H5Util::readArray1DCoerce(group, "data", data);
455
456 // Optional errors field
457 std::vector<double> errors;
458 if (group.exists("errors")) {
459 H5Util::readArray1DCoerce(group, "errors", errors);
460 }
461
462 MatrixWorkspace_sptr ws = WorkspaceFactory::Instance().create("Workspace2D", axis2Length, axis1Length, axis1Length);
463 Axis *axis1 = ws->getAxis(0);
464 axis1->title() = axis1Name;
465 // Set caption
466 auto lblUnit = std::make_shared<Units::Label>();
467 lblUnit->setLabel(axis1Name, "");
468 axis1->unit() = lblUnit;
469
470 auto axis2 = std::make_unique<NumericAxis>(axis2Length);
471 auto axis2Raw = axis2.get();
472 axis2->title() = axis2Name;
473 // Set caption
474 lblUnit = std::make_shared<Units::Label>();
475 lblUnit->setLabel(axis2Name, "");
476 axis2->unit() = lblUnit;
477
478 ws->setYUnit(axis2Name);
479 ws->replaceAxis(1, std::move(axis2));
480
481 for (size_t wsIndex = 0; wsIndex < axis2Length; ++wsIndex) {
482 auto &dataX = ws->mutableX(wsIndex);
483 auto &dataY = ws->mutableY(wsIndex);
484 auto &dataE = ws->mutableE(wsIndex);
485
486 for (size_t j = 0; j < axis1Length; ++j) {
487 // Data is stored in column-major order so we are translating to
488 // row major for Mantid
489 const size_t fileDataIndex = j * axis2Length + wsIndex;
490
491 dataX[j] = axis1Values[j];
492 dataY[j] = data[fileDataIndex];
493 if (!errors.empty())
494 dataE[j] = errors[fileDataIndex];
495 }
496 axis2Raw->setValue(wsIndex, axis2Values[wsIndex]);
497 }
498
499 // set the workspace title
500 ws->setTitle(nameAttrValueTITLE);
501
502 // use the workspace title to create the workspace name
503 std::replace(nameAttrValueTITLE.begin(), nameAttrValueTITLE.end(), ' ', '_');
504
505 // ensure that specified name is given to workspace (eventWS) when added to
506 // outputGroup
507 const std::string outputWS = getProperty("OutputWorkspace");
508 const std::string nameUserSee = nameAttrValueTITLE + "_" + outputWS;
509 AnalysisDataService::Instance().addOrReplace(nameUserSee, ws);
510
511 histoWSNames.emplace_back(ws->getName());
512 }
513 return histoWSNames;
514}

522int LoadMcStas::confidence(Nexus::NexusDescriptorLazy &descriptor) const {
523 if (!descriptor.isEntry("/entry1/simulation/name", Nexus::SCIENTIFIC_DATA_SET)) {
524 return 0;
525 }
526 std::string value = descriptor.getStrData("/entry1/simulation/name");
527 std::transform(value.begin(), value.end(), value.begin(), [](unsigned char c) { return std::toupper(c); });
528 if (value == "MCCODE") {
529 return 98;
530 }
531 return 0;
532}
