77void store_value(const DataObjects::Workspace2D_sptr &ws, int specID, double value, double error, double wavelength,
78 double dwavelength) {
79 auto &X = ws->mutableX(specID);
80 auto &Y = ws->mutableY(specID);
81 auto &E = ws->mutableE(specID);
82 // The following is mostly to make Mantid happy by defining a histogram with
83 // a single bin around the neutron wavelength
84 X[0] = wavelength - dwavelength / 2.0;
85 X[1] = wavelength + dwavelength / 2.0;
86 Y[0] = value;
87 E[0] = error;
88 ws->getSpectrum(specID).setSpectrumNo(specID);
89}

97int LoadSpice2D::confidence(Kernel::FileDescriptor &descriptor) const {
98 if (descriptor.extension() != ".xml")
99 return 0;
100
101 std::istream &is = descriptor.data();
102 int confidence(0);
103
104 { // start of inner scope
105 Poco::XML::InputSource src(is);
106 // Set up the DOM parser and parse xml file
107 DOMParser pParser;
108 Poco::AutoPtr<Document> pDoc;
109 try {
110 pDoc = pParser.parse(&src);
111 } catch (Poco::Exception &e) {
112 throw Kernel::Exception::FileError("Unable to parse File (" + descriptor.filename() + ")", e.displayText());
113 } catch (...) {
114 throw Kernel::Exception::FileError("Unable to parse File:", descriptor.filename());
115 }
116 // Get pointer to root element
117 Element *pRootElem = pDoc->documentElement();
118 if (pRootElem) {
119 if (pRootElem->tagName() == "SPICErack") {
120 confidence = 80;
121 }
122 }
123 } // end of inner scope
124
125 return confidence;
126}

157void LoadSpice2D::exec() {
158
159 setInputPropertiesAsMemberProperties();
160 setTimes();
161 const std::vector<std::string> tags_to_ignore{"Detector", "DetectorWing"};
162 std::map<std::string, std::string> metadata = m_xmlHandler.get_metadata(tags_to_ignore);
163
164 setSansSpiceXmlFormatVersion(metadata);
165 setWavelength(metadata);
166
167 std::vector<int> data = getData("//Data");
168
169 double monitorCounts = 0;
170 from_string<double>(monitorCounts, metadata["Counters/monitor"], std::dec);
171 double countingTime = 0;
172 from_string<double>(countingTime, metadata["Counters/time"], std::dec);
173 createWorkspace(data, metadata["Header/Scan_Title"], monitorCounts, countingTime);
174
175 // Add all metadata to the WS
176 addMetadataAsRunProperties(metadata);
177
178 setMetadataAsRunProperties(metadata);
179
180 // run load instrument
181 std::string instrument = metadata["Header/Instrument"];
182
183 // ugly hack for Biosans wing detector:
184 // it tests if there is metadata tagged with the wing detector
185 // if so, puts the detector in the right angle
186 if (metadata.find("Motor_Positions/det_west_wing_rot") != metadata.end()) {
187 auto angle = boost::lexical_cast<double>(metadata["Motor_Positions/det_west_wing_rot"]);
188 rotateDetector(-angle);
189 }
190 // sample_detector_distances
191 detectorDistance(metadata);
192 detectorTranslation(metadata);
193 runLoadInstrument(instrument, m_workspace);
194 setProperty("OutputWorkspace", m_workspace);
195}

201std::pair<int, int> LoadSpice2D::parseDetectorDimensions(const std::string &dims_str) {
202
203 // Read in the detector dimensions from the Detector tag
204
205 std::pair<int, int> dims = std::make_pair(0, 0);
206
207 boost::regex b_re_sig(R"(INT\d+\[(\d+),(\d+)\])");
208 if (boost::regex_match(dims_str, b_re_sig)) {
209 boost::match_results<std::string::const_iterator> match;
210 boost::regex_search(dims_str, match, b_re_sig);
211 // match[0] is the full string
212 Kernel::Strings::convert(match[1], dims.first);
213 Kernel::Strings::convert(match[2], dims.second);
214 }
215 if (dims.first == 0 || dims.second == 0)
216 g_log.notice() << "Could not read in the number of pixels!" << '\n';
217 return dims;
218}

259

287std::vector<int> LoadSpice2D::getData(const std::string &dataXpath = "//Data") {
288
289 // data container
290 std::vector<int> data;
291 unsigned int totalDataSize = 0;
292
293 // let's see how many detectors we have
294 std::vector<std::string> detectors = m_xmlHandler.get_subnodes(dataXpath);
295 g_log.debug() << "Number the detectors found in Xpath " << dataXpath << " = " << detectors.size() << '\n';
296
297 // iterate every detector in the xml file
298 for (const auto &detector : detectors) {
299 std::string detectorXpath = std::string(dataXpath).append("/").append(detector);
300 // type : INT32[192,256]
301 std::map<std::string, std::string> attributes = m_xmlHandler.get_attributes_from_tag(detectorXpath);
302 std::pair<int, int> dims = parseDetectorDimensions(attributes["type"]);
303
304 // Horrible hack:
305 // Some old files had a: //Data/DetectorWing with dimensions:
306 // 16 x 256 = 4096. This must be ignored as it is not in the IDF.
307 if (detectorXpath.find("DetectorWing") != std::string::npos && dims.first * dims.second <= 4096)
308 break;
309
310 totalDataSize += dims.first * dims.second;
311 g_log.debug() << "Parsing detector XPath " << detectorXpath << " with dimensions: " << dims.first << " x "
312 << dims.second << " = " << dims.first * dims.second << '\n';
313
314 std::string data_str = m_xmlHandler.get_text_from_tag(detectorXpath);
315 g_log.debug() << "The size of detector contents (xpath = " << detectorXpath << ") is " << data_str.size()
316 << " bytes." << '\n';
317
318 // convert string data into a vector<int>
319 std::stringstream iss(data_str);
320 double number;
321 while (iss >> number) {
322 data.emplace_back(static_cast<int>(number));
323 }
324 g_log.debug() << "Detector XPath: " << detectorXpath
325 << " parsed. Total size of data processed up to now = " << data.size() << " from a total of "
326 << totalDataSize << '\n';
327 }
328
329 if (data.size() != totalDataSize) {
330 g_log.error() << "Total data size = " << totalDataSize << ". Parsed data size = " << data.size() << '\n';
331 throw Kernel::Exception::NotImplementedError("Inconsistent data set: There were more data pixels found than "
332 "declared in the Spice XML meta-data.");
333 }
334 return data;
335}

340void LoadSpice2D::createWorkspace(const std::vector<int> &data, const std::string &title, double monitor1_counts,
341 double monitor2_counts) {
342 int nBins = 1;
343 int numSpectra = static_cast<int>(data.size()) + LoadSpice2D::nMonitors;
344
345 m_workspace = std::dynamic_pointer_cast<DataObjects::Workspace2D>(
346 API::WorkspaceFactory::Instance().create("Workspace2D", numSpectra, nBins + 1, nBins));
347 m_workspace->setTitle(title);
348 m_workspace->getAxis(0)->unit() = Kernel::UnitFactory::Instance().create("Wavelength");
349 m_workspace->setYUnit("");
350
351 int specID = 0;
352 // Store monitor counts in the beggining
353 store_value(m_workspace, specID++, monitor1_counts, monitor1_counts > 0 ? sqrt(monitor1_counts) : 0.0, m_wavelength,
354 m_dwavelength);
355 store_value(m_workspace, specID++, monitor2_counts, 0.0, m_wavelength, m_dwavelength);
356
357 // Store detector pixels
358 for (auto count : data) {
359 // Data uncertainties, computed according to the HFIR/IGOR reduction code
360 // The following is what I would suggest instead...
361 // error = count > 0 ? sqrt((double)count) : 0.0;
362 double error = sqrt(0.5 + fabs(static_cast<double>(count) - 0.5));
363 store_value(m_workspace, specID++, count, error, m_wavelength, m_dwavelength);
364 }
365}

400void LoadSpice2D::setBeamTrapRunProperty(std::map<std::string, std::string> &metadata) {
401
402 std::vector<double> trapDiameters = {76.2, 50.8, 76.2, 101.6};
403 // default use the shortest trap
404 double trapDiameterInUse = trapDiameters[1];
405
406 std::vector<double> trapMotorPositions;
407 trapMotorPositions.emplace_back(boost::lexical_cast<double>(metadata["Motor_Positions/trap_y_25mm"]));
408 trapMotorPositions.emplace_back(boost::lexical_cast<double>(metadata["Motor_Positions/trap_y_50mm"]));
409 trapMotorPositions.emplace_back(boost::lexical_cast<double>(metadata["Motor_Positions/trap_y_76mm"]));
410 trapMotorPositions.emplace_back(boost::lexical_cast<double>(metadata["Motor_Positions/trap_y_101mm"]));
411
412 // Check how many traps are in use (store indexes):
413 std::vector<size_t> trapIndexInUse;
414 for (size_t i = 0; i < trapMotorPositions.size(); i++) {
415 if (trapMotorPositions[i] > 26.0) {
416 // Resting positions are below 25. Make sure we have one trap in use!
417 trapIndexInUse.emplace_back(i);
418 }
419 }
420
421 g_log.debug() << "trapIndexInUse length:" << trapIndexInUse.size() << "\n";
422
423 // store trap diameters in use
424 std::vector<double> trapDiametersInUse;
425 trapDiametersInUse.reserve(trapIndexInUse.size());
426 std::transform(trapIndexInUse.cbegin(), trapIndexInUse.cend(), std::back_inserter(trapDiametersInUse),
427 [&trapDiameters](const auto index) { return trapDiameters[index]; });
428
429 g_log.debug() << "trapDiametersInUse length:" << trapDiametersInUse.size() << "\n";
430
431 // The maximum value for the trapDiametersInUse is the trap in use
432 auto trapDiameterInUseIt = std::max_element(trapDiametersInUse.begin(), trapDiametersInUse.end());
433 if (trapDiameterInUseIt != trapDiametersInUse.end())
434 trapDiameterInUse = *trapDiameterInUseIt;
435
436 g_log.debug() << "trapDiameterInUse:" << trapDiameterInUse << "\n";
437
438 addRunProperty<double>("beam-trap-diameter", trapDiameterInUse, "mm");
439}

514void LoadSpice2D::detectorDistance(std::map<std::string, std::string> &metadata) {
515
516 double sample_detector_distance = 0, sample_detector_distance_offset = 0, sample_si_window_distance = 0;
517
518 // check if it's the new format
519 if (metadata.find("Motor_Positions/sample_det_dist") != metadata.end()) {
520 // Old Format
521 from_string<double>(sample_detector_distance, metadata["Motor_Positions/sample_det_dist"], std::dec);
522 sample_detector_distance *= 1000.0;
523 addRunProperty<double>("sample-detector-distance", sample_detector_distance, "mm");
524 sample_detector_distance_offset =
525 addRunProperty<double>(metadata, "Header/tank_internal_offset", "sample-detector-distance-offset", "mm");
526 sample_si_window_distance =
527 addRunProperty<double>(metadata, "Header/sample_to_flange", "sample-si-window-distance", "mm");
528
529 } else {
530 // New format:
531 from_string<double>(sample_detector_distance, metadata["Motor_Positions/flange_det_dist"], std::dec);
532 sample_detector_distance *= 1000.0;
533 addRunProperty<double>("sample-detector-distance-offset", 0, "mm");
534 addRunProperty<double>("sample-detector-distance", sample_detector_distance, "mm");
535 sample_si_window_distance =
536 addRunProperty<double>(metadata, "Header/sample_to_flange", "sample-si-window-distance", "mm");
537 }
538
539 double total_sample_detector_distance;
540 if (metadata.find("Motor_Positions/sdd") != metadata.end()) {
541
542 // When sdd exists overrides all the distances
543 from_string<double>(total_sample_detector_distance, metadata["Motor_Positions/sdd"], std::dec);
544 total_sample_detector_distance *= 1000.0;
545 sample_detector_distance = total_sample_detector_distance;
546
547 addRunProperty<double>("sample-detector-distance-offset", 0, "mm");
548 addRunProperty<double>("sample-detector-distance", sample_detector_distance, "mm");
549 addRunProperty<double>("sample-si-window-distance", 0, "mm");
550
551 g_log.debug() << "Sample-Detector-Distance from SDD tag = " << total_sample_detector_distance << '\n';
552
553 } else {
554 total_sample_detector_distance =
555 sample_detector_distance + sample_detector_distance_offset + sample_si_window_distance;
556 }
557 addRunProperty<double>("total-sample-detector-distance", total_sample_detector_distance, "mm");
558
559 // Add to the log!
560 API::Run &runDetails = m_workspace->mutableRun();
561 auto p = std::make_unique<Mantid::Kernel::TimeSeriesProperty<double>>("sdd");
562 p->addValue(DateAndTime::getCurrentTime(), total_sample_detector_distance);
563 runDetails.addLogData(std::move(p));
564
565 // Store sample-detector distance
566 declareProperty("SampleDetectorDistance", sample_detector_distance, Kernel::Direction::Output);
567}
