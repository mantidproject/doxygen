58void FunctionQDepends::setAttribute(const std::string &attName, const Attr &attValue) {
59 // Q value is tied to WorkspaceIndex if we have a list of Q values
60 if (attName == "WorkspaceIndex") {
61 size_t wi{static_cast<size_t>(attValue.asInt())}; // ah!, the "joys" of C++ strong typing.
62 if (!m_vQ.empty() && wi < m_vQ.size()) {
63 Mantid::API::IFunction::setAttribute(attName, attValue);
64 auto qAttr = getAttribute("Q");
65 qAttr.setDouble(m_vQ.at(wi));
66 // Can't call setAttributeValue, as it will recurse back into here
67 Mantid::API::IFunction::setAttribute("Q", qAttr);
68 }
69 }
70 // Q can be manually changed by user only if list of Q values is empty
71 else if (attName == "Q") {
72 if (m_vQ.empty()) {
73 Mantid::API::IFunction::setAttribute(attName, attValue);
74 }
75 } else {
76 Mantid::API::IFunction::setAttribute(attName, attValue);
77 }
78}

116std::vector<double> FunctionQDepends::extractQValues(const Mantid::API::MatrixWorkspace &workspace) {
117 std::vector<double> qs;
118 // Check if the vertical axis has units of momentum transfer, then extract Q
119 // values...
120 auto axis_ptr = dynamic_cast<Mantid::API::NumericAxis *>(workspace.getAxis(1));
121 if (axis_ptr) {
122 const std::shared_ptr<Kernel::Unit> &unit_ptr = axis_ptr->unit();
123 if (unit_ptr->unitID() == "MomentumTransfer") {
124 qs = axis_ptr->getValues();
125 }
126 }
127 // ...otherwise, compute the momentum transfer for each spectrum, if possible
128 else {
129 const auto &spectrumInfo = workspace.spectrumInfo();
130 size_t numHist = workspace.getNumberHistograms();
131 for (size_t wi = 0; wi < numHist; wi++) {
132 if (spectrumInfo.hasDetectors(wi)) {
133 const auto detID = spectrumInfo.detector(wi).getID();
134 double efixed = workspace.getEFixed(detID);
135 double usignTheta = 0.5 * spectrumInfo.twoTheta(wi);
136 double q = Mantid::Kernel::UnitConversion::convertToElasticQ(usignTheta, efixed);
137 qs.emplace_back(q);
138 } else {
139 g_log.debug("Cannot populate Q values from workspace");
140 qs.clear();
141 break;
142 }
143 }
144 }
145 return qs;
146}
