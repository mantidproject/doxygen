44void LoadRawSpectrum0::exec() {
45 // Retrieve the filename from the properties
46 m_filename = getPropertyValue("Filename");
47
48 bool bLoadlogFiles = getProperty("LoadLogFiles");
49
50 // open the raw file
51 FILE *file = openRawFile(m_filename);
52
53 // Need to check that the file is not a text file as the ISISRAW routines
54 // don't deal with these very well, i.e
55 // reading continues until a bad_alloc is encountered.
56 if (Kernel::FileDescriptor::isAscii(file)) {
57 g_log.error() << "File \"" << m_filename << "\" is not a valid RAW file.\n";
58 throw std::invalid_argument("Incorrect file type encountered.");
59 }
60
61 std::string title;
62 readTitle(file, title);
63
64 readworkspaceParameters(m_numberOfSpectra, m_numberOfPeriods, m_lengthIn, m_noTimeRegimes);
65
66 // Calculate the size of a workspace, given its number of periods & spectra to
67 // read
68 const int64_t total_specs = 1;
69
70 // Get the time channel array(s) and store in a vector inside a shared pointer
71 auto timeChannelsVec = getTimeChannels(m_noTimeRegimes, m_lengthIn);
72
73 auto histTotal = static_cast<double>(total_specs * m_numberOfPeriods);
74 int64_t histCurrent = -1;
75
76 // Create the 2D workspace for the output
77 DataObjects::Workspace2D_sptr localWorkspace = createWorkspace(total_specs, m_lengthIn, m_lengthIn - 1, title);
78
79 Run &run = localWorkspace->mutableRun();
80
81 if (bLoadlogFiles) {
82 runLoadLog(m_filename, localWorkspace, 0.0, 0.0);
83 const int period_number = 1;
84 createPeriodLogs(period_number, localWorkspace);
85 }
86 // Set the total proton charge for this run
87 setProtonCharge(run);
88
89 WorkspaceGroup_sptr wsgrp_sptr = createGroupWorkspace();
90 setWorkspaceProperty("OutputWorkspace", title, wsgrp_sptr, localWorkspace, m_numberOfPeriods, false, this);
91
92 // Loop over the number of periods in the raw file, putting each period in a
93 // separate workspace
94 for (int period = 0; period < m_numberOfPeriods; ++period) {
95 if (period > 0) {
96 localWorkspace =
97 std::dynamic_pointer_cast<DataObjects::Workspace2D>(WorkspaceFactory::Instance().create(localWorkspace));
98
99 if (bLoadlogFiles) {
100 // remove previous period data
101 std::stringstream prevPeriod;
102 prevPeriod << "PERIOD " << (period);
103 Run &runObj = localWorkspace->mutableRun();
104 runObj.removeLogData(prevPeriod.str());
105 runObj.removeLogData("current_period");
106 // add current period data
107 auto period_number = static_cast<int>(period + 1);
108 createPeriodLogs(period_number, localWorkspace);
109 }
110 // skip all spectra except the first one in each period
111 for (int i = 1; i <= m_numberOfSpectra; ++i) {
112 skipData(file, i + (period - 1) * (m_numberOfSpectra + 1));
113 }
114 }
115
116 int64_t wsIndex = 0;
117 // for each period read first spectrum
118 int64_t histToRead = period * (static_cast<int64_t>(m_numberOfSpectra) + 1);
119
120 progress(m_prog, "Reading raw file data...");
121 // isisRaw->readData(file, histToRead);
122 // readData(file, histToRead);
123 // read spectrum
124 if (!readData(file, histToRead)) {
125 throw std::runtime_error("Error reading raw file");
126 }
127
128 // set the workspace data
129 setWorkspaceData(localWorkspace, timeChannelsVec, wsIndex, 0, m_noTimeRegimes, m_lengthIn, 1);
130
131 if (m_numberOfPeriods == 1) {
132 if (++histCurrent % 100 == 0) {
133 m_prog = double(histCurrent) / histTotal;
134 }
135 interruption_point();
136 }
137 if (m_numberOfPeriods > 1) {
138 setWorkspaceProperty(localWorkspace, wsgrp_sptr, period, false, this);
139 // progress for workspace groups
140 m_prog = static_cast<double>(period) / static_cast<double>(m_numberOfPeriods - 1);
141 }
142
143 } // loop over periods
144
145 // Clean up
146 reset();
147 fclose(file);
148}
