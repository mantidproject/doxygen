152 for (size_t j = 0; j < m_data[i].size(); j++) {
153 double num = fabs((double)m_data[i][j] - (double)otherData[i][j]);
154 double den = (fabs((double)m_data[i][j]) + fabs((double)otherData[i][j])) / 2;
155 if (den < tolerance && num > tolerance) {
