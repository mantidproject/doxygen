49void loadSampleDataISIScompatibilityInfo(Nexus::File &file, Mantid::API::MatrixWorkspace_sptr const &WS) {
50 try {
51 file.openGroup("isis_vms_compat", "IXvms");
52 } catch (Nexus::Exception const &) {
53 // No problem, it just means that this entry does not exist
54 return;
55 }
56
57 // read the data
58 try {
59 std::vector<int32_t> spb;
60 std::vector<float> rspb;
61 file.readData("SPB", spb);
62 file.readData("RSPB", rspb);
63
64 WS->mutableSample().setGeometryFlag(spb[2]); // the flag is in the third value
65 WS->mutableSample().setThickness(rspb[3]);
66 WS->mutableSample().setHeight(rspb[4]);
67 WS->mutableSample().setWidth(rspb[5]);
68 } catch (Nexus::Exception const &ex) {
69 // it means that the data was not as expected, report the problem
70 std::stringstream s;
71 s << "Wrong definition found in isis_vms_compat :> " << ex.what();
72 file.closeGroup();
73 throw std::runtime_error(s.str());
74 }
75
76 file.closeGroup();
77}

153//------------------------------------------------------------------------------
158void LoadNexusMonitors2::exec() {
159 // Retrieve the filename from the properties
160 m_filename = this->getPropertyValue("Filename");
161
162 API::Progress prog1(this, 0.0, 0.2, 2);
163
164 if (!canOpenAsNeXus(m_filename)) {
165 throw std::runtime_error("Failed to recognize this file as a NeXus file, cannot continue.");
166 }
167
168 m_top_entry_name = this->getPropertyValue("NXentryName");
169
170 // top level file information
171 Nexus::File file(m_filename);
172
173 // open the correct entry
174 using string_map_t = std::map<std::string, std::string>;
175 string_map_t entries = file.getEntries();
176
177 if (m_top_entry_name.empty()) {
178 const auto it = std::find_if(entries.cbegin(), entries.cend(), [](const auto &entry) {
179 return ((entry.first == "entry" || entry.first == "raw_data_1") && entry.second == "NXentry");
180 });
181 if (it != entries.cend()) {
182 file.openGroup(it->first, it->second);
183 m_top_entry_name = it->first;
184 }
185 } else {
186 if (!keyExists(m_top_entry_name, entries)) {
187 throw std::invalid_argument(m_filename + " does not contain an entry named " + m_top_entry_name);
188 }
189 file.openGroup(m_top_entry_name, "NXentry"); // Open as will need to be
190 // open for subsequent operations
191 }
192 prog1.report();
193
194 size_t numPeriods = 0;
195 m_monitor_count = getMonitorInfo(file, numPeriods);
196 // Fix the detector numbers if the defaults above are not correct
197 // fixUDets(detector_numbers, file, spectra_numbers, m_monitor_count);
198 // a temporary place to put the spectra/detector numbers
199 // this gets the ids from the "isis_vms_compat" group
200 fixUDets(file);
201
202 if (numPeriods > 0) {
203 m_multiPeriodCounts.resize(m_monitor_count);
204 m_multiPeriodBinEdges.resize(m_monitor_count);
205 }
206
207 // Nothing to do
208 if (0 == m_monitor_count) {
209 // previous version just used to return, but that
210 // threw an error when the OutputWorkspace property was not set.
211 // and the error message was confusing.
212 // This has changed to throw a specific error.
213 throw std::invalid_argument(m_filename + " does not contain any monitors");
214 }
215
216 // only used if using event monitors
217 // EventWorkspace_sptr eventWS;
218 // Create the output workspace
219 std::vector<bool> loadMonitorFlags;
220 bool useEventMon = createOutputWorkspace(loadMonitorFlags);
221
222 API::Progress prog3(this, 0.6, 1.0, m_monitor_count);
223
224 // cache address to entry for later
225 const std::string entryAddress = file.getAddress();
226
227 // TODO-NEXT: load event monitor if it is required to do so
228 // load histogram monitor if it is required to do so
229 // Require a tuple: monitorNames[i], loadAsEvent[i], loadAsHistogram[i]
230 for (std::size_t ws_index = 0; ws_index < m_monitor_count; ++ws_index) {
231 // already know spectrum and detector numbers
232 g_log.debug() << "monIndex = " << m_monitorInfo[ws_index].detNum << '\n';
233 g_log.debug() << "spectrumNo = " << m_monitorInfo[ws_index].specNum << '\n';
234 m_workspace->getSpectrum(ws_index).setSpectrumNo(m_monitorInfo[ws_index].specNum);
235 m_workspace->getSpectrum(ws_index).setDetectorID(m_monitorInfo[ws_index].detNum);
236
237 // Don't actually read all of the monitors
238 g_log.information() << "Loading " << m_monitorInfo[ws_index].name;
239 if (loadMonitorFlags[ws_index]) {
240 g_log.information() << "\n";
241 file.openGroup(m_monitorInfo[ws_index].name, "NXmonitor");
242 if (useEventMon) {
243 // load as an event monitor
244 readEventMonitorEntry(file, ws_index);
245 } else {
246 // load as a histogram monitor
247 readHistoMonitorEntry(file, ws_index, numPeriods);
248 }
249 file.closeGroup(); // NXmonitor
250 } else {
251 g_log.information() << " is skipped.\n";
252 }
253
254 prog3.report();
255 }
256
257 if (useEventMon) // set the x-range to be the range for min/max events
258 {
259 EventWorkspace_sptr eventWS = std::dynamic_pointer_cast<EventWorkspace>(m_workspace);
260 double xmin, xmax;
261 eventWS->getEventXMinMax(xmin, xmax);
262
263 if (xmin > xmax) {
264 xmin = 0;
265 xmax = 1;
266 if (eventWS->getNumberEvents() == 0) {
267 g_log.warning("No events loading. Resetting time-of-flight range 0 to 1");
268 } else {
269 g_log.warning("time-of-flight range of events are unusual. Resetting time-of-flight range 0 to 1");
270 }
271 } else {
272 // move out by one just like LoadEventNexus
273 xmin = xmin - 1;
274 xmax = xmax + 1;
275 }
276
277 auto axis = HistogramData::BinEdges{xmin, xmax};
278 eventWS->setAllX(axis); // Set the binning axis using this.
279
280 // a certain generation of ISIS files modify the time-of-flight
281 const std::string currentAddress = file.getAddress();
282 adjustTimeOfFlightISISLegacy(file, eventWS, m_top_entry_name, "NXmonitor");
283 file.openAddress(currentAddress); // reset to where it was earlier
284 }
285
286 // Check for and ISIS compat block to get the detector IDs for the loaded
287 // spectrum numbers
288 // @todo: Find out if there is a better (i.e. more generic) way to do this
289 try {
290 g_log.debug() << "Load Sample data isis\n";
291 loadSampleDataISIScompatibilityInfo(file, m_workspace);
292 } catch (Nexus::Exception const &) {
293 }
294
295 // Need to get the instrument name from the file
296 std::string instrumentName;
297 file.openAddress(entryAddress); // reset address in case of unusual behavior
298 file.openGroup("instrument", "NXinstrument");
299 try {
300 file.openData("name");
301 instrumentName = file.getStrData();
302 // Now let's close the file as we don't need it anymore to load the
303 // instrument.
304 file.closeData();
305 file.closeGroup(); // Close the NXentry
306 file.close();
307
308 } catch (std::runtime_error &) // no correct instrument definition (old ISIS
309 // file, fall back to isis_vms_compat)
310 {
311 file.closeGroup(); // Close the instrument NXentry
312 instrumentName = LoadEventNexus::readInstrumentFromISIS_VMSCompat(file);
313 file.close();
314 }
315
316 g_log.debug() << "Instrument name read from NeXus file is " << instrumentName << '\n';
317
318 m_workspace->getAxis(0)->unit() = Kernel::UnitFactory::Instance().create("TOF");
319 m_workspace->setYUnit("Counts");
320
321 // Load the logs
322 this->runLoadLogs(m_filename, m_workspace);
323
324 // Old SNS files don't have this
325 try {
326 // The run_start will be loaded from the pulse times.
327 Types::Core::DateAndTime run_start(0, 0);
328 run_start = m_workspace->getFirstPulseTime();
329 m_workspace->mutableRun().addProperty("run_start", run_start.toISO8601String(), true);
330 } catch (...) {
331 // Old files have the start_time defined, so all SHOULD be good.
332 // The start_time, however, has been known to be wrong in old files.
333 }
334 // Load the instrument
335 LoadEventNexus::loadInstrument(m_filename, m_workspace, m_top_entry_name, this);
336
337 // Load the meta data, but don't stop on errors
338 g_log.debug() << "Loading metadata\n";
339 try {
340 LoadEventNexus::loadEntryMetadata<API::MatrixWorkspace_sptr>(m_filename, m_workspace, m_top_entry_name);
341 } catch (std::exception &e) {
342 g_log.warning() << "Error while loading meta data: " << e.what() << '\n';
343 }
344
345 // add filename
346 m_workspace->mutableRun().addProperty("Filename", m_filename);
347
348 if ((numPeriods > 0) && (!useEventMon)) {
349 splitMutiPeriodHistrogramData(numPeriods);
350 } else {
351 this->setProperty("OutputWorkspace", m_workspace);
352 }
353}

362void LoadNexusMonitors2::fixUDets(Nexus::File &file) {
363 const size_t nmonitors = m_monitorInfo.size();
364 boost::scoped_array<detid_t> det_ids(new detid_t[nmonitors]);
365 boost::scoped_array<specnum_t> spec_ids(new specnum_t[nmonitors]);
366 // convert the monitor info into two arrays for resorting
367 for (size_t i = 0; i < nmonitors; ++i) {
368 spec_ids[i] = m_monitorInfo[i].specNum;
369 det_ids[i] = m_monitorInfo[i].detNum;
370 }
371
372 try {
373 file.openGroup("isis_vms_compat", "IXvms");
374 } catch (Nexus::Exception const &) {
375 return;
376 }
377 // UDET
378 file.openData("UDET");
379 std::vector<int32_t> udet;
380 file.getData(udet);
381 file.closeData();
382 // SPEC
383 file.openData("SPEC");
384 std::vector<int32_t> spec;
385 file.getData(spec);
386 file.closeData();
387
388 // This is a little complicated: Each value in the spec_id array is a value
389 // found in the
390 // SPEC block of isis_vms_compat. The index that this value is found at then
391 // corresponds
392 // to the index within the UDET block that holds the detector ID
393 std::vector<int32_t>::const_iterator beg = spec.begin();
394 for (size_t mon_index = 0; mon_index < nmonitors; ++mon_index) {
395 std::vector<int32_t>::const_iterator itr = std::find(spec.begin(), spec.end(), spec_ids[mon_index]);
396 if (itr == spec.end()) {
397 det_ids[mon_index] = -1;
398 continue;
399 }
400 std::vector<int32_t>::difference_type udet_index = std::distance(beg, itr);
401 det_ids[mon_index] = udet[udet_index];
402 }
403 file.closeGroup();
404
405 // copy the information back into the monitorinfo
406 for (size_t i = 0; i < nmonitors; ++i) {
407 m_monitorInfo[i].specNum = spec_ids[i];
408 m_monitorInfo[i].detNum = det_ids[i];
409 }
410}

412void LoadNexusMonitors2::runLoadLogs(const std::string &filename, const API::MatrixWorkspace_sptr &localWorkspace) {
413 // get the properties for which logs to use
414 const std::vector<std::string> allow_list = getProperty(PropertyNames::LOGS_ALLOW);
415 const std::vector<std::string> block_list = getProperty(PropertyNames::LOGS_BLOCK);
416
417 // do the actual work
418 auto loadLogs = createChildAlgorithm("LoadNexusLogs");
419
420 // Now execute the Child Algorithm. Catch and log any error, but don't stop.
421 try {
422 g_log.information() << "Loading logs from NeXus file...\n";
423 loadLogs->setPropertyValue("Filename", filename);
424 loadLogs->setProperty<API::MatrixWorkspace_sptr>("Workspace", localWorkspace);
425 // copy properties for which logs to use/not use
426 loadLogs->setProperty<std::vector<std::string>>(PropertyNames::LOGS_ALLOW, allow_list);
427 loadLogs->setProperty<std::vector<std::string>>(PropertyNames::LOGS_BLOCK, block_list);
428 loadLogs->execute();
429 } catch (...) {
430 g_log.error() << "Error while loading Logs from Nexus. Some sample logs "
431 "may be missing.\n";
432 }
433}

462void LoadNexusMonitors2::splitMutiPeriodHistrogramData(const size_t numPeriods) {
463 // Y array should be divisible by the number of periods
464 if (m_multiPeriodCounts[0].size() % numPeriods != 0) {
465 g_log.warning() << "Attempted to split multiperiod histogram workspace with " << m_multiPeriodCounts[0].size()
466 << "data entries, into " << numPeriods
467 << "periods."
468 " Aborted.\n";
469 return;
470 }
471
472 size_t yLength = m_multiPeriodCounts[0].size() / numPeriods;
473 size_t xLength = yLength + 1;
474 size_t numSpectra = m_workspace->getNumberHistograms();
475 API::ISISRunLogs monLogCreator(m_workspace->run());
476
477 BinEdges edges = m_multiPeriodBinEdges[0];
478 std::vector<API::MatrixWorkspace_sptr> outputWorkspaces;
479 outputWorkspaces.reserve(numPeriods);
480
481 for (size_t i = 0; i < numPeriods; i++) {
482 // create the period workspace
483 API::MatrixWorkspace_sptr wsPeriod =
484 API::WorkspaceFactory::Instance().create(m_workspace, numSpectra, xLength, yLength);
485
486 auto offset = yLength * i;
487
488 for (size_t wsIndex = 0; wsIndex < numSpectra; wsIndex++) {
489 auto inYBegin = m_multiPeriodCounts[wsIndex].cbegin() + offset;
490
491 wsPeriod->setHistogram(wsIndex, edges, Counts(inYBegin, inYBegin + yLength));
492 }
493 // add period logs
494 monLogCreator.addStatusLog(wsPeriod->mutableRun());
495 monLogCreator.addPeriodLogs(static_cast<int>(i + 1), wsPeriod->mutableRun());
496
497 outputWorkspaces.push_back(std::move(wsPeriod));
498 }
499
500 // set output workspace to either single period ws, or group ws.
501 if (outputWorkspaces.size() > 1) {
502 WorkspaceGroup_sptr wsGroup(new WorkspaceGroup);
503 for (const auto &ws : outputWorkspaces) {
504 wsGroup->addWorkspace(ws);
505 }
506 this->setProperty("OutputWorkspace", wsGroup);
507 } else if (outputWorkspaces.size() == 1) {
508 this->setProperty("OutputWorkspace", outputWorkspaces[0]);
509 } else {
510 g_log.error() << "No period workspaces generated to output.";
511 }
512}

514size_t LoadNexusMonitors2::getMonitorInfo(Nexus::File &file, size_t &numPeriods) {
515 // should already be open to the correct NXentry
516
517 m_monitorInfo.clear();
518
519 using string_map_t = std::map<std::string, std::string>;
520
521 // Now we want to go through and find the monitors
522 string_map_t entries = file.getEntries();
523 numPeriods = 0;
524 // we want to sort monitors by monitor_number if they are present
525
526 API::Progress prog2(this, 0.2, 0.6, entries.size());
527
528 string_map_t::const_iterator it = entries.begin();
529 for (; it != entries.end(); ++it) {
530 std::string entry_name(it->first);
531 std::string entry_class(it->second);
532 if ((entry_class == "NXmonitor")) {
533 MonitorInfo info;
534
535 // check for event/histogram monitor
536 // -> This will prefer event monitors over histogram
537 // if they are found in the same group.
538 file.openGroup(entry_name, "NXmonitor");
539 info.name = entry_name;
540 info.hasEvent = isEventMonitor(file);
541 info.hasHisto = isHistoMonitor(file);
542
543 // get the detector number
544 string_map_t inner_entries = file.getEntries(); // get list of entries
545 if (inner_entries.find("monitor_number") != inner_entries.end()) {
546 // get monitor number from field in file
547 const auto detNum = Nexus::IOHelper::readNexusValue<int64_t>(file, "monitor_number");
548 if (detNum > std::numeric_limits<detid_t>::max()) {
549 throw std::runtime_error("Monitor number too larger to represent");
550 }
551 info.detNum = static_cast<detid_t>(detNum);
552 } else {
553 // default creates it from monitor name
554 std::filesystem::path monPath(entry_name);
555 std::string monitorName = monPath.stem().string();
556
557 // check for monitor name - in our case will be of the form either
558 // monitor1
559 // or monitor_1
560 std::string::size_type loc = monitorName.rfind('_');
561 if (loc == std::string::npos) {
562 loc = monitorName.rfind('r');
563 }
564
565 info.detNum = -1 * boost::lexical_cast<int>(monitorName.substr(loc + 1)); // SNS default
566 }
567
568 // get the spectrum number
569 if (inner_entries.find("spectrum_index") != inner_entries.end()) {
570 file.openData("spectrum_index");
571 file.getData(&info.specNum);
572 file.closeData();
573 } else {
574 // default is to match the detector number
575 info.specNum = std::abs(info.detNum);
576 }
577
578 if (info.hasHisto && (numPeriods == 0) && (inner_entries.find("period_index") != inner_entries.end())) {
579 MantidVec period_data;
580 file.openData("period_index");
581 file.getDataCoerce(period_data);
582 file.closeData();
583 numPeriods = period_data.size();
584 }
585
586 file.closeGroup(); // close NXmonitor
587 m_monitorInfo.emplace_back(info);
588 }
589 prog2.report();
590 }
591
592 // sort based on the absolute value of the monitor number
593 // this takes care of the fact that SNS monitors have negative numbers
594 std::sort(m_monitorInfo.begin(), m_monitorInfo.end(), [](const MonitorInfo &left, const MonitorInfo &right) {
595 return std::abs(left.detNum) < std::abs(right.detNum);
596 });
597
598 return m_monitorInfo.size();
599}

601bool LoadNexusMonitors2::createOutputWorkspace(std::vector<bool> &loadMonitorFlags) {
602 loadMonitorFlags.clear();
603
604 size_t numEventMon =
605 std::count_if(m_monitorInfo.begin(), m_monitorInfo.end(), [](const MonitorInfo &info) { return info.hasEvent; });
606 size_t numHistoMon =
607 std::count_if(m_monitorInfo.begin(), m_monitorInfo.end(), [](const MonitorInfo &info) { return info.hasHisto; });
608
609 bool useEventMon; // which type of workspace to create/is created
610 const std::string loadType = getProperty("LoadOnly");
611 if (loadType == LOAD_EVENTS) {
612 useEventMon = true;
613 if (numEventMon == 0) { // make sure there are some
614 throw std::runtime_error("Loading event data. Trying to load event data but failed to "
615 "find event monitors. This file may be corrupted or it may not be "
616 "supported");
617 }
618 } else if (loadType == LOAD_HISTO) {
619 useEventMon = false;
620 if (numHistoMon == 0) { // make sure there are some
621 throw std::runtime_error("Not loading event data. Trying to load histogram data but failed to "
622 "find monitors with histogram data or could not interpret the data. "
623 "This file may be corrupted or it may not be supported");
624 }
625 } else { // only other option is to go with the default
626 if (numEventMon > 0 && numHistoMon > 0) {
627 std::stringstream errmsg;
628 errmsg << "There are " << numHistoMon << " histogram monitors and " << numEventMon
629 << " event monitors. Loading Histogram by default. "
630 << "Use \"LoadOnly\" or \"MonitorLoadOnly\" to specify which to "
631 "load.";
632 m_log.warning(errmsg.str());
633 useEventMon = false;
634 } else {
635 // more than one event monitor means use that since both can't be nonzero
636 useEventMon = (numEventMon > 0);
637 }
638 }
639
640 // set up the flags to load monitor
641 if (useEventMon) {
642 // load event
643 for (size_t i_mon = 0; i_mon < m_monitor_count; ++i_mon) {
644 loadMonitorFlags.emplace_back(m_monitorInfo[i_mon].hasEvent);
645 }
646 } else {
647 // load histogram
648 for (size_t i_mon = 0; i_mon < m_monitor_count; ++i_mon) {
649 loadMonitorFlags.emplace_back(m_monitorInfo[i_mon].hasHisto);
650 }
651 }
652
653 // create workspace
654 if (useEventMon) {
655 // Use event monitors and create event workspace
656
657 // only used if using event monitors
658 EventWorkspace_sptr eventWS = EventWorkspace_sptr(new EventWorkspace());
659 eventWS->initialize(m_monitorInfo.size(), 1, 1);
660
661 // Set the units
662 eventWS->getAxis(0)->unit() = Mantid::Kernel::UnitFactory::Instance().create("TOF");
663 eventWS->setYUnit("Counts");
664 m_workspace = eventWS;
665 } else { // only other option is histograms
666 // Use histogram monitors and event monitors' histogram data.
667 // And thus create a Workspace2D.
668
669 m_workspace = API::WorkspaceFactory::Instance().create("Workspace2D", m_monitorInfo.size(), 2, 1);
670 }
671
672 return useEventMon;
673}

675void LoadNexusMonitors2::readEventMonitorEntry(Nexus::File &file, size_t ws_index) {
676 // setup local variables
677 EventWorkspace_sptr eventWS = std::dynamic_pointer_cast<EventWorkspace>(m_workspace);
678 std::string tof_units, event_time_zero_units;
679
680 // read in the data
681 auto event_index = Nexus::IOHelper::readNexusVector<uint64_t>(file, "event_index");
682
683 file.openData("event_time_offset"); // time of flight
684 MantidVec time_of_flight = Nexus::IOHelper::readNexusVector<double>(file);
685 file.getAttr("units", tof_units);
686 Kernel::Units::timeConversionVector(time_of_flight, tof_units, "microseconds");
687 file.closeData();
688
689 // warn the user if no events were found
690 if (time_of_flight.empty()) {
691 g_log.error() << "No events found in \"" << m_monitorInfo[ws_index].name << "\"\n";
692 return; // early
693 }
694
695 file.openData("event_time_zero"); // pulse time
696 MantidVec seconds = Nexus::IOHelper::readNexusVector<double>(file);
697 file.getAttr("units", event_time_zero_units);
698 Kernel::Units::timeConversionVector(seconds, event_time_zero_units, "seconds");
699 Mantid::Types::Core::DateAndTime pulsetime_offset;
700 {
701 std::string startTime;
702 file.getAttr("offset", startTime);
703 pulsetime_offset = createFromSanitizedISO8601(startTime);
704 }
705 file.closeData();
706
707 // load up the event list
708 DataObjects::EventList &event_list = eventWS->getSpectrum(ws_index);
709
710 Mantid::Types::Core::DateAndTime pulsetime;
711 Mantid::Types::Core::DateAndTime lastpulsetime(0);
712 std::size_t numEvents = time_of_flight.size();
713 bool pulsetimesincreasing = true;
714 size_t pulse_index(0);
715 size_t numPulses = seconds.size();
716 for (std::size_t j = 0; j < numEvents; ++j) {
717 while (!((j >= event_index[pulse_index]) && (j < event_index[pulse_index + 1]))) {
718 pulse_index += 1;
719 if (pulse_index > (numPulses + 1))
720 break;
721 }
722 if (pulse_index >= (numPulses))
723 pulse_index = numPulses - 1; // fix it
724 pulsetime = pulsetime_offset + seconds[pulse_index];
725 if (pulsetime < lastpulsetime)
726 pulsetimesincreasing = false;
727 lastpulsetime = pulsetime;
728 event_list.addEventQuickly(Types::Event::TofEvent(time_of_flight[j], pulsetime));
729 }
730 if (pulsetimesincreasing)
731 event_list.setSortOrder(DataObjects::PULSETIME_SORT);
732}
