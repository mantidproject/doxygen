34template <herr_t (*H5Xclose)(hid_t)> std::string readNXClass(Mantid::Nexus::UniqueID<H5Xclose> const &oid) {
35 std::string nxClass = UNKNOWN_CLASS;
36 if (H5Aexists(oid, Mantid::Nexus::GROUP_CLASS_SPEC.c_str()) > 0) {
37 Mantid::Nexus::UniqueID<&H5Aclose> attrID = H5Aopen(oid, Mantid::Nexus::GROUP_CLASS_SPEC.c_str(), H5P_DEFAULT);
38 if (attrID.isValid()) {
39 Mantid::Nexus::UniqueID<&H5Tclose> atype(H5Aget_type(attrID));
40 if (H5Tis_variable_str(atype)) {
41 // variable length string
42 char *rdata;
43 if (H5Aread(attrID, atype, &rdata) >= 0) {
44 nxClass = std::string(rdata);
45 }
46 // reclaim memory allocated for rdata by HDF5
47 H5free_memory(rdata);
48 } else {
49 // fixed length string
50 std::size_t size = H5Tget_size(atype);
51 nxClass.resize(size);
52 H5Aread(attrID, atype, nxClass.data());
53 }
54 }
55 }
56 return nxClass;
57}

68bool NexusDescriptorLazy::isEntry(std::string const &entryName) const {
69
70 std::map<std::string, std::string>::iterator it; // get it out of the lock scope
71 {
72 // wait for writes to end
73 std::shared_lock<std::shared_mutex> lock(m_readNexusMutex);
74 it = m_allEntries.find(entryName);
75 }
76 if (it != m_allEntries.end()) {
77 return it->second != NONEXISTENT;
78 } else {
79 // modifying m_allEntries, need write lock
80 std::lock_guard<std::shared_mutex> lock(m_readNexusMutex);
81
82 // see if something with the name exists
83 if (H5Oexists_by_name(m_fileID, entryName.c_str(), H5P_DEFAULT) <= 0) {
84 // register failure
85 m_allEntries[entryName] = NONEXISTENT;
86 return false;
87 }
88
89 // open the object to determine its type
90 UniqueID<&H5Oclose> entryID(H5Oopen(m_fileID, entryName.c_str(), H5P_DEFAULT));
91 if (entryID.isValid()) {
92 m_allEntries[entryName] = UNKNOWN_CLASS;
93 H5O_info_t oinfo;
94 H5Oget_info(entryID, &oinfo, H5O_INFO_BASIC);
95 if (oinfo.type == H5O_TYPE_DATASET) {
96 m_allEntries[entryName] = SCIENTIFIC_DATA_SET;
97 } else {
98 // read NX_class attribute
99 m_allEntries[entryName] = readNXClass(entryID);
100 }
101 return true;
102 } else {
103 // register failure
104 m_allEntries[entryName] = NONEXISTENT;
105 return false;
106 }
107 }

118bool NexusDescriptorLazy::classTypeExistsChild(const std::string &parentPath, const std::string &classType) const {
119 // if the parent doesn't exist, the child doesn't either
120 if (!this->isEntry(parentPath))
121 return false;
122
123 // wait for writes to end
124 std::shared_lock<std::shared_mutex> lock(m_readNexusMutex);
125
126 // linear search through all entries - stop at first match
127 const auto delimitedEntryName = parentPath + '/';
128 for (auto const &[name, cls] : m_allEntries) {
129 // match the class first since that limits the list more
130 if (cls == classType && name.starts_with(delimitedEntryName)) {
131 return true;
132 }
133 }
134 return false;
135}

137bool NexusDescriptorLazy::hasRootAttr(std::string const &name) const {
138 { // wait for writes to end
139 std::shared_lock<std::shared_mutex> lock(m_readNexusMutex);
140 if (m_rootAttrs.count(name) == 1) {
141 return true;
142 }
143 }
144 // check the file since it wasn't in the cache
145 {
146 // mutex has the wrong name, but it's what we have
147 std::lock_guard<std::shared_mutex> lock(m_readNexusMutex);
148
149 if (H5Aexists(m_fileID, name.c_str()) > 0) {
150 m_rootAttrs.emplace(name);
151 return true;
152 } else {
153 return false;

159std::string NexusDescriptorLazy::getStrData(std::string const &address) {
160 std::string strData;
161 if (isEntry(address, SCIENTIFIC_DATA_SET)) {
162 // open the data set and get its string data
163 // using H5Cpp interface because trying to read string data is an absolute nightmare with the C API
164 UniqueID<&H5Dclose> did(H5Dopen(m_fileID, address.c_str(), H5P_DEFAULT));
165 H5::DataSet dataset(did);
166 H5::DataType dtype = dataset.getDataType();
167 if (dtype.isVariableStr() || dtype.getClass() == H5T_STRING) {
168 dataset.read(strData, dtype, dataset.getSpace());
169 }
170 }
171 return strData;
172}

176void NexusDescriptorLazy::loadGroups(std::map<std::string, std::string> &allEntries, std::string const &address,
177 unsigned int depth, const unsigned int maxDepth) {
178 UniqueID<&H5Gclose> groupID(H5Gopen(m_fileID, address.c_str(), H5P_DEFAULT));
179 if (!groupID.isValid()) {
180 return;
181 }
182
183 // get NX_class attribute
184 allEntries[address] = readNXClass(groupID);
185
186 if (depth >= maxDepth)
187 return;
188
189 // iterate over members
190 hsize_t numObjs = 0;
191 H5Gget_num_objs(groupID.get(), &numObjs);
192 for (hsize_t i = 0; i < numObjs; i++) {
193 H5G_obj_t type = H5Gget_objtype_by_idx(groupID, i);
194 ssize_t name_len = H5Gget_objname_by_idx(groupID, i, nullptr, 0);
195 if (name_len <= 0)
196 continue;
197 std::string memberName(name_len, 'X'); // fill with X for obvious errors
198 H5Gget_objname_by_idx(groupID, i, memberName.data(), name_len + 1); // +1 for null terminator,
199 std::string memberAddress = address;
200 if (!memberAddress.ends_with("/"))
201 memberAddress += "/";
202 memberAddress += memberName;
203
204 if (type == H5G_GROUP) {
205 loadGroups(allEntries, memberAddress, depth + 1, maxDepth);
206 } else if (type == H5G_DATASET) {
207 allEntries[memberAddress] = SCIENTIFIC_DATA_SET;
208 }
209 }

212std::map<std::string, std::string> NexusDescriptorLazy::initAllEntries() {
213
214 H5Eset_auto(H5E_DEFAULT, nullptr, nullptr);
215
216 std::map<std::string, std::string> allEntries;
217
218 // if the file exists read it
219 if (std::filesystem::exists(m_filename)) {
220 // if the file exists but cannot be opened, throw invalid
221 // NOTE must be std::invalid_argument for expected errors to be raised in python API
222 if (!H5::H5File::isAccessible(m_filename, Mantid::Nexus::H5Util::defaultFileAcc())) {
223 throw std::invalid_argument("ERROR: NexusDescriptorLazy couldn't open hdf5 file " + m_filename + "\n");
224 } else {
225 m_fileID = H5Fopen(m_filename.c_str(), H5F_ACC_RDONLY, Mantid::Nexus::H5Util::defaultFileAcc().getId());
226 }
227 if (!m_fileID.isValid()) {
228 throw std::invalid_argument("ERROR: NexusDescriptorLazy couldn't open hdf5 file " + m_filename + "\n");
229 }
230
231 // get all top-level entries
232 unsigned int depth = 0;
233 loadGroups(allEntries, "/", depth, INIT_DEPTH);
234 // set the first entry name/type
235 if (allEntries.size() > 1) {
236 m_firstEntryNameType = *(++allEntries.begin());
237 m_firstEntryNameType.first = m_firstEntryNameType.first.substr(1); // remove leading /
238 } else {
239 m_firstEntryNameType = std::make_pair("", UNKNOWN_CLASS);
240 }
241
242 // for levels beyond 2, only load special entries
243 depth = INIT_DEPTH;
244 for (std::string const &specialAddress : SPECIAL_ADDRESS) {
245 if (allEntries.contains(specialAddress))
246 loadGroups(allEntries, specialAddress, depth, ENTRY_DEPTH);
247 }
248
249 // get instrument up to a depth of 5
250 depth = ENTRY_DEPTH;
251 for (std::string const &specialAddress : SPECIAL_ADDRESS) {
252 if (allEntries.contains(specialAddress)) {
253 std::string instrumentAddress = specialAddress + "/instrument";
254 if (allEntries.contains(instrumentAddress)) {
255 loadGroups(allEntries, instrumentAddress, depth, INSTR_DEPTH);
256 }
257 }
258 }
259 } else {
260 // if the file does not exist, then leave allEntries empty
261 }
262 // rely on move semantics for single return
263 return allEntries;
264}
