44SparseWorkspace::SparseWorkspace(const API::MatrixWorkspace &modelWS, const size_t wavelengthPoints, const size_t rows,
45 const size_t columns)
46 : Workspace2D() {
47 double minLat, maxLat, minLong, maxLong;
48 std::tie(minLat, maxLat, minLong, maxLong) = extremeAngles(modelWS);
49 m_gridDef = std::make_unique<Algorithms::DetectorGridDefinition>(minLat, maxLat, rows, minLong, maxLong, columns);
50 if ((rows < 3) || (columns < 3)) {
51 g_log.warning("Can't calculate errors on a sparse workspace with lat or "
52 "long dimension < 3");
53 }
54 const size_t numSpectra = rows * columns;
55 const auto h = modelHistogram(modelWS, wavelengthPoints);
56 initialize(numSpectra, h);
57
58 // Build a quite standard and somewhat complete instrument.
59 auto instrument = std::make_shared<Geometry::Instrument>("MC_simulation_instrument");
60 const auto refFrame = modelWS.getInstrument()->getReferenceFrame();
61
62 instrument->setReferenceFrame(std::make_shared<Geometry::ReferenceFrame>(*refFrame));
63 // The sparse instrument is build around origin.
64 constexpr Kernel::V3D samplePos{0.0, 0.0, 0.0};
65 auto sample = std::make_unique<Geometry::Component>("sample", instrument.get());
66 sample->setPos(samplePos);
67 instrument->add(sample.get());
68 instrument->markAsSamplePos(sample.release());
69 // Add source behind the sample.
70 const Kernel::V3D sourcePos = [&]() {
71 Kernel::V3D p;
72 p[refFrame->pointingAlongBeam()] = -2.0 * R;
73 return p;
74 }();
75 auto source = std::make_unique<Geometry::ObjComponent>("source", nullptr, instrument.get());
76 source->setPos(sourcePos);
77 instrument->add(source.get());
78 instrument->markAsSource(source.release());
79
80 auto detShape = makeCubeShape();
81 for (size_t col = 0; col < columns; ++col) {
82 const auto lon = m_gridDef->longitudeAt(col);
83 for (size_t row = 0; row < rows; ++row) {
84 const auto lat = m_gridDef->latitudeAt(row);
85 const size_t index = col * rows + row;
86 const auto detID = static_cast<int>(index);
87 std::ostringstream detName;
88 detName << "det-" << detID;
89 auto det = std::make_unique<Geometry::Detector>(detName.str(), detID, detShape, instrument.get());
90 const Kernel::V3D pos = [&]() {
91 Kernel::V3D p;
92 p[refFrame->pointingHorizontal()] = R * std::sin(lon) * std::cos(lat);
93 p[refFrame->pointingUp()] = R * std::sin(lat);
94 p[refFrame->pointingAlongBeam()] = R * std::cos(lon) * std::cos(lat);
95 return p;
96 }();
97 det->setPos(pos);
98 getSpectrum(index).setDetectorID(detID);
99 instrument->add(det.get());
100 instrument->markAsDetector(det.release());
101 }
102 }
103 setInstrument(instrument);
104
105 // Copy things needed for the simulation from the model workspace.
106 auto &paramMap = instrumentParameters();
107 auto parametrizedInstrument = getInstrument();
108 // Copy beam parameters.
109 const auto modelSource = modelWS.getInstrument()->getSource();
110 const auto parametrizedSource = parametrizedInstrument->getSource();
111 const auto beamShapeParam = modelSource->getParameterAsString("beam-shape");
112 if (!beamShapeParam.empty())
113 paramMap.add("string", parametrizedSource.get(), "beam-shape", beamShapeParam);
114 const auto beamWidthParam = modelSource->getNumberParameter("beam-width");
115 const auto beamHeightParam = modelSource->getNumberParameter("beam-height");
116 if (beamWidthParam.size() == 1 && beamHeightParam.size() == 1) {
117 paramMap.add("double", parametrizedSource.get(), "beam-width", beamWidthParam[0]);
118 paramMap.add("double", parametrizedSource.get(), "beam-height", beamHeightParam[0]);
119 }
120 const auto beamRadiusParam = modelSource->getNumberParameter("beam-radius");
121 if (beamRadiusParam.size() == 1) {
122 paramMap.add("double", parametrizedSource.get(), "beam-radius", beamRadiusParam[0]);
123 }
124 // Add information about EFixed in a proper place.
125 const auto eMode = modelWS.getEMode();
126 mutableRun().addProperty("deltaE-mode", Kernel::DeltaEMode::asString(eMode));
127 if (eMode == Kernel::DeltaEMode::Direct) {
128 mutableRun().addProperty("Ei", modelWS.getEFixed());
129 } else if (eMode == Kernel::DeltaEMode::Indirect) {
130 std::vector<detid_t> detIDs;
131 auto modelInstrument = modelWS.getInstrument();
132 for (size_t i = 0; i < modelWS.getNumberHistograms(); i++) {
133 const auto &spec = modelWS.getSpectrum(i);
134 const auto &specDetIDs = spec.getDetectorIDs();
135 if (!modelInstrument->isMonitor(specDetIDs))
136 detIDs.insert(detIDs.end(), specDetIDs.begin(), specDetIDs.end());
137 }
138 if (!constantIndirectEFixed(modelWS, detIDs)) {
139 throw std::runtime_error("Sparse instrument with variable EFixed not supported.");
140 }
141 const auto e = modelWS.getEFixed(detIDs[0]);
142 const auto &sparseDetIDs = detectorInfo().detectorIDs();
143 for (int sparseDetID : sparseDetIDs) {
144 setEFixed(sparseDetID, e);
145 }
146 }
147}

158std::tuple<double, double, double, double> SparseWorkspace::extremeAngles(const API::MatrixWorkspace &ws) {
159 const auto &spectrumInfo = ws.spectrumInfo();
160 double minLat = std::numeric_limits<double>::max();
161 double maxLat = std::numeric_limits<double>::lowest();
162 double minLong = std::numeric_limits<double>::max();
163 double maxLong = std::numeric_limits<double>::lowest();
164 for (size_t i = 0; i < ws.getNumberHistograms(); ++i) {
165 if (spectrumInfo.hasDetectors(i)) {
166 double lat, lon;
167 std::tie(lat, lon) = spectrumInfo.geographicalAngles(i);
168 if (lat < minLat) {
169 minLat = lat;
170 }
171 if (lat > maxLat) {
172 maxLat = lat;
173 }
174 if (lon < minLong) {
175 minLong = lon;
176 }
177 if (lon > maxLong) {
178 maxLong = lon;
179 }
180 }
181 }
182 return std::make_tuple(minLat, maxLat, minLong, maxLong);
183}

189std::tuple<double, double> SparseWorkspace::extremeWavelengths(const API::MatrixWorkspace &ws) {
190 double currentMin = std::numeric_limits<double>::max();
191 double currentMax = std::numeric_limits<double>::lowest();
192 for (size_t i = 0; i < ws.getNumberHistograms(); ++i) {
193 const auto h = ws.histogram(i);
194 const auto first = h.begin();
195 currentMin = std::min(first->center(), currentMin);
196 const auto last = std::prev(h.end());
197 currentMax = std::max(last->center(), currentMax);
198 }
199 return std::make_tuple(currentMin, currentMax);
200}

207Mantid::HistogramData::Histogram SparseWorkspace::modelHistogram(const API::MatrixWorkspace &modelWS,
208 const size_t wavelengthPoints) {
209 double minWavelength, maxWavelength;
210 std::tie(minWavelength, maxWavelength) = extremeWavelengths(modelWS);
211 HistogramData::Frequencies ys(wavelengthPoints, 0.0);
212 HistogramData::FrequencyVariances es(wavelengthPoints, 0.0);
213 HistogramData::Points ps(wavelengthPoints, 0.0);
214 HistogramData::Histogram h(ps, ys, es);
215 auto &xs = h.mutableX();
216 if (wavelengthPoints > 1) {
217 const double step = (maxWavelength - minWavelength) / static_cast<double>(wavelengthPoints - 1);
218 for (size_t i = 0; i < xs.size() - 1; ++i) {
219 xs[i] = minWavelength + step * static_cast<double>(i);
220 }
221 // Force last point as otherwise it might be slightly off due to
222 // small rounding errors in the calculation above.
223 xs.back() = maxWavelength;
224 } else {
225 xs.front() = (minWavelength + maxWavelength) / 2.0;
226 }
227 return h;
228}

278double SparseWorkspace::greatCircleDistance(const double lat1, const double long1, const double lat2,
279 const double long2) {
280 const double latD = std::sin((lat2 - lat1) / 2.0);
281 const double longD = std::sin((long2 - long1) / 2.0);
282 const double S = latD * latD + std::cos(lat1) * std::cos(lat2) * longD * longD;
283 return 2.0 * std::asin(std::sqrt(S));
284}

290std::array<double, 4> SparseWorkspace::inverseDistanceWeights(const std::array<double, 4> &distances) {
291 std::array<double, 4> weights;
292 for (size_t i = 0; i < weights.size(); ++i) {
293 if (distances[i] == 0.0) {
294 weights.fill(0.0);
295 weights[i] = 1.0;
296 return weights;
297 }
298 weights[i] = 1.0 / distances[i] / distances[i];
299 }
300 return weights;
301}

308HistogramData::HistogramY SparseWorkspace::secondDerivative(const std::array<size_t, 3> &indices,
309 const double distanceStep) const {
310 HistogramData::HistogramY avgSecondDeriv(blocksize());
311
312 HistogramData::HistogramY sumSecondDeriv(blocksize());
313 auto firstDerivB = (y(indices[2]) - y(indices[1])) / distanceStep;
314 auto firstDerivA = (y(indices[1]) - y(indices[0])) / distanceStep;
315 auto secondDerivA = (firstDerivB - firstDerivA) / distanceStep;
316
317 return secondDerivA;
318}

325HistogramData::Histogram SparseWorkspace::interpolateFromDetectorGrid(const double lat, const double lon) const {
326 const auto indices = m_gridDef->nearestNeighbourIndices(lat, lon);
327
328 auto h = histogram(0);
329
330 std::array<double, 4> distances;
331 for (size_t i = 0; i < 4; ++i) {
332 double detLat, detLong;
333 std::tie(detLat, detLong) = spectrumInfo().geographicalAngles(indices[i]);
334 distances[i] = greatCircleDistance(lat, lon, detLat, detLong);
335 }
336 const auto weights = inverseDistanceWeights(distances);
337 auto weightSum = weights[0];
338 h.mutableY() = weights[0] * y(indices[0]);
339 for (size_t i = 1; i < 4; ++i) {
340 weightSum += weights[i];
341 h.mutableY() += weights[i] * y(indices[i]);
342 }
343 h.mutableY() /= weightSum;
344 return h;
345}

371HistogramData::Histogram SparseWorkspace::bilinearInterpolateFromDetectorGrid(const double lat,
372 const double lon) const {
373
374 size_t nearestLatIndex, nearestLonIndex;
375 std::tie(nearestLatIndex, nearestLonIndex) = m_gridDef->getNearestVertex(lat, lon);
376
377 std::array<std::array<size_t, 2>, 2> detIndices;
378 for (size_t i = 0; i < 2; i++) {
379 for (size_t j = 0; j < 2; j++) {
380 detIndices[i][j] = m_gridDef->getDetectorIndex(nearestLatIndex + j, nearestLonIndex + i);
381 }
382 }
383
384 double latLow, longLow, latHigh, longHigh;
385 std::tie(latLow, longLow) = spectrumInfo().geographicalAngles(detIndices[0][0]);
386 std::tie(latHigh, longHigh) = spectrumInfo().geographicalAngles(detIndices[1][1]);
387
388 // interpolate across different longitudes
389 auto ylat1 = ((longHigh - lon) * y(detIndices[0][0]) + (lon - longLow) * y(detIndices[1][0])) / (longHigh - longLow);
390 auto errLat1 = esqrt(esq((longHigh - lon) * e(detIndices[0][0])) + esq((lon - longLow) * e(detIndices[1][0]))) /
391 (longHigh - longLow);
392
393 auto ylat2 = ((longHigh - lon) * y(detIndices[0][1]) + (lon - longLow) * y(detIndices[1][1])) / (longHigh - longLow);
394 auto errLat2 = esqrt(esq((longHigh - lon) * e(detIndices[0][1])) + esq((lon - longLow) * e(detIndices[1][1]))) /
395 (longHigh - longLow);
396
397 // interpolate across different latitudes
398 auto interpY = ((latHigh - lat) * ylat1 + (lat - latLow) * ylat2) / (latHigh - latLow);
399 auto errFromSourcePoints = esqrt(esq((latHigh - lat) * errLat1) + esq((lat - latLow) * errLat2)) / (latHigh - latLow);
400
401 // calculate interpolation errors if possible
402 HistogramData::HistogramE interpolationError(e(0).size(), 0);
403 if ((m_gridDef->numberColumns() > 2) && (m_gridDef->numberRows() > 2)) {
404 // step back a further row\col (if possible) to estimate the second
405 // derivative
406 auto nearestLatIndexSec = nearestLatIndex > 0 ? nearestLatIndex - 1 : 0;
407 auto nearestLonIndexSec = nearestLonIndex > 0 ? nearestLonIndex - 1 : 0;
408 std::array<size_t, 3> threeIndices;
409
410 // 2nd derivative in longitude
411 for (int i = 0; i < 3; i++) {
412 threeIndices[i] = m_gridDef->getDetectorIndex(nearestLatIndex, nearestLonIndexSec + i);
413 }
414 auto avgSecondDerivLong = secondDerivative(threeIndices, longHigh - longLow);
415
416 // 2nd derivative in latitude
417 for (int i = 0; i < 3; i++) {
418 threeIndices[i] = m_gridDef->getDetectorIndex(nearestLatIndexSec + i, nearestLonIndex);
419 }
420 auto avgSecondDerivLat = secondDerivative(threeIndices, latHigh - latLow);
421
422 // calculate the interpolation error according to a Taylor expansion from
423 // the low lat and low long points. Doesn't make any difference if do this
424 // in angle or distance because scaling factor (~ radius) cancels out
425 HistogramData::HistogramY interpolationErrorAsYData =
426 0.5 * (lon - longLow) * (longHigh - lon) * avgSecondDerivLong +
427 0.5 * (lat - latLow) * (latHigh - lat) * avgSecondDerivLat;
428 // convert from HistogramY to HistogramE
429 interpolationError = interpolationErrorAsYData.rawData();
430 }
431
432 auto h = histogram(0);
433 h.mutableY() = interpY;
434 // combine the (independent) errors
435 // Note - interpolationError may be negative if the 2nd derivative is negative
436 // so the following line also has useful side effect of taking abs value
437 h.mutableE() = esqrt(esq(interpolationError) + esq(errFromSourcePoints));
438 return h;
439}
