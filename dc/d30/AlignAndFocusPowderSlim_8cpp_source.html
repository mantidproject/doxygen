78double getFocussedPostion(const detid_t detid, const std::vector<double> &difc_focus) {
79 // grouping taken from the IDF for VULCAN
80 if (detid < 0) {
81 throw std::runtime_error("detid < 0 is not supported");
82 } else if (detid < 100000) { // bank1 0-99095
83 return difc_focus[0];
84 } else if (detid < 200000) { // bank2 100000-199095
85 return difc_focus[1];
86 } else if (detid < 300000) { // bank3 200000-289095
87 return difc_focus[2];
88 } else if (detid < 400000) { // bank4 300000-389095
89 return difc_focus[3];
90 } else if (detid < 500000) { // bank5 400000-440095
91 return difc_focus[4];
92 } else if (detid < 600000) { // bank6 500000-554095
93 return difc_focus[5];
94 } else {
95 throw std::runtime_error("detid > 600000 is not supported");
96 }
97}

99std::vector<double> calculate_difc_focused(const double l1, const std::vector<double> &l2s,
100 const std::vector<double> &polars) {
101 constexpr double deg2rad = std::numbers::pi_v<double> / 180.;
102
103 std::vector<double> difc;
104
105 std::transform(l2s.cbegin(), l2s.cend(), polars.cbegin(), std::back_inserter(difc),
106 [l1, deg2rad](const auto &l2, const auto &polar) {
107 return 1. / Kernel::Units::tofToDSpacingFactor(l1, l2, deg2rad * polar, 0.);
108 });
109
110 return difc;
111}

223std::map<std::string, std::string> AlignAndFocusPowderSlim::validateInputs() {
224 std::map<std::string, std::string> errors;
225
226 // make sure that data is read in larger chunks than the events processed in a single thread
227 const int disk_chunk = getProperty(PropertyNames::READ_SIZE_FROM_DISK);
228 const int grainsize_events = getProperty(PropertyNames::EVENTS_PER_THREAD);
229 if (disk_chunk < grainsize_events) {
230 const std::string msg(PropertyNames::READ_SIZE_FROM_DISK + " must be larger than " +
231 PropertyNames::EVENTS_PER_THREAD);
232 errors[PropertyNames::READ_SIZE_FROM_DISK] = msg;
233 errors[PropertyNames::EVENTS_PER_THREAD] = msg;
234 }
235
236 // validate binning information is consistent with each other
237 const std::vector<double> xmins = getProperty(PropertyNames::X_MIN);
238 const std::vector<double> xmaxs = getProperty(PropertyNames::X_MAX);
239 const std::vector<double> deltas = getProperty(PropertyNames::X_DELTA);
240
241 const auto numMin = xmins.size();
242 const auto numMax = xmaxs.size();
243 const auto numDelta = deltas.size();
244
245 if (std::any_of(deltas.cbegin(), deltas.cend(), [](double d) { return !std::isfinite(d) || d == 0; }))
246 errors[PropertyNames::X_DELTA] = "All must be nonzero";
247 else if (!(numDelta == 1 || numDelta == NUM_HIST))
248 errors[PropertyNames::X_DELTA] = "Must have 1 or 6 values";
249
250 if (!(numMin == 1 || numMin == NUM_HIST))
251 errors[PropertyNames::X_MIN] = "Must have 1 or 6 values";
252
253 if (!(numMax == 1 || numMax == NUM_HIST))
254 errors[PropertyNames::X_MAX] = "Must have 1 or 6 values";
255
256 // only specify allow or block list for logs
257 if ((!isDefault(PropertyNames::ALLOW_LOGS)) && (!isDefault(PropertyNames::BLOCK_LOGS))) {
258 errors[PropertyNames::ALLOW_LOGS] = "Cannot specify both allow and block lists";
259 errors[PropertyNames::BLOCK_LOGS] = "Cannot specify both allow and block lists";
260 }
261
262 return errors;
263}

268void AlignAndFocusPowderSlim::exec() {
269 // create a histogram workspace
270
271 // These give the limits in each file as to which events we actually load (when filtering by time).
272 loadStart.resize(1, 0);
273 loadSize.resize(1, 0);
274
275 this->progress(.0, "Create output workspace");
276
277 MatrixWorkspace_sptr wksp = createOutputWorkspace();
278
279 const std::string filename = getPropertyValue(PropertyNames::FILENAME);
280 { // TODO TEMPORARY - this algorithm is hard coded for VULCAN
281 // it needs to be made more generic
282 if (filename.find("VULCAN") == std::string::npos) {
283 throw std::runtime_error("File does not appear to be for VULCAN");
284 }
285 }
286 const Nexus::NexusDescriptor descriptor(filename);
287
288 // instrument is needed for lots of things
289 const std::string ENTRY_TOP_LEVEL("entry");
290 LoadEventNexus::loadInstrument<MatrixWorkspace_sptr>(filename, wksp, ENTRY_TOP_LEVEL, this, &descriptor);
291
292 // TODO parameters should be input information
293 const double l1{43.755};
294 const std::vector<double> polars{90, 90, 120, 150, 157, 65.5}; // two-theta
295 const std::vector<double> azimuthals{180, 0, 0, 0, 0, 0}; // angle from positive x-axis
296 const std::vector<double> l2s{2.296, 2.296, 2.070, 2.070, 2.070, 2.530};
297 const std::vector<specnum_t> specids;
298 const auto difc_focused = calculate_difc_focused(l1, l2s, polars);
299
300 // create values for focusing time-of-flight
301 this->progress(.05, "Creating calibration constants");
302 const std::string cal_filename = getPropertyValue(PropertyNames::CAL_FILE);
303 if (!cal_filename.empty()) {
304 this->loadCalFile(wksp, cal_filename, difc_focused);
305 } else {
306 this->initCalibrationConstants(wksp, difc_focused);
307 }
308
309 // calculate correction for tof of the neutron at the sample position
310 if (this->getProperty(PropertyNames::FULL_TIME)) {
311 this->initScaleAtSample(wksp);
312 }
313
314 // set the instrument
315 this->progress(.07, "Set instrument geometry");
316 wksp = this->editInstrumentGeometry(wksp, l1, polars, specids, l2s, azimuthals);
317
318 // convert to TOF if not already
319 this->progress(.1, "Convert bins to TOF");
320 wksp = this->convertToTOF(wksp);
321
322 // TODO parameters should be read in from a file
323 std::map<size_t, std::vector<detid_t>> grouping;
324 constexpr detid_t NUM_DETS_PER_BANK{100000};
325 for (size_t outputSpecNum : std::views::iota(0, 6)) {
326 grouping[outputSpecNum] = std::vector<detid_t>(NUM_DETS_PER_BANK);
327 std::iota(grouping[outputSpecNum].begin(), grouping[outputSpecNum].end(),
328 static_cast<detid_t>(NUM_DETS_PER_BANK * outputSpecNum));
329 }
330
331 // load run metadata
332 this->progress(.11, "Loading metadata");
333 // prog->doReport("Loading metadata"); TODO add progress bar stuff
334 try {
335 LoadEventNexus::loadEntryMetadata(filename, wksp, ENTRY_TOP_LEVEL, descriptor);
336 } catch (std::exception &e) {
337 g_log.warning() << "Error while loading meta data: " << e.what() << '\n';
338 }
339
340 // load logs
341 this->progress(.12, "Loading logs");
342 auto periodLog = std::make_unique<const TimeSeriesProperty<int>>("period_log"); // not used
343 const std::vector<std::string> &allow_logs = getProperty(PropertyNames::ALLOW_LOGS);
344 const std::vector<std::string> &block_logs = getProperty(PropertyNames::BLOCK_LOGS);
345 int nPeriods{1};
346 LoadEventNexus::runLoadNexusLogs<MatrixWorkspace_sptr>(filename, wksp, *this, false, nPeriods, periodLog, allow_logs,
347 block_logs);
348
349 const auto timeSplitter = this->timeSplitterFromSplitterWorkspace(wksp->run().startTime());
350 const auto filterROI = this->getFilterROI(wksp);
351 // determine the pulse indices from the time and splitter workspace
352 this->progress(.15, "Determining pulse indices");
353
354 // Now we want to go through all the bankN_event entries
355 const std::map<std::string, std::set<std::string>> &allEntries = descriptor.getAllEntries();
356 auto itClassEntries = allEntries.find("NXevent_data");
357
358 // load the events
359 H5::H5File h5file(filename, H5F_ACC_RDONLY, Nexus::H5Util::defaultFileAcc());
360 if (itClassEntries == allEntries.end()) {
361 h5file.close();
362 throw std::runtime_error("No NXevent_data entries found in file");
363 }
364
365 this->progress(.17, "Reading events");
366
367 // hard coded for VULCAN 6 banks
368 std::vector<std::string> bankEntryNames;
369 std::size_t num_banks_to_read;
370 int outputSpecNum = getProperty(PropertyNames::OUTPUT_SPEC_NUM);
371 if (outputSpecNum == EMPTY_INT()) {
372 for (size_t i = 1; i <= NUM_HIST; ++i) {
373 bankEntryNames.push_back("bank" + std::to_string(i) + "_events");
374 }
375 num_banks_to_read = NUM_HIST;
376 } else {
377 // fill this vector with blanks -- this is for the ProcessBankTask to correctly access it
378 for (size_t i = 1; i <= NUM_HIST; ++i) {
379 bankEntryNames.push_back("");
380 }
381 // the desired bank gets the correct entry name
382 bankEntryNames[outputSpecNum - 1] = "bank" + std::to_string(outputSpecNum) + "_events";
383 num_banks_to_read = 1;
384 }
385
386 // threaded processing of the banks
387 const int DISK_CHUNK = getProperty(PropertyNames::READ_SIZE_FROM_DISK);
388 const int GRAINSIZE_EVENTS = getProperty(PropertyNames::EVENTS_PER_THREAD);
389 g_log.debug() << (DISK_CHUNK / GRAINSIZE_EVENTS) << " threads per chunk\n";
390
391 if (timeSplitter.empty()) {
392 const auto pulse_indices = this->determinePulseIndices(wksp, filterROI);
393
394 auto progress = std::make_shared<API::Progress>(this, .17, .9, num_banks_to_read);
395 ProcessBankTask task(bankEntryNames, h5file, is_time_filtered, wksp, m_calibration, m_scale_at_sample, grouping,
396 m_masked, static_cast<size_t>(DISK_CHUNK), static_cast<size_t>(GRAINSIZE_EVENTS),
397 pulse_indices, progress);
398 // generate threads only if appropriate
399 if (num_banks_to_read > 1) {
400 tbb::parallel_for(tbb::blocked_range<size_t>(0, num_banks_to_read), task);
401 } else {
402 // a "range" of 1; note -1 to match 0-indexed array with 1-indexed bank labels
403 task(tbb::blocked_range<size_t>(outputSpecNum - 1, outputSpecNum));
404 }
405
406 // close the file so child algorithms can do their thing
407 h5file.close();
408
409 // update the run TimeROI and remove log data outside the time ROI
410 wksp->mutableRun().setTimeROI(filterROI);
411 wksp->mutableRun().removeDataOutsideTimeROI();
412
413 setProperty(PropertyNames::OUTPUT_WKSP, std::move(wksp));
414 } else {
415 std::string ws_basename = this->getPropertyValue(PropertyNames::OUTPUT_WKSP);
416 std::vector<std::string> wsNames;
417 std::vector<int> workspaceIndices;
418 std::vector<MatrixWorkspace_sptr> workspaces;
419 for (const int &splitter_target : timeSplitter.outputWorkspaceIndices()) {
420 std::string ws_name = ws_basename + "_" + timeSplitter.getWorkspaceIndexName(splitter_target);
421 wsNames.push_back(std::move(ws_name));
422 workspaceIndices.push_back(splitter_target);
423 workspaces.emplace_back(wksp->clone());
424 }
425
426 auto progress = std::make_shared<API::Progress>(this, .17, .9, num_banks_to_read * workspaceIndices.size());
427 if (this->getProperty(PropertyNames::FULL_TIME)) {
428 g_log.information() << "Using ProcessBankSplitFullTimeTask for splitter processing\n";
429
430 // Get the combined time ROI for all targets so we only load necessary events.
431 // Need to offset the start time to account for tof's greater than pulsetime. 66.6ms is 4 pulses.
432 auto combined_time_roi = timeSplitter.combinedTimeROI(PULSETIME_OFFSET);
433 if (!filterROI.useAll()) {
434 combined_time_roi.update_intersection(filterROI);
435 }
436
437 const auto pulse_indices = this->determinePulseIndices(wksp, combined_time_roi);
438
439 const auto &splitterMap = timeSplitter.getSplittersMap();
440
441 ProcessBankSplitFullTimeTask task(bankEntryNames, h5file, is_time_filtered, workspaceIndices, workspaces,
442 m_calibration, m_scale_at_sample, grouping, m_masked,
443 static_cast<size_t>(DISK_CHUNK), static_cast<size_t>(GRAINSIZE_EVENTS),
444 pulse_indices, splitterMap, progress);
445
446 // generate threads only if appropriate
447 if (num_banks_to_read > 1) {
448 tbb::parallel_for(tbb::blocked_range<size_t>(0, num_banks_to_read), task);
449 } else {
450 // a "range" of 1; note -1 to match 0-indexed array with 1-indexed bank labels
451 task(tbb::blocked_range<size_t>(outputSpecNum - 1, outputSpecNum));
452 }
453
454 } else if (this->getProperty(PropertyNames::PROCESS_BANK_SPLIT_TASK)) {
455 g_log.information() << "Using ProcessBankSplitTask for splitter processing\n";
456 // determine the pulse indices from the time and splitter workspace
457 const auto target_to_pulse_indices = this->determinePulseIndicesTargets(wksp, filterROI, timeSplitter);
458
459 ProcessBankSplitTask task(bankEntryNames, h5file, true, workspaceIndices, workspaces, m_calibration,
460 m_scale_at_sample, grouping, m_masked, static_cast<size_t>(DISK_CHUNK),
461 static_cast<size_t>(GRAINSIZE_EVENTS), target_to_pulse_indices, progress);
462 // generate threads only if appropriate
463 if (num_banks_to_read > 1) {
464 tbb::parallel_for(tbb::blocked_range<size_t>(0, num_banks_to_read), task);
465 } else {
466 // a "range" of 1; note -1 to match 0-indexed array with 1-indexed bank labels
467 task(tbb::blocked_range<size_t>(outputSpecNum - 1, outputSpecNum));
468 }
469 } else {
470 g_log.information() << "Using ProcessBankTask for splitter processing\n";
471 // loop over the targets in the splitter workspace, each target gets its own output workspace
472 tbb::parallel_for(
473 tbb::blocked_range<size_t>(0, workspaceIndices.size()),
474 [&](const tbb::blocked_range<size_t> &target_indices) {
475 for (size_t target_index = target_indices.begin(); target_index != target_indices.end(); ++target_index) {
476 const int splitter_target = workspaceIndices[target_index];
477
478 auto splitter_roi = timeSplitter.getTimeROI(splitter_target);
479 // copy the roi so we can modify it just for this target
480 auto target_roi = filterROI;
481 if (target_roi.useAll())
482 target_roi = std::move(splitter_roi); // use the splitter ROI if no time filtering is specified
483 else if (!splitter_roi.useAll())
484 target_roi.update_intersection(splitter_roi); // otherwise intersect with the splitter ROI
485
486 // clone wksp for this target
487 MatrixWorkspace_sptr target_wksp = workspaces[target_index];
488
489 const auto pulse_indices = this->determinePulseIndices(target_wksp, target_roi);
490
491 ProcessBankTask task(bankEntryNames, h5file, is_time_filtered, target_wksp, m_calibration,
492 m_scale_at_sample, grouping, m_masked, static_cast<size_t>(DISK_CHUNK),
493 static_cast<size_t>(GRAINSIZE_EVENTS), pulse_indices, progress);
494 // generate threads only if appropriate
495 if (num_banks_to_read > 1) {
496 tbb::parallel_for(tbb::blocked_range<size_t>(0, num_banks_to_read), task);
497 } else {
498 // a "range" of 1; note -1 to match 0-indexed array with 1-indexed bank labels
499 task(tbb::blocked_range<size_t>(outputSpecNum - 1, outputSpecNum));
500 }
501 }
502 });
503 }
504
505 // close the file so child algorithms can do their thing
506 h5file.close();

532 }
533

542 const BINMODE binmode = getPropertyValue(PropertyNames::BINMODE);
543 const bool linearBins = bool(binmode == BinningMode::LINEAR);
544 const std::string binUnits = getPropertyValue(PropertyNames::BIN_UNITS);

584 Kernel::VectorHelper::createAxisFromRebinParams(params, XValues_new.mutableRawData(), resize_xnew,
585 full_bins_only);
586 }
587 HistogramData::Histogram hist(XValues_new, HistogramData::Counts(XValues_new.size() - 1, 0.0));
588 wksp->setHistogram(i, hist);
589 }
590 }

592 wksp->getAxis(0)->setUnit(binUnits);

708Kernel::TimeROI AlignAndFocusPowderSlim::getFilterROI(const API::MatrixWorkspace_sptr &wksp) {
709 Kernel::TimeROI roi;
710 const auto startOfRun = wksp->run().startTime();
711
712 // filter by time
713 double filter_time_start_sec = getProperty(PropertyNames::FILTER_TIMESTART);
714 double filter_time_stop_sec = getProperty(PropertyNames::FILTER_TIMESTOP);
715 if (filter_time_start_sec != EMPTY_DBL() || filter_time_stop_sec != EMPTY_DBL()) {
716 this->progress(.15, "Creating time filtering");
717 g_log.information() << "Filtering pulses from " << filter_time_start_sec << " to " << filter_time_stop_sec << "s\n";
718
719 try {
720 roi.addROI(startOfRun + (filter_time_start_sec == EMPTY_DBL() ? 0.0 : filter_time_start_sec),
721 startOfRun + filter_time_stop_sec); // start and stop times in seconds
722 } catch (const std::runtime_error &e) {
723 throw std::invalid_argument("Invalid time range for filtering: " + std::string(e.what()));
724 }
725 }
726
727 // filter bad pulses
728 if (getProperty(PropertyNames::FILTER_BAD_PULSES)) {
729 this->progress(.16, "Filtering bad pulses");
730
731 // get limits from proton_charge
732 const auto [min_pcharge, max_pcharge, mean] =
733 wksp->run().getBadPulseRange(LOG_CHARGE_NAME, getProperty(PropertyNames::FILTER_BAD_PULSES_LOWER_CUTOFF));
734 g_log.information() << "Filtering bad pulses; pcharge outside of " << min_pcharge << " to " << max_pcharge << '\n';
735
736 const auto run_start = wksp->getFirstPulseTime();
737 const auto run_stop = wksp->getLastPulseTime();
738
739 const auto log = dynamic_cast<const TimeSeriesProperty<double> *>(wksp->run().getLogData(LOG_CHARGE_NAME));
740 if (log) {
741 // need to have centre=true for proton_charge
742 roi = log->makeFilterByValue(min_pcharge, max_pcharge, true, Mantid::Kernel::TimeInterval(run_start, run_stop),
743 0.0, true, &roi);
744 }
745 }
746 return roi;
747}

790std::vector<std::pair<int, PulseROI>>
791AlignAndFocusPowderSlim::determinePulseIndicesTargets(const API::MatrixWorkspace_sptr &wksp, const TimeROI &filterROI,
792 const TimeSplitter &timeSplitter) {
793 // get pulse times from frequency log on workspace
794 const auto frequency_log = dynamic_cast<const TimeSeriesProperty<double> *>(wksp->run().getProperty("frequency"));
795 if (!frequency_log) {
796 throw std::runtime_error("Frequency log not found in workspace run");
797 }
798 const auto pulse_times =
799 std::make_unique<std::vector<Mantid::Types::Core::DateAndTime>>(frequency_log->timesAsVector());
800
801 std::vector<PulseROI> pulse_indices;
802 if (filterROI.useAll()) {
803 pulse_indices.emplace_back(0, std::numeric_limits<size_t>::max());
804 } else {
805 pulse_indices = filterROI.calculate_indices(*pulse_times);
806 if (pulse_indices.empty())
807 throw std::invalid_argument("No valid pulse time indices found for filtering");
808 }
809
810 const auto target_to_pulse_indices = timeSplitter.calculate_target_indices(*pulse_times);
811
812 // calculate intersection of target pulse indices and time filter pulse indices (removes pulses outside filterROI)
813 std::vector<std::pair<int, PulseROI>> intersected_target_pulse_indices;
814 auto pulse_it = pulse_indices.cbegin();
815 for (const auto &target_pair : target_to_pulse_indices) {
816 // move pulse_it to the first pulse that could overlap
817 while (pulse_it != pulse_indices.cend() && pulse_it->second <= target_pair.second.first) {
818 ++pulse_it;
819 }
820 // check for overlaps
821 auto check_it = pulse_it;
822 while (check_it != pulse_indices.cend() && check_it->first < target_pair.second.second) {
823 // there is an overlap
824 size_t start_index = std::max(check_it->first, target_pair.second.first);
825 size_t stop_index = std::min(check_it->second, target_pair.second.second);
826 if (start_index < stop_index) {
827 intersected_target_pulse_indices.emplace_back(target_pair.first, PulseROI(start_index, stop_index));
828 }
829 ++check_it;
830 }
831 }
832
833 return intersected_target_pulse_indices;
834}
