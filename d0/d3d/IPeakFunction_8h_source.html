23 size_t maxParam(size_t iY) {
24 double max = -DBL_MAX;
25 size_t maxIndex = 0;
26 for (size_t i = 0; i < m_p; ++i) {
27 double current = get(iY, i);

29 maxIndex = i;
30 max = current;
31 }
32 }
33
34 return maxIndex;
35 }
