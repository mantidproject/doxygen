53void ProcessBankSplitFullTimeTask::operator()(const tbb::blocked_range<size_t> &range) const {
54 auto entry = m_h5file.openGroup("entry"); // type=NXentry
55 for (size_t wksp_index = range.begin(); wksp_index < range.end(); ++wksp_index) {
56 const auto &bankName = m_bankEntries[wksp_index];
57 // empty bank names indicate spectra to skip; control should never get here, but just in case
58 if (bankName.empty()) {
59 continue;
60 }
61 Kernel::Timer timer;
62 g_log.debug() << bankName << " start" << std::endl;
63
64 // open the bank
65 auto event_group = entry.openGroup(bankName); // type=NXevent_data
66
67 // skip empty dataset
68 auto tof_SDS = event_group.openDataSet(NxsFieldNames::TIME_OF_FLIGHT);
69 const int64_t total_events = static_cast<size_t>(tof_SDS.getSpace().getSelectNpoints());
70 if (total_events == 0) {
71 m_progress->report();
72 continue;
73 }
74
75 std::unique_ptr<std::vector<uint64_t>> event_index = std::make_unique<std::vector<uint64_t>>();
76 auto eventRanges = m_loader->getEventIndexRanges(event_group, total_events, &event_index);
77
78 // Get all spectra for this bank.
79 // Create temporary y arrays for each workspace.
80 std::vector<API::ISpectrum *> spectra;
81 std::vector<std::vector<uint32_t>> y_temps;
82 for (const auto &wksp : m_wksps) {
83 spectra.push_back(&wksp->getSpectrum(wksp_index));
84 y_temps.emplace_back(spectra.back()->dataY().size());
85 }
86
87 // get handle to the data
88 auto detID_SDS = event_group.openDataSet(NxsFieldNames::DETID);
89 std::string tof_unit;
90 Nexus::H5Util::readStringAttribute(tof_SDS, "units", tof_unit);
91 const double time_conversion = Kernel::Units::timeConversionValue(tof_unit, MICROSEC);
92
93 // now the calibration for the output group can be created
94 // which detectors go into the current group - assumes ouput spectrum number is one more than workspace index
95 const auto calibration = m_calibFactory.getCalibration(time_conversion, wksp_index);
96
97 const auto frequency_log =
98 dynamic_cast<const Kernel::TimeSeriesProperty<double> *>(m_wksps.at(0)->run().getProperty("frequency"));
99 if (!frequency_log) {
100 throw std::runtime_error("Frequency log not found");
101 }
102 const auto pulse_times =
103 std::make_unique<std::vector<Mantid::Types::Core::DateAndTime>>(frequency_log->timesAsVector());
104
105 // declare arrays once so memory can be reused
106 auto event_detid = std::make_unique<std::vector<uint32_t>>(); // uint32 for ORNL nexus file
107 auto event_time_of_flight = std::make_unique<std::vector<float>>(); // float for ORNL nexus files
108 auto pulse_times_idx = std::make_unique<std::vector<size_t>>(); // index into pulse_times for every event
109
110 // read parts of the bank at a time until all events are processed
111 while (!eventRanges.empty()) {
112 // Create offsets and slab sizes for the next chunk of events.
113 // This will read at most m_events_per_chunk events from the file
114 // and will split the ranges if necessary for the next iteration.
115 std::vector<size_t> offsets;
116 std::vector<size_t> slabsizes;
117
118 size_t total_events_to_read = 0;
119 // Process the event ranges until we reach the desired number of events to read or run out of ranges
120 while (!eventRanges.empty() && total_events_to_read < m_events_per_chunk) {
121 // Get the next event range from the stack
122 auto eventRange = eventRanges.top();
123 eventRanges.pop();
124
125 size_t range_size = eventRange.second - eventRange.first;
126 size_t remaining_chunk = m_events_per_chunk - total_events_to_read;
127
128 // If the range size is larger than the remaining chunk, we need to split it
129 if (range_size > remaining_chunk) {
130 // Split the range: process only part of it now, push the rest back for later
131 offsets.push_back(eventRange.first);
132 slabsizes.push_back(remaining_chunk);
133 total_events_to_read += remaining_chunk;
134 // Push the remainder of the range back to the front for next iteration
135 eventRanges.emplace(eventRange.first + remaining_chunk, eventRange.second);
136 break;
137 } else {
138 offsets.push_back(eventRange.first);
139 slabsizes.push_back(range_size);
140 total_events_to_read += range_size;
141 // Continue to next range
142 }
143 }
144
145 // log the event ranges being processed
146 std::ostringstream oss;
147 oss << "Processing " << bankName << " with " << total_events_to_read << " events in the ranges: ";
148 for (size_t i = 0; i < offsets.size(); ++i) {
149 oss << "[" << offsets[i] << ", " << (offsets[i] + slabsizes[i]) << "), ";
150 }
151 oss << "\n";
152 g_log.debug() << oss.str();
153
154 // load detid and tof at the same time
155 tbb::parallel_invoke(
156 [&] { // load detid
157 m_loader->loadData(detID_SDS, event_detid, offsets, slabsizes);
158 },
159 [&] { // load time-of-flight
160 m_loader->loadData(tof_SDS, event_time_of_flight, offsets, slabsizes);
161 });
162
163 pulse_times_idx->resize(total_events_to_read);
164 // get the pulsetime of every event, event_index maps the first event of each pulse
165 size_t pos = 0;
166 auto event_index_it = event_index->cbegin();
167 for (size_t i = 0; i < offsets.size(); ++i) {
168 const size_t off = offsets[i];
169 const size_t slab = slabsizes[i];
170 for (size_t j = 0; j < slab; ++j) {
171 const uint64_t global_idx = static_cast<uint64_t>(off + j);
172 // find pulse: event_index holds the first event index of each pulse
173 while (event_index_it != event_index->cend() && *event_index_it <= global_idx) {
174 ++event_index_it;
175 }
176 size_t pulse_idx = 0;
177 if (event_index_it != event_index->cbegin()) {
178 pulse_idx = static_cast<size_t>(std::distance(event_index->cbegin(), event_index_it) - 1);
179 }
180 (*pulse_times_idx)[pos++] = pulse_idx;
181 }
182 }
183
184 // loop over targets
185 tbb::parallel_for(
186 tbb::blocked_range<size_t>(0, m_workspaceIndices.size()), [&](const tbb::blocked_range<size_t> &r) {
187 for (size_t idx = r.begin(); idx != r.end(); ++idx) {
188 int i = m_workspaceIndices[idx];
189
190 // Precompute indices for this target
191 std::vector<size_t> indices;
192 auto splitter_it = m_splitterMap.cbegin();
193 for (size_t k = 0; k < event_detid->size(); ++k) {
194 // Calculate the full time at sample: full_time = pulse_time + (tof * correctionFactor), where
195 // correctionFactor is either scale_at_sample[detid] or 1.0
196 const double correctionFactor =
197 calibration.value_scale_at_sample(static_cast<detid_t>((*event_detid)[k]));
198
199 const auto tof_in_nanoseconds =
200 static_cast<int64_t>(static_cast<double>((*event_time_of_flight)[k]) * correctionFactor);
201 const auto pulsetime = (*pulse_times)[(*pulse_times_idx)[k]];
202 const Mantid::Types::Core::DateAndTime full_time = pulsetime + tof_in_nanoseconds;
203 // Linear search for pulsetime in splitter map, assume pulsetime and splitter map are both sorted. This
204 // is the starting point for the full_time search so we need to subtract some time (66.6ms) to ensure we
205 // don't skip it when adding the tof.
206 // Advance splitter_it until it points to the first element greater than (pulsetime - offset).
207 // This gives us a reasonable starting point for the full_time search.
208 while (splitter_it != m_splitterMap.end() &&
209 splitter_it->first <= pulsetime - static_cast<int64_t>(PULSETIME_OFFSET)) {
210 ++splitter_it;
211 }
212
213 // Now, starting at splitter_it, find full_time (TOFs will be unsorted)
214 // Advance until we find the first element > full_time, then step back to get the greatest key <=
215 // full_time.
216 auto full_time_it = splitter_it;
217 while (full_time_it != m_splitterMap.end() && full_time_it->first <= full_time) {
218 ++full_time_it;
219 }
220
221 // If there is no element <= full_time then full_time_it will be begin(); otherwise step back to the
222 // element that is <= full_time.
223 if (full_time_it == m_splitterMap.begin()) {
224 // no splitter entry <= full_time; skip this event
225 } else {
226 --full_time_it;
227 if (full_time_it != m_splitterMap.end() && full_time_it->second == i) {
228 indices.push_back(k);
229 }
230 }
231 }
232
233 auto event_id_view_for_target =
234 indices | std::views::transform([&event_detid](const auto &k) { return (*event_detid)[k]; });
235 auto event_tof_view_for_target = indices | std::views::transform([&event_time_of_flight](const auto &k) {
236 return (*event_time_of_flight)[k];
237 });
238
239 ProcessEventsTask task(&event_id_view_for_target, &event_tof_view_for_target, &calibration,
240 &spectra[idx]->readX());
241
242 const tbb::blocked_range<size_t> range_info(0, indices.size(), m_grainsize_event);
243 tbb::parallel_reduce(range_info, task);
244
245 std::transform(y_temps[idx].begin(), y_temps[idx].end(), task.y_temp.begin(), y_temps[idx].begin(),
246 std::plus<uint32_t>());
247 }
248 });
249 }
250
251 // copy the data out into the correct spectrum and calculate errors
252 tbb::parallel_for(size_t(0), m_wksps.size(), [&](size_t i) {
253 auto &y_values = spectra[i]->dataY();
254 std::copy(y_temps[i].cbegin(), y_temps[i].cend(), y_values.begin());
255 auto &e_values = spectra[i]->dataE();
256 std::transform(y_temps[i].cbegin(), y_temps[i].cend(), e_values.begin(),
257 [](uint32_t y) { return std::sqrt(static_cast<double>(y)); });
258 });
259
260 g_log.debug() << bankName << " stop" << timer << std::endl;
261 m_progress->report();
262 }
263}
