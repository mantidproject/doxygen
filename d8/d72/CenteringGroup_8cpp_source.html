42std::vector<SymmetryOperation>
43CenteringGroupCreatorImpl::getSymmetryOperations(CenteringGroup::CenteringType centeringType) const {
44 switch (centeringType) {
45 case CenteringGroup::P:
46 return getPrimitive();
47 case CenteringGroup::I:
48 return getBodyCentered();
49 case CenteringGroup::A:
50 return getACentered();
51 case CenteringGroup::B:
52 return getBCentered();
53 case CenteringGroup::C:
54 return getCCentered();
55 case CenteringGroup::F:
56 return getFCentered();
57 case CenteringGroup::Robv:
58 return getRobvCentered();
59 case CenteringGroup::Rrev:
60 return getRrevCentered();
61 default:
62 throw std::invalid_argument("Unknown centering type.");
63 }
64}
