35void ProcessBankSplitTask::operator()(const tbb::blocked_range<size_t> &range) const {
36 auto entry = m_h5file.openGroup("entry"); // type=NXentry
37 for (size_t wksp_index = range.begin(); wksp_index < range.end(); ++wksp_index) {
38 const auto &bankName = this->bankName(wksp_index);
39 // empty bank names indicate spectra to skip; control should never get here, but just in case
40 if (bankName.empty()) {
41 continue;
42 }
43 Kernel::Timer timer;
44 g_log.debug() << bankName << " start" << std::endl;
45
46 // open the bank
47 auto event_group = entry.openGroup(bankName); // type=NXevent_data
48
49 // skip empty dataset
50 auto tof_SDS = event_group.openDataSet(NxsFieldNames::TIME_OF_FLIGHT);
51 const int64_t total_events = static_cast<size_t>(tof_SDS.getSpace().getSelectNpoints());
52 if (total_events == 0) {
53 m_progress->report();
54 continue;
55 }
56
57 auto eventSplitRanges = this->getEventIndexSplitRanges(event_group, total_events);
58
59 // Get all spectra for this bank.
60 // Create temporary y arrays for each workspace.
61 std::vector<API::ISpectrum *> spectra;
62 std::vector<std::vector<uint32_t>> y_temps;
63 for (const auto &wksp : m_wksps) {
64 spectra.push_back(&wksp->getSpectrum(wksp_index));
65 y_temps.emplace_back(spectra.back()->dataY().size());
66 }
67
68 // get handle to the data
69 auto detID_SDS = event_group.openDataSet(NxsFieldNames::DETID);
70 // auto tof_SDS = event_group.openDataSet(NxsFieldNames::TIME_OF_FLIGHT);
71 // and the units
72 std::string tof_unit;
73 Nexus::H5Util::readStringAttribute(tof_SDS, "units", tof_unit);
74 // now the calibration for the output group can be created
75 // which detectors go into the current group - assumes ouput spectrum number is one more than workspace index
76 const auto calibration = this->getCalibration(tof_unit, wksp_index);
77
78 // declare arrays once so memory can be reused
79 auto event_detid = std::make_unique<std::vector<uint32_t>>(); // uint32 for ORNL nexus file
80 auto event_time_of_flight = std::make_unique<std::vector<float>>(); // float for ORNL nexus files
81
82 // read parts of the bank at a time until all events are processed
83 while (!eventSplitRanges.empty()) {
84 // Create offsets and slab sizes for the next chunk of events.
85 // This will read at most m_events_per_chunk events from the file
86 // and will split the ranges if necessary for the next iteration.
87 std::vector<size_t> offsets;
88 std::vector<size_t> slabsizes;
89 std::vector<std::pair<int, EventROI>> relative_target_ranges;
90
91 size_t total_events_to_read = 0;
92 // Process the event ranges until we reach the desired number of events to read or run out of ranges
93 while (!eventSplitRanges.empty() && total_events_to_read < m_events_per_chunk) {
94 // Get the next event range from the stack
95 auto [target, eventRange] = eventSplitRanges.top();
96 eventSplitRanges.pop();
97
98 size_t range_size = eventRange.second - eventRange.first;
99 size_t remaining_chunk = m_events_per_chunk - total_events_to_read;
100
101 // If the range size is larger than the remaining chunk, we need to split it
102 if (range_size > remaining_chunk) {
103 // Split the range: process only part of it now, push the rest back for later
104 relative_target_ranges.emplace_back(target,
105 EventROI(total_events_to_read, total_events_to_read + remaining_chunk));
106 offsets.push_back(eventRange.first);
107 slabsizes.push_back(remaining_chunk);
108 total_events_to_read += remaining_chunk;
109 // Push the remainder of the range back to the front for next iteration
110 eventSplitRanges.emplace(target, EventROI(eventRange.first + remaining_chunk, eventRange.second));
111 break;
112 } else {
113 relative_target_ranges.emplace_back(target,
114 EventROI(total_events_to_read, total_events_to_read + range_size));
115 offsets.push_back(eventRange.first);
116 slabsizes.push_back(range_size);
117 total_events_to_read += range_size;
118 // Continue to next range
119 }
120 }
121
122 // log the event ranges being processed
123 g_log.debug(toLogString(bankName, total_events_to_read, offsets, slabsizes));
124
125 if (total_events_to_read == 0) {
126 continue; // nothing to do
127 }
128
129 // load detid and tof at the same time
130 this->loadEvents(detID_SDS, tof_SDS, offsets, slabsizes, event_detid, event_time_of_flight);
131
132 // loop over targets
133 tbb::parallel_for(
134 tbb::blocked_range<size_t>(0, m_workspaceIndices.size()), [&](const tbb::blocked_range<size_t> &r) {
135 for (size_t idx = r.begin(); idx != r.end(); ++idx) {
136 int i = m_workspaceIndices[idx];
137
138 // Precompute indices for this target
139 std::vector<size_t> indices;
140 for (const auto &pair : relative_target_ranges) {
141 if (pair.first == static_cast<int>(i)) {
142 auto [start, end] = pair.second;
143 for (size_t k = start; k < end; ++k) {
144 indices.push_back(k);
145 }
146 }
147 }
148
149 auto event_id_view_for_target =
150 indices | std::views::transform([&event_detid](const auto &k) { return (*event_detid)[k]; });
151 auto event_tof_view_for_target = indices | std::views::transform([&event_time_of_flight](const auto &k) {
152 return (*event_time_of_flight)[k];
153 });
154
155 ProcessEventsTask task(&event_id_view_for_target, &event_tof_view_for_target, &calibration,
156 &spectra[idx]->readX());
157
158 const tbb::blocked_range<size_t> range_info(0, indices.size(), m_grainsize_event);
159 tbb::parallel_reduce(range_info, task);
160
161 std::transform(y_temps[idx].begin(), y_temps[idx].end(), task.y_temp.begin(), y_temps[idx].begin(),
162 std::plus<uint32_t>());
163 }
164 });
165 }
166
167 // copy the data out into the correct spectrum and calculate errors
168 tbb::parallel_for(size_t(0), m_wksps.size(), [&](size_t i) { copyDataToSpectrum(y_temps[i], spectra[i]); });
169
170 g_log.debug() << bankName << " stop " << timer << std::endl;
171 m_progress->report();
172 }
173}
