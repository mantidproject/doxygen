189size_t FunctionGenerator::getParameterIndex(const ParameterReference &ref) const {
190 if (ref.getLocalFunction() == this) {
191 auto index = ref.getLocalIndex();
192 auto np = nParams();
193 if (index < np) {
194 return index;
195 }
196 return np;
197 }
198 checkTargetFunction();
199 return m_target->getParameterIndex(ref) + m_nOwnParams;
