50void XDataConverter::exec() {
51 MatrixWorkspace_sptr inputWS = getProperty("InputWorkspace");
52 if (!isProcessingRequired(inputWS)) {
53 setProperty("OutputWorkspace", inputWS);
54 return;
55 }
56
57 const bool ragged = inputWS->isRaggedWorkspace();
58
59 const auto numSpectra = static_cast<int>(inputWS->getNumberHistograms());
60 size_t numYValues{1};
61 if (!ragged)
62 numYValues = getNewYSize(inputWS);
63 const size_t numXValues = getNewXSize(numYValues);
64 m_sharedX = API::WorkspaceHelpers::sharedXData(inputWS);
65 // Create the new workspace
66 MatrixWorkspace_sptr outputWS = WorkspaceFactory::Instance().create(inputWS, numSpectra, numXValues, numYValues);
67
68 // Copy over the 'vertical' axis
69 if (inputWS->axes() > 1)
70 outputWS->replaceAxis(1, std::unique_ptr<API::Axis>(inputWS->getAxis(1)->clone(outputWS.get())));
71
72 bool isRebinnedWorkspace = inputWS->id() == "RebinnedOutput";
73 RebinnedOutput_sptr outRB = std::dynamic_pointer_cast<RebinnedOutput>(outputWS);
74 // Force fractions to unity (converting from histo to point discards bin info).
75
76 Progress prog(this, 0.0, 1.0, numSpectra);
77 PARALLEL_FOR_IF(Kernel::threadSafe(*inputWS, *outputWS))
78 for (int i = 0; i < int(numSpectra); ++i) {
79 PARALLEL_START_INTERRUPT_REGION
80
81 if (ragged)
82 outputWS->resizeHistogram(i, inputWS->histogramSize(i));
83
84 // Copy over the Y and E data
85 outputWS->setSharedY(i, inputWS->sharedY(i));
86 outputWS->setSharedE(i, inputWS->sharedE(i));
87 if (isRebinnedWorkspace && outRB) {
88 MantidVecPtr outF;
89 outF.access().resize(inputWS->getNumberBins(i), 1.0);
90 outRB->setF(i, outF);
91 }
92 setXData(outputWS, inputWS, i);
93 if (inputWS->hasDx(i)) {
94 outputWS->setSharedDx(i, inputWS->sharedDx(i));
95 }
96 prog.report();
97
98 PARALLEL_END_INTERRUPT_REGION
99 }
100 PARALLEL_CHECK_INTERRUPT_REGION
101
102 // Store the output
103 setProperty("OutputWorkspace", outputWS);
104}

117void XDataConverter::setXData(const API::MatrixWorkspace_sptr &outputWS, const API::MatrixWorkspace_sptr &inputWS,
118 const int index) {
119 if (m_sharedX) {

124 }
125 outputWS->setSharedX(index, m_cachedX);
126 } else {
127 outputWS->setSharedX(index, calculateXPoints(inputWS->sharedX(index)));
