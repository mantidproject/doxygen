39 const Mantid::Geometry::Instrument_const_sptr &instrument,

42 if (algProperty->isDefault()) {

68template <typename T>
69std::optional<T> checkForOptionalInstrumentDefault(const Mantid::API::Algorithm *const alg, const std::string &propName,
70 const Mantid::Geometry::Instrument_const_sptr &instrument,
71 const std::string &idf_name) {
72 const auto *algProperty = alg->getPointerToProperty(propName);
73 if (algProperty->isDefault()) {
74 auto defaults = instrument->getNumberParameter(idf_name);
75 if (!defaults.empty()) {
76 return std::optional<T>(static_cast<T>(defaults[0]));
77 } else {
78 return std::nullopt;
79 }
80 } else {
81 double value = boost::lexical_cast<double, std::string>(algProperty->value());
82 return std::optional<T>(static_cast<T>(value));
83 }
