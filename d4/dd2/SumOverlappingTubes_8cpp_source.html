87void SumOverlappingTubes::exec() {
88 getInputParameters();
89
90 m_progress = std::make_unique<Progress>(this, 0.0, 1.0, m_workspaceList.size());
91
92 // we need histogram data with m_numPoints bins
93 HistogramData::BinEdges x(m_numPoints + 1, LinearGenerator(m_startScatteringAngle, m_stepScatteringAngle));
94
95 MatrixWorkspace_sptr outputWS = create<Workspace2D>(m_numHistograms, x);
96 outputWS->setDistribution(false);
97 outputWS->setSharedRun(m_workspaceList.front()->sharedRun());
98
99 auto newAxis = std::make_unique<NumericAxis>(m_heightAxis);
100 newAxis->setUnit("Label");
101 auto yLabelUnit = std::dynamic_pointer_cast<Kernel::Units::Label>(newAxis->unit());
102 yLabelUnit->setLabel("Height", "m");
103 newAxis->unit() = yLabelUnit;
104 outputWS->replaceAxis(1, std::move(newAxis));
105
106 outputWS->getAxis(0)->unit() = Kernel::UnitFactory::Instance().create("Label");
107 Unit_sptr xUnit = outputWS->getAxis(0)->unit();
108 std::shared_ptr<Units::Label> xLabel = std::dynamic_pointer_cast<Units::Label>(xUnit);
109 xLabel->setLabel("Scattering Angle", "degrees");
110
111 const auto normalisation = performBinning(outputWS);
112
113 if (getProperty("Normalise")) {
114 PARALLEL_FOR_NO_WSP_CHECK()
115 for (int64_t ii = 0; ii < static_cast<int64_t>(m_numPoints); ++ii)
116 for (size_t j = 0; j < m_numHistograms; ++j) {
117 // Avoid spurious normalisation for low counting cells
118 const auto i = static_cast<size_t>(ii);
119 if (normalisation[j][i] < 1e-15)
120 continue;
121 outputWS->mutableY(j)[i] /= normalisation[j][i];
122 outputWS->mutableE(j)[i] /= normalisation[j][i];
123 }
124 }
125
126 setProperty("OutputWorkspace", outputWS);
127}

148void SumOverlappingTubes::getScatteringAngleBinning() {
149 m_startScatteringAngle = 180.0;
150 m_endScatteringAngle = -180.0;
151
152 // Loop to check minimum and maximum extents for workspace
153 for (auto &ws : m_workspaceList) {
154 const auto &specInfo = ws->spectrumInfo();
155 for (size_t i = 0; i < specInfo.size(); ++i) {
156 if (specInfo.isMonitor(i) || specInfo.isMasked(i))
157 continue;
158 const auto &pos = specInfo.position(i);
159 const double theta = atan2(pos.X(), pos.Z()) * m_mirrorDetectors * 180 / M_PI;
160 m_startScatteringAngle = std::min(m_startScatteringAngle, theta);
161 m_endScatteringAngle = std::max(m_endScatteringAngle, theta);
162 }
163 }
164
165 const std::vector<double> scatteringBinning = getProperty("ScatteringAngleBinning");
166 if (scatteringBinning.size() == 1) {
167 m_stepScatteringAngle = scatteringBinning[0];
168 // Extend the boundaries by half of the step size
169 m_startScatteringAngle -= m_stepScatteringAngle / 2.;
170 m_endScatteringAngle += m_stepScatteringAngle / 2.;
171 } else if (scatteringBinning.size() == 3) {
172 if (scatteringBinning[0] > m_startScatteringAngle || scatteringBinning[2] < m_endScatteringAngle)
173 g_log.warning() << "Some detectors outside of scattering angle range.\n";
174 m_startScatteringAngle = scatteringBinning[0];
175 m_stepScatteringAngle = scatteringBinning[1];
176 m_endScatteringAngle = scatteringBinning[2];
177 }
178
179 if (getProperty("CropNegativeScatteringAngles")) {
180 if (m_endScatteringAngle < 0) {
181 throw std::runtime_error("No positive scattering angle range");
182 }
183 if (m_startScatteringAngle < 0) {
184 m_startScatteringAngle += std::floor(-m_startScatteringAngle / m_stepScatteringAngle) * m_stepScatteringAngle;
185 }
186 }
187
188 m_numPoints =
189 static_cast<size_t>(std::floor((m_endScatteringAngle - m_startScatteringAngle) / m_stepScatteringAngle));
190 g_log.information() << "Number of bins:" << m_numPoints << std::endl;
191 g_log.information() << "Scattering angle binning:" << m_startScatteringAngle << ", " << m_stepScatteringAngle << ", "
192 << m_endScatteringAngle << "\n";
193
194 if (m_startScatteringAngle >= m_endScatteringAngle) {
195 throw std::runtime_error("Wrong scattering angle range, check your binning/data");
196 }
197}

199void SumOverlappingTubes::getHeightAxis(const std::string &componentName) {
200 std::vector<double> heightBinning = getProperty("HeightAxis");
201 m_heightAxis.clear();
202 if (componentName.length() == 0 && heightBinning.empty())
203 throw std::runtime_error("No detector_for_height_axis parameter for this "
204 "instrument. Please enter a value for the "
205 "HeightAxis parameter.");
206 if ((componentName.length() > 0 && heightBinning.empty()) || (m_outputType != "1D" && heightBinning.size() == 2)) {
207 // Try to get the component. It should be a tube with pixels in the
208 // y-direction, the height bins are then taken as the detector positions.
209 const auto &componentInfo = m_workspaceList.front()->componentInfo();
210 const auto componentIndex = componentInfo.indexOfAny(componentName);
211 const auto &detsInSubtree = componentInfo.detectorsInSubtree(componentIndex);
212 for (const auto detIndex : detsInSubtree) {
213 const auto posY = componentInfo.position({detIndex, 0}).Y();
214 if (heightBinning.size() == 2 && (posY < heightBinning[0] || posY > heightBinning[1]))
215 continue;
216 m_heightAxis.emplace_back(posY);
217 }
218 } else {
219 if (heightBinning.size() != 3) {
220 if (heightBinning.size() == 2 && m_outputType == "1D") {
221 m_heightAxis.emplace_back(heightBinning[0]);
222 m_heightAxis.emplace_back(heightBinning[1]);
223 } else
224 throw std::runtime_error("Height binning must have start, step and end "
225 "values (except for 1D option).");
226 } else if (m_outputType == "1D") {
227 m_heightAxis.emplace_back(heightBinning[0]);
228 m_heightAxis.emplace_back(heightBinning[2]);
229 } else {
230 double height = heightBinning[0];
231 while (height < heightBinning[2]) {
232 m_heightAxis.emplace_back(height);
233 height += heightBinning[1];
234 }
235 }
236 }
237
238 if (m_heightAxis.size() == 0) {
239 throw std::runtime_error("Height axis is undefined");
240 }
241
242 m_startHeight = *min_element(m_heightAxis.begin(), m_heightAxis.end());
243 m_endHeight = *max_element(m_heightAxis.begin(), m_heightAxis.end());
244
245 if (m_outputType == "1D") {
246 const double singleValue = (m_heightAxis.front() + m_heightAxis.back()) * 0.5;
247 m_heightAxis.clear();
248 m_heightAxis.emplace_back(singleValue);
249 }
250
251 m_numHistograms = m_heightAxis.size();
252
253 g_log.information() << "Number of histograms in output workspace:" << m_numHistograms << ".\n";
254 g_log.information() << "Height axis:" << m_heightAxis[0] << " to " << m_heightAxis[m_numHistograms - 1] << " with "
255 << m_heightAxis.size() << " entries.\n";
256}

258std::vector<std::vector<double>> SumOverlappingTubes::performBinning(MatrixWorkspace_sptr &outputWS) {
259 const double scatteringAngleTolerance = getProperty("ScatteringAngleTolerance");
260 const bool splitCounts = getProperty("SplitCounts");
261
262 std::vector<std::vector<double>> normalisation(m_numHistograms, std::vector<double>(m_numPoints, 0.0));
263
264 // loop over all workspaces
265 for (auto &ws : m_workspaceList) {
266 m_progress->report("Processing workspace " + std::string(ws->getName()));
267 // loop over spectra
268 const auto &specInfo = ws->spectrumInfo();
269 PARALLEL_FOR_IF(Kernel::threadSafe(*ws, *outputWS))
270 for (int i = 0; i < static_cast<int>(specInfo.size()); ++i) {
271 PARALLEL_START_INTERRUPT_REGION
272 if (specInfo.isMonitor(i) || specInfo.isMasked(i))
273 continue;
274
275 const auto &pos = specInfo.position(i);
276 const auto height = pos.Y();
277
278 const double tolerance = 1e-6;
279 if (height < m_startHeight - tolerance || height > m_endHeight + tolerance)
280 continue;
281
282 size_t heightIndex;
283 try {
284 heightIndex = Kernel::VectorHelper::indexOfValueFromCenters(m_heightAxis, height);
285 } catch (std::out_of_range &) {
286 continue;
287 }
288
289 double angle;
290 if (m_outputType == "2DTubes")
291 angle = atan2(pos.X(), pos.Z());
292 else
293 angle = specInfo.signedTwoTheta(i);
294 angle *= m_mirrorDetectors * 180.0 / M_PI;
295
296 const auto angleIndex = static_cast<int>(std::floor((angle - m_startScatteringAngle) / m_stepScatteringAngle));
297
298 // point is out of range, a warning should have been generated already for
299 // the theta index
300 if (angleIndex < 0 || angleIndex >= int(m_numPoints))
301 continue;
302
303 const double deltaAngle = distanceFromAngle(angleIndex, angle);
304 const auto counts = ws->histogram(i).y()[0];
305 const auto error = ws->histogram(i).e()[0];
306
307 PARALLEL_CRITICAL(Histogramming2ThetaVsHeight) {
308 auto &yData = outputWS->mutableY(heightIndex);
309 auto &eData = outputWS->mutableE(heightIndex);
310 // counts are split between bins if outside this tolerance
311 if (splitCounts && deltaAngle > m_stepScatteringAngle * scatteringAngleTolerance) {
312 int angleIndexNeighbor;
313 if (distanceFromAngle(angleIndex - 1, angle) < distanceFromAngle(angleIndex + 1, angle))
314 angleIndexNeighbor = angleIndex - 1;
315 else
316 angleIndexNeighbor = angleIndex + 1;
317
318 double deltaAngleNeighbor = distanceFromAngle(angleIndexNeighbor, angle);
319
320 const auto scalingFactor = deltaAngleNeighbor / m_stepScatteringAngle;
321 const auto newError = error * scalingFactor;
322 yData[angleIndex] += counts * scalingFactor;
323 eData[angleIndex] = sqrt(eData[angleIndex] * eData[angleIndex] + newError * newError);
324
325 normalisation[heightIndex][angleIndex] += (deltaAngleNeighbor / m_stepScatteringAngle);
326
327 if (angleIndexNeighbor >= 0 && angleIndexNeighbor < int(m_numPoints)) {
328 const auto scalingFactorNeighbor = deltaAngle / m_stepScatteringAngle;
329 const auto newErrorNeighbor = error * scalingFactorNeighbor;
330 yData[angleIndexNeighbor] += counts * scalingFactorNeighbor;
331 eData[angleIndexNeighbor] =
332 sqrt(eData[angleIndexNeighbor] * eData[angleIndexNeighbor] + newErrorNeighbor * newErrorNeighbor);
333 normalisation[heightIndex][angleIndexNeighbor] += (deltaAngle / m_stepScatteringAngle);
334 }
335 } else {
336 yData[angleIndex] += counts;
337 eData[angleIndex] = sqrt(eData[angleIndex] * eData[angleIndex] + error * error);
338 normalisation[heightIndex][angleIndex]++;
339 }
340 }
341 PARALLEL_END_INTERRUPT_REGION
342 }
343 PARALLEL_CHECK_INTERRUPT_REGION
344 }
345
346 return normalisation;
347}

349double SumOverlappingTubes::distanceFromAngle(const int angleIndex, const double angle) const {
350 return fabs(m_startScatteringAngle + double(angleIndex) * m_stepScatteringAngle - angle);
351}
