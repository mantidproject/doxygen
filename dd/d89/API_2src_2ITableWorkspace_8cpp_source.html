70void ITableWorkspace::modified() {
71 if (!AnalysisDataService::Instance().doesExist(this->getName()))
72 return;
73 Workspace_sptr ws = AnalysisDataService::Instance().retrieve(this->getName());
74 if (!ws)
75 return;
76 ITableWorkspace_sptr tws = std::dynamic_pointer_cast<ITableWorkspace>(ws);
77 if (!tws)
78 return;
79 AnalysisDataService::Instance().notificationCenter.postNotification(
80 new Kernel::DataService<API::Workspace>::AfterReplaceNotification(this->getName(), tws));
81}
