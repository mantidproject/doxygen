68void copyDataToSpectrum(const std::vector<uint32_t> &y_temp, API::ISpectrum *spectrum) {
69 // copy the data out into the correct spectrum and calculate errors
70 auto &y_values = spectrum->dataY();
71 std::copy(y_temp.cbegin(), y_temp.cend(), y_values.begin());
72 auto &e_values = spectrum->dataE();
73 std::transform(y_temp.cbegin(), y_temp.cend(), e_values.begin(),
74 [](uint32_t y) { return std::sqrt(static_cast<double>(y)); });
75}
