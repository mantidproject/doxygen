31void SeqDomain::getDomainAndValues(size_t i, API::FunctionDomain_sptr &domain, API::FunctionValues_sptr &values) const {
32 if (i >= m_creators.size())

73void SeqDomain::additiveCostFunctionVal(const CostFunctions::CostFuncFitting &costFunction) {
74 API::FunctionDomain_sptr domain;
75 API::FunctionValues_sptr values;
76 const size_t n = getNDomains();
77 for (size_t i = 0; i < n; ++i) {
78 values.reset();
79 getDomainAndValues(i, domain, values);
80 if (!values) {
81 throw std::runtime_error("CostFunction: undefined FunctionValues.");
82 }
83 costFunction.addVal(domain, values);
