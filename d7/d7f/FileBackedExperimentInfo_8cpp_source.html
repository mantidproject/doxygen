50void FileBackedExperimentInfo::populateFromFile() const {
51 try {
52 Nexus::File nxFile(m_filename);
53 nxFile.openAddress(m_nxpath);
54 // The loadExperimentInfo calls things such as mutableSample()
55 // and if m_loaded is not true then this function is
56 // will be called recursively.
57 m_loaded = true;
58
59 std::string parameterStr;
60 const_cast<FileBackedExperimentInfo *>(this)->loadExperimentInfoNexus(m_filename, &nxFile, parameterStr);
61 const_cast<FileBackedExperimentInfo *>(this)->readParameterMap(parameterStr);
62 } catch (Nexus::Exception const &exc) {
63 std::ostringstream os;
64 os << "Unable to load experiment information from NeXus file: " << exc.what() << "\n";
65 throw std::runtime_error(os.str());
66 }
67}
