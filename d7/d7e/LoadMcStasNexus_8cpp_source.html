66void LoadMcStasNexus::exec() {
67 std::string filename = getPropertyValue("Filename");
68 g_log.debug() << "Opening file " << filename << '\n';
69
70 Nexus::File nxFile(filename);
71 auto entries = nxFile.getEntries();
72 auto itend = entries.end();
73 WorkspaceGroup_sptr outputGroup(new WorkspaceGroup);
74
75 for (auto it = entries.begin(); it != itend; ++it) {
76 std::string entryName = it->first;
77 std::string type = it->second;
78 nxFile.openGroup(entryName, type);
79 auto dataEntries = nxFile.getEntries();
80
81 for (auto &dataEntry : dataEntries) {
82 const std::string &dataName = dataEntry.first;
83 const std::string &dataType = dataEntry.second;
84 if (dataName == "content_nxs" || dataType != "NXdata")
85 continue;
86 g_log.debug() << "Opening " << dataName << " " << dataType << '\n';
87
88 nxFile.openGroup(dataName, dataType);
89
90 // Find the axis names
91 auto nxdataEntries = nxFile.getEntries();
92 std::string axis1Name, axis2Name;
93 for (const auto &nxdataEntry : nxdataEntries) {
94 if (nxdataEntry.second == "NXparameters")
95 continue;
96 nxFile.openData(nxdataEntry.first);
97 if (nxFile.hasAttr("axis")) {
98 int axisNo(0);
99 nxFile.getAttr("axis", axisNo);
100 if (axisNo == 1)
101 axis1Name = nxdataEntry.first;
102 else if (axisNo == 2)
103 axis2Name = nxdataEntry.first;
104 else
105 throw std::invalid_argument("Unknown axis number");
106 }
107 nxFile.closeData();
108 }
109
110 std::vector<double> axis1Values, axis2Values;
111 nxFile.readData<double>(axis1Name, axis1Values);
112 nxFile.readData<double>(axis2Name, axis2Values);
113
114 const auto axis1Length = axis1Values.size();
115 const auto axis2Length = axis2Values.size();
116 g_log.debug() << "Axis lengths=" << axis1Length << " " << axis2Length << '\n';
117
118 // Require "data" field
119 std::vector<double> data;
120 nxFile.readData<double>("data", data);
121
122 // Optional errors field
123 std::vector<double> errors;
124 try {
125 nxFile.readData<double>("errors", errors);
126 } catch (Nexus::Exception const &) {
127 g_log.information() << "Field " << dataName << " contains no error information.\n";
128 }
129
130 MatrixWorkspace_sptr ws =
131 WorkspaceFactory::Instance().create("Workspace2D", axis2Length, axis1Length, axis1Length);
132 Axis *axis1 = ws->getAxis(0);
133 axis1->title() = axis1Name;
134 // Set caption
135 auto lblUnit = std::make_shared<Units::Label>();
136 lblUnit->setLabel(axis1Name, "");
137 axis1->unit() = lblUnit;
138
139 auto axis2 = std::make_unique<NumericAxis>(axis2Length);
140 auto axis2Raw = axis2.get();
141 axis2->title() = axis2Name;
142 // Set caption
143 lblUnit = std::make_shared<Units::Label>();
144 lblUnit->setLabel(axis2Name, "");
145 axis2->unit() = lblUnit;
146
147 ws->setYUnit(axis2Name);
148 ws->replaceAxis(1, std::move(axis2));
149
150 ws->mutableX(0) = axis1Values;
151
152 for (size_t wsIndex = 0; wsIndex < axis2Length; ++wsIndex) {
153 auto &dataY = ws->mutableY(wsIndex);
154 auto &dataE = ws->mutableE(wsIndex);
155 ws->setSharedX(wsIndex, ws->sharedX(0));
156
157 for (size_t j = 0; j < axis1Length; ++j) {
158 // Data is stored in column-major order so we are translating to
159 // row major for Mantid
160 const size_t fileDataIndex = j * axis2Length + wsIndex;
161
162 dataY[j] = data[fileDataIndex];
163 if (!errors.empty())
164 dataE[j] = errors[fileDataIndex];
165 }
166 axis2Raw->setValue(wsIndex, axis2Values[wsIndex]);
167 }
168 // Make Mantid store the workspace in the group
169 outputGroup->addWorkspace(ws);
170
171 nxFile.closeGroup();
172 }
173 nxFile.closeGroup();
174 }
175
176 setProperty("OutputWorkspace", outputGroup);
177}
