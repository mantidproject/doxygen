72void LoadPreNexusMonitors::exec() {
73 // time of flight channel parameters
74 double tmin = 0.0;
75 double tstep = 0.0;
76 int tchannels = 0;
77 std::string instrumentName;
78
79 // Vectors to store monitor parameters
80 std::vector<std::string> monitorFilenames;
81 std::vector<int> monitorIDs;
82
83 // Get the Runinfo filename from the property
84 std::string runinfo_filename = this->getPropertyValue(RUNINFO_FILENAME);
85
86 // TODO: Extract the directory that the runinfo file is in.
87 // Create a Poco Path object for runinfo filename
88 std::filesystem::path runinfoPath(runinfo_filename);
89 // Now lets get the directory
90 std::filesystem::path dirPath(runinfoPath.parent_path());
91
92 this->g_log.information("Monitor File Dir: " + dirPath.string());
93
94 // Some XML parsing magic...
95 std::ifstream in(runinfo_filename.c_str());
96 Poco::XML::InputSource src(in);
97
98 Poco::XML::DOMParser parser;
99 Poco::AutoPtr<Poco::XML::Document> pDoc = parser.parse(&src);
100
101 Poco::XML::NodeIterator it(pDoc, Poco::XML::NodeFilter::SHOW_ELEMENT);
102 Poco::XML::Node *pNode = it.nextNode();
103 while (pNode) {
104
105 // Get the beamline name.
106 if (pNode->nodeName() == "RunInfo") {
107 auto *pRunInfoElement = static_cast<Poco::XML::Element *>(pNode);
108 instrumentName = pRunInfoElement->getAttribute("instrument");
109 }
110
111 // Look for the section that contains the binning parameters.
112 // This will end up with using the binning for the last monitor that is
113 // found
114 // TODO: Modify to store (and use) the TCBs for each monitor separately.
115 if (pNode->nodeName() == "BeamMonitorInfo") {
116 // Increment the number of monitors we've found
117 ++nMonitors;
118
119 auto *pE = static_cast<Poco::XML::Element *>(pNode);
120 g_log.debug() << "Beam Monitor " << pE->getAttribute("id") << '\n';
121 g_log.debug() << "\tname: " << pE->getAttribute("name") << '\n';
122 g_log.debug() << "\tdescription: " << pE->getAttribute("description") << '\n';
123
124 // Now lets get the tof binning settings
125 Poco::XML::Element *pTimeChannels = pE->getChildElement("NumTimeChannels");
126 tmin = boost::lexical_cast<double>(pTimeChannels->getAttribute("startbin"));
127 tstep = boost::lexical_cast<double>(pTimeChannels->getAttribute("width"));
128 }
129
130 // Look for the 'DataList' node to get the monitor dims.
131 // TODO: Again we will only use the mast monitor value.
132 if (pNode->nodeName() == "DataList") {
133 // Get a list of the child elements
134 Poco::AutoPtr<Poco::XML::NodeList> pDataListChildren = pNode->childNodes();
135 for (unsigned long i = 0; i < pDataListChildren->length(); ++i) {
136 // We only care about monitors
137 if (pDataListChildren->item(i)->nodeName() == "monitor") {
138 auto *element = static_cast<Poco::XML::Element *>(pDataListChildren->item(i));
139 monitorIDs.emplace_back(boost::lexical_cast<int>(element->getAttribute("id")));
140 monitorFilenames.emplace_back(element->getAttribute("name"));
141 }
142 }
143 }
144
145 // Get the size of the files
146 if (pNode->nodeName() == "FileFormats") {
147 // Get a list of the child elements
148 Poco::AutoPtr<Poco::XML::NodeList> pDataListChildren = pNode->childNodes();
149 for (unsigned long i = 0; i < pDataListChildren->length(); ++i) {
150 // We only care about monitors
151 if (pDataListChildren->item(i)->nodeName() == "monitor") {
152 std::string dims = static_cast<Poco::XML::Element *>(pDataListChildren->item(i))->getAttribute("dims");
153 tchannels = boost::lexical_cast<int>(dims);
154 }
155 }
156 }
157
158 pNode = it.nextNode();
159 }
160
161 g_log.information() << "Found " << nMonitors << " beam monitors.\n";
162
163 g_log.information() << "Number of Time Channels = " << tchannels << '\n';
164
165 // Now lets create the time of flight array.
166 const int numberTimeBins = tchannels + 1;
167 MantidVec time_bins(numberTimeBins);
168 for (int i = 0; i < numberTimeBins; ++i) {
169 time_bins[i] = tmin + (i)*tstep;
170 }
171
172 // Create the new workspace
173 MatrixWorkspace_sptr localWorkspace =
174 WorkspaceFactory::Instance().create("Workspace2D", nMonitors, numberTimeBins, tchannels);
175
176 BinEdges edges(time_bins);
177 for (int i = 0; i < nMonitors; i++) {
178 // Now lets actually read the monitor files..
179 std::filesystem::path pMonitorFilename = dirPath / monitorFilenames[i];
180
181 g_log.debug() << "Loading monitor file :" << pMonitorFilename.string() << '\n';
182
183 Kernel::BinaryFile<uint32_t> monitorFile(pMonitorFilename.string());
184 // temp buffer for file reading
185 std::vector<uint32_t> buffer = monitorFile.loadAllIntoVector();
186
187 localWorkspace->setHistogram(i, edges, Counts(buffer.begin(), buffer.end()));
188
189 // Just have spectrum number be the same as the monitor number but -ve.
190 auto &spectrum = localWorkspace->getSpectrum(i);
191 spectrum.setSpectrumNo(monitorIDs[i]);
192 spectrum.setDetectorID(-monitorIDs[i]);
193 }
194
195 g_log.debug() << "Setting axis zero to TOF\n";
196
197 // Set the units
198 localWorkspace->getAxis(0)->unit() = UnitFactory::Instance().create("TOF");
199 localWorkspace->setYUnit("Counts");
200 // TODO localWorkspace->setTitle(title);
201
202 // Actually load the instrument
203 this->runLoadInstrument(instrumentName, localWorkspace);
204
205 // Set the property
206 setProperty("OutputWorkspace", localWorkspace);
207}

215void LoadPreNexusMonitors::runLoadInstrument(const std::string &instrument,
216 const MatrixWorkspace_sptr &localWorkspace) {
217
218 auto loadInst = createChildAlgorithm("LoadInstrument");
219
220 // Now execute the Child Algorithm. Catch and log any error, but don't stop.
221 bool executionSuccessful(true);
222 try {
223 loadInst->setPropertyValue("InstrumentName", instrument);
224 loadInst->setProperty<MatrixWorkspace_sptr>("Workspace", localWorkspace);
225 loadInst->setProperty("RewriteSpectraMap",
226 false); // We have a custom mapping
227 loadInst->execute();
228
229 // Populate the instrument parameters in this workspace - this works around
230 // a bug
231 localWorkspace->populateInstrumentParameters();
232 } catch (std::invalid_argument &e) {
233 g_log.information() << "Invalid argument to LoadInstrument Child Algorithm : " << e.what() << '\n';
234 executionSuccessful = false;
235 } catch (std::runtime_error &e) {
236 g_log.information() << "Unable to successfully run LoadInstrument Child Algorithm : " << e.what() << '\n';
237 executionSuccessful = false;
238 }
239
240 // If loading instrument definition file fails
241 if (!executionSuccessful) {
242 g_log.error() << "Error loading Instrument definition file\n";
243 } else {
244 this->instrument_loaded_correctly = true;
245 }
246}
