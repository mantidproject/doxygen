340template <typename T> void LoadEventNexus::filterDuringPause(T workspace) {
341 try {
342 if ((!ConfigService::Instance().hasProperty("loadeventnexus.keeppausedevents")) &&
343 (m_ws->run().getLogData("pause")->size() > 1)) {
344 g_log.notice("Filtering out events when the run was marked as paused. "
345 "Set the loadeventnexus.keeppausedevents configuration "
346 "property to override this.");
347
348 auto filter = createChildAlgorithm("FilterByLogValue");
349 filter->setProperty("InputWorkspace", workspace);
350 filter->setProperty("OutputWorkspace", workspace);
351 filter->setProperty("LogName", "pause");
352 // The log value is set to 1 when the run is paused, 0 otherwise.
353 filter->setProperty("MinimumValue", 0.0);
354 filter->setProperty("MaximumValue", 0.0);
355 filter->setProperty("LogBoundary", "Left");
356 filter->execute();
357 }
358 } catch (Exception::NotFoundError &) {
359 // No "pause" log, just carry on
360 }
361}

405void LoadEventNexus::exec() {
406 // Retrieve the filename from the properties
407 m_filename = getPropertyValue("Filename");
408
409 compressEvents = !isDefault(PropertyNames::COMPRESS_TOL);
410 compressTolerance = getProperty(PropertyNames::COMPRESS_TOL);
411 if (compressEvents) {
412 BINMODE mode = getPropertyValue(PropertyNames::COMPRESS_MODE);
413 if (mode == BinningMode::LINEAR)
414 compressTolerance = std::fabs(compressTolerance);
415 else if (mode == BinningMode::LOGARITHMIC)
416 compressTolerance = -1. * std::fabs(compressTolerance);
417 }
418
419 loadlogs = getProperty("LoadLogs");
420
421 // Check to see if the monitors need to be loaded later
422 bool load_monitors = this->getProperty("LoadMonitors");
423
424 // this must make absolutely sure that m_file is a valid (and open)
425 // Nexus::File object
426 safeOpenFile(m_filename);
427
428 setTopEntryName();
429
430 // Initialize progress reporting.
431 int reports = 3;
432 if (load_monitors)
433 reports++;
434 Progress prog(this, 0.0, 0.3, reports);
435
436 // Load the detector events
437 m_ws = std::make_shared<EventWorkspaceCollection>(); // Algorithm currently
438 // relies on an
439
440 // object-level workspace ptr
441 loadEvents(&prog, false); // Do not load monitor blocks
442
443 if (discarded_events > 0) {
444 g_log.information() << discarded_events
445 << " events were encountered coming from pixels which "
446 "are not in the Instrument Definition File."
447 "These events were discarded.\n";
448 }
449
450 // If the run was paused at any point, filter out those events (SNS only, I
451 // think)
452 filterDuringPause(m_ws->getSingleHeldWorkspace());
453
454 // add filename
455 m_ws->mutableRun().addProperty("Filename", m_filename);
456 // Save output
457 this->setProperty("OutputWorkspace", m_ws->combinedWorkspace());
458
459 // close the file since LoadNexusMonitors will take care of its own file
460 // handle
461 m_file->close();
462
463 // Load the monitors with child algorithm 'LoadNexusMonitors'
464 if (load_monitors) {
465 prog.report("Loading monitors");
466 this->runLoadMonitors();
467 }
468}

470std::pair<DateAndTime, DateAndTime> firstLastPulseTimes(Nexus::File &file, Kernel::Logger &logger) {
471 file.openData("event_time_zero");
472 DateAndTime offset;
473 // According to the Nexus standard, if the offset is not present, it implies
474 // the offset is and absolute timestamp, which is relative to the start of
475 // Unix epoch (https://manual.nexusformat.org/classes/base_classes/NXlog.html)
476 if (!file.hasAttr("offset")) {
477 offset = DateAndTime("1970-01-01T00:00:00Z");
478 logger.warning("In firstLastPulseTimes: no ISO8601 offset attribute "
479 "provided for event_time_zero, using UNIX epoch instead");
480 } else {
481 std::string isooffset; // ISO8601 offset
482 file.getAttr("offset", isooffset);
483 offset = DateAndTime(isooffset);
484 }
485 std::string units; // time units
486 if (file.hasAttr("units"))
487 file.getAttr("units", units);
488 // Read in the pulse times
489 auto pulse_times = Nexus::IOHelper::readNexusVector<double>(file, "event_time_zero");
490 // Remember to close the entry
491 file.closeData();
492 if (pulse_times.empty()) {
493 throw std::invalid_argument("Cannot find run start; event_time_zero contains no pulse times");
494 }
495 // Convert to seconds
496 auto conv = Kernel::Units::timeConversionValue(units, "s");
497 return std::make_pair(DateAndTime(pulse_times.front() * conv, 0.0) + offset.totalNanoseconds(),
498 DateAndTime(pulse_times.back() * conv, 0.0) + offset.totalNanoseconds());
499} // namespace DataHandling

512std::size_t numEvents(Nexus::File &file, bool &hasTotalCounts, bool &oldNeXusFileNames, const std::string &prefix) {
513 // try getting the value of total_counts
514 if (hasTotalCounts) {
515 hasTotalCounts = false;
516 if (file.hasData(prefix + "/total_counts")) {
517 try {
518 file.openData("total_counts");
519 auto info = file.getInfo();
520 file.closeData();
521 if (info.type == NXnumtype::UINT64) {
522 uint64_t eventCount;
523 file.readData("total_counts", eventCount);
524 hasTotalCounts = true;
525 return eventCount;
526 }
527 } catch (Nexus::Exception const &) {
528 }
529 }
530 }
531
532 // just get the length of the event pixel ids
533 try {
534 if (oldNeXusFileNames)
535 file.openData("event_pixel_id");
536 else
537 file.openData("event_id");
538 } catch (Nexus::Exception const &) {
539 // Older files (before Nov 5, 2010) used this field.
540 try {
541 file.openData("event_pixel_id");
542 oldNeXusFileNames = true;
543 } catch (Nexus::Exception const &) {
544 // Some groups have neither indicating there are not events here
545 return 0;
546 }
547 }
548
549 size_t numEvents = static_cast<std::size_t>(file.getInfo().dims[0]);
550 file.closeData();
551 return numEvents;
552}

567template <typename T>
568std::shared_ptr<BankPulseTimes>
569LoadEventNexus::runLoadNexusLogs(const std::string &nexusfilename, T localWorkspace, API::Algorithm &alg,
570 bool returnpulsetimes, int &nPeriods,
571 std::unique_ptr<const TimeSeriesProperty<int>> &periodLog) {
572 // --------------------- Load DAS Logs -----------------
573 // The pulse times will be empty if not specified in the DAS logs.
574 // BankPulseTimes * out = NULL;
575 std::shared_ptr<BankPulseTimes> out;
576 auto loadLogs = alg.createChildAlgorithm("LoadNexusLogs");
577
578 // Now execute the Child Algorithm. Catch and log any error, but don't stop.
579 try {
580 alg.getLogger().information() << "Loading logs from NeXus file..."
581 << "\n";
582 loadLogs->setPropertyValue("Filename", nexusfilename);
583 loadLogs->setProperty<API::MatrixWorkspace_sptr>("Workspace", localWorkspace);
584 try {
585 loadLogs->setPropertyValue("NXentryName", alg.getPropertyValue("NXentryName"));
586 } catch (...) {
587 }
588
589 loadLogs->execute();
590
591 const Run &run = localWorkspace->run();
592 // Get the number of periods
593 if (run.hasProperty("nperiods")) {
594 nPeriods = run.getPropertyValueAsType<int>("nperiods");
595 }
596 // Get the period log. Map of DateAndTime to Period int values.
597 if (run.hasProperty("period_log")) {
598 const auto *temp = run.getProperty("period_log");
599 // Check for corrupted period logs
600 std::string status = "";
601 std::unique_ptr<TimeSeriesProperty<int>> tempPeriodLog(dynamic_cast<TimeSeriesProperty<int> *>(temp->clone()));
602 nPeriods = checkForCorruptedPeriods(std::move(tempPeriodLog), periodLog, nPeriods, nexusfilename, status);
603 if (!status.empty())
604 alg.getLogger().warning(status);
605 }
606
607 // If successful, we can try to load the pulse times
608 std::vector<Types::Core::DateAndTime> temp;
609 if (localWorkspace->run().hasProperty("proton_charge")) {
610 const auto *log =
611 dynamic_cast<Kernel::TimeSeriesProperty<double> *>(localWorkspace->mutableRun().getProperty("proton_charge"));
612 if (log)
613 temp = log->timesAsVector();
614 }
615 if (returnpulsetimes)
616 out = std::make_shared<BankPulseTimes>(temp);
617
618 // Use the first pulse as the run_start time.
619 if (!temp.empty()) {
620 if (temp[0] < Types::Core::DateAndTime("1991-01-01T00:00:00"))
621 alg.getLogger().warning() << "Found entries in the proton_charge "
622 "sample log with invalid pulse time!\n";
623
624 Types::Core::DateAndTime run_start = localWorkspace->getFirstPulseTime();
625 // add the start of the run as a ISO8601 date/time string. The start =
626 // first non-zero time.
627 // (this is used in LoadInstrument to find the right instrument file to
628 // use).
629 localWorkspace->mutableRun().addProperty("run_start", run_start.toISO8601String(), true);
630 } else if (run.hasProperty("start_time")) {
631 localWorkspace->mutableRun().addProperty("run_start", run.getProperty("start_time")->value(), true);
632 } else {
633 alg.getLogger().warning() << "Empty proton_charge sample log. You will "
634 "not be able to filter by time.\n";
635 }
637 try {
638 Geometry::Goniometer gm;
639 gm.makeUniversalGoniometer();
640 localWorkspace->mutableRun().setGoniometer(gm, true);
641 } catch (std::runtime_error &) {
642 }
643 } catch (const InvalidLogPeriods &) {
644 // Rethrow so LoadEventNexus fails.
645 // If we don't, Mantid will crash.
646 throw;
647 } catch (...) {
648 alg.getLogger().error() << "Error while loading Logs from SNS Nexus. Some "
649 "sample logs may be missing."
650 << "\n";
651 return out;
652 }
653 return out;

671template <typename T>
672std::shared_ptr<BankPulseTimes> LoadEventNexus::runLoadNexusLogs(
673 const std::string &nexusfilename, T localWorkspace, API::Algorithm &alg, bool returnpulsetimes, int &nPeriods,
674 std::unique_ptr<const TimeSeriesProperty<int>> &periodLog, const std::vector<std::string> &allow_list,
675 const std::vector<std::string> &block_list) {
676 // --------------------- Load DAS Logs -----------------
677 // The pulse times will be empty if not specified in the DAS logs.
678 // BankPulseTimes * out = NULL;
679 std::shared_ptr<BankPulseTimes> out;
680 auto loadLogs = alg.createChildAlgorithm("LoadNexusLogs");
681
682 // Now execute the Child Algorithm. Catch and log any error, but don't stop.
683 try {
684 alg.getLogger().information() << "Loading logs from NeXus file..."
685 << "\n";
686 loadLogs->setPropertyValue("Filename", nexusfilename);
687 loadLogs->setProperty<API::MatrixWorkspace_sptr>("Workspace", localWorkspace);
688 loadLogs->setProperty<std::vector<std::string>>("AllowList", allow_list);
689 loadLogs->setProperty<std::vector<std::string>>("BlockList", block_list);
690
691 try {
692 loadLogs->setPropertyValue("NXentryName", alg.getPropertyValue("NXentryName"));
693 } catch (...) {
694 }
695
696 loadLogs->execute();
697
698 const Run &run = localWorkspace->run();
699 // Get the number of periods
700 if (run.hasProperty("nperiods")) {
701 nPeriods = run.getPropertyValueAsType<int>("nperiods");
702 }
703 // Get the period log. Map of DateAndTime to Period int values.
704 if (run.hasProperty("period_log")) {
705 auto *temp = run.getProperty("period_log");
706 std::string status = "";
707 // Check for corrupted period logs
708 std::unique_ptr<TimeSeriesProperty<int>> tempPeriodLog(dynamic_cast<TimeSeriesProperty<int> *>(temp->clone()));
709 nPeriods = checkForCorruptedPeriods(std::move(tempPeriodLog), periodLog, nPeriods, nexusfilename, status);
710 if (!status.empty())
711 alg.getLogger().warning(status);
712 }
713
714 // If successful, we can try to load the pulse times
715 std::vector<Types::Core::DateAndTime> temp;
716 if (localWorkspace->run().hasProperty("proton_charge")) {
717 auto *log =
718 dynamic_cast<Kernel::TimeSeriesProperty<double> *>(localWorkspace->mutableRun().getProperty("proton_charge"));
719 if (log)
720 temp = log->timesAsVector();
721 }
722 if (returnpulsetimes)
723 out = std::make_shared<BankPulseTimes>(temp);
724
725 // Use the first pulse as the run_start time.
726 if (!temp.empty()) {
727 if (temp[0] < Types::Core::DateAndTime("1991-01-01T00:00:00"))
728 alg.getLogger().warning() << "Found entries in the proton_charge "
729 "sample log with invalid pulse time!\n";
730
731 Types::Core::DateAndTime run_start = localWorkspace->getFirstPulseTime();
732 // add the start of the run as a ISO8601 date/time string. The start =
733 // first non-zero time.
734 // (this is used in LoadInstrument to find the right instrument file to
735 // use).
736 localWorkspace->mutableRun().addProperty("run_start", run_start.toISO8601String(), true);
737 } else {
738 alg.getLogger().warning() << "Empty proton_charge sample log. You will "
739 "not be able to filter by time.\n";
740 }
742 try {
743 Geometry::Goniometer gm;
744 gm.makeUniversalGoniometer();
745 localWorkspace->mutableRun().setGoniometer(gm, true);
746 } catch (std::runtime_error &) {
747 }
748 } catch (const InvalidLogPeriods &) {
749 // Rethrow so LoadEventNexus fails.
750 // If we don't, Mantid will crash.
751 throw;
752 } catch (...) {
753 alg.getLogger().error() << "Error while loading Logs from SNS Nexus. Some "
754 "sample logs may be missing."
755 << "\n";
756 return out;
757 }
758 return out;
759}

775int LoadEventNexus::checkForCorruptedPeriods(std::unique_ptr<TimeSeriesProperty<int>> tempPeriodLog,
776 std::unique_ptr<const TimeSeriesProperty<int>> &periodLog,
777 const int &nPeriods, const std::string &nexusfilename,
778 std::string &status) {
779 const auto valuesAsVector = tempPeriodLog->valuesAsVector();
780 const auto nPeriodsInLog = *std::max_element(valuesAsVector.begin(), valuesAsVector.end());
781 int numberOfValidPeriods = nPeriodsInLog;
782 status = "";
783
784 // Check for historic files
785 if (nPeriodsInLog == 0 && nPeriods == 1) {
786 // "modernize" the local copy here by making period_log
787 // a vector of 1s
788 const std::vector<int> newValues(tempPeriodLog->realSize(), 1);
789 const auto times = tempPeriodLog->timesAsVector();
790 periodLog.reset(new const TimeSeriesProperty<int>("period_log", times, newValues));
791 numberOfValidPeriods = 1;
792 } else if (nPeriodsInLog < nPeriods) {
793 status = boost::str(
794 boost::format(
795 "The number of periods specified in the file (%1%) is greater than the maximum period in the data (%2%).") %
796 nPeriods % nPeriodsInLog);
797 } else if (nPeriodsInLog > nPeriods) {
798 // Sanity check here that period_log only contains period numbers up to
799 // nperiods. These values can be different due to instrument noise, and
800 // cause undescriptive crashes if not caught.
801 // We throw here to make it clear
802 // that the file is corrupted and must be manually assessed.
803 const auto msg = "File " + nexusfilename +
804 " has been corrupted. The log framelog/period_log/value "
805 "contains " +
806 std::to_string(nPeriodsInLog) + " periods, but periods/number contains " +
807 std::to_string(nPeriods) + ". This file should be manually inspected and corrected.";
808 throw InvalidLogPeriods(msg);
809 } else {
810 // periodLog should point to a copy of the period logs
811 periodLog = std::make_unique<const TimeSeriesProperty<int>>(*tempPeriodLog);
812 tempPeriodLog.reset();
813 }
814 return numberOfValidPeriods;
815}

867//-----------------------------------------------------------------------------
876void LoadEventNexus::loadEvents(API::Progress *const prog, const bool monitors) {
877 bool metaDataOnly = getProperty("MetaDataOnly");
878
879 // Get the time filters
880 setTimeFilters(monitors);
881
882 // Get the log filter if provided
883 std::vector<std::string> allow_list = getProperty("AllowList"); // if not empty, only these logs will be loaded
884 std::vector<std::string> block_list = getProperty("BlockList"); // if not empty, these logs won't be loaded
885
886 // The run_start will be loaded from the pulse times.
887 DateAndTime run_start(0, 0);
888 bool takeTimesFromEvents = false;
889 // Initialize the counter of bad TOFs
890 bad_tofs = 0;
891 int nPeriods = 1;
892 auto periodLog = std::make_unique<const TimeSeriesProperty<int>>("period_log");
893
894 bool loadAllLogs = getProperty("LoadAllLogs"); // load all logs, not just DAS logs.
895
896 if (loadlogs) {
897 if (!loadAllLogs) {
898 prog->doReport("Loading DAS logs");
899
900 if (allow_list.empty() && block_list.empty()) {
901 m_allBanksPulseTimes =
902 runLoadNexusLogs<EventWorkspaceCollection_sptr>(m_filename, m_ws, *this, true, nPeriods, periodLog);
903 } else {
904 m_allBanksPulseTimes = runLoadNexusLogs<EventWorkspaceCollection_sptr>(m_filename, m_ws, *this, true, nPeriods,
905 periodLog, allow_list, block_list);
906 }
907
908 try {
909 run_start = m_ws->getFirstPulseTime();
910 } catch (Kernel::Exception::NotFoundError &) {
911 /*
912 This is added to (a) support legacy behaviour of continuing to take
913 times from the proto_charge log, but (b) allowing a fall back of
914 getting run start and end from actual pulse times within the
915 NXevent_data group. Note that the latter is better Nexus compliant.
916 */
917 takeTimesFromEvents = true;
918 }
919 } else {
920 prog->doReport("Loading all logs");
921 try {
922 // Open NeXus file
923 Nexus::File nxHandle(m_filename, NXaccess::READ);
924 LoadHelper::addNexusFieldsToWsRun(nxHandle, m_ws->mutableRun(), "", true);
925 } catch (Nexus::Exception const &e) {
926 g_log.debug() << "Failed to open nexus file \"" << m_filename << "\" in read mode: " << e.what() << "\n";
927 }
928 }
929 } else {
930 g_log.information() << "Skipping the loading of sample logs!\n"
931 << "Reading the start time directly from /" << m_top_entry_name << "/start_time\n";
932 // start_time is read and set
933 m_file->openAddress("/");
934 m_file->openGroup(m_top_entry_name, "NXentry");
935 std::string tmp;
936 m_file->readData("start_time", tmp);
937 m_file->closeGroup();
938 run_start = createFromSanitizedISO8601(tmp);
939 m_ws->mutableRun().addProperty("run_start", run_start.toISO8601String(), true);
940 }
941
942 // set more properties on the workspace
943 Nexus::NexusDescriptor const &descriptor = m_file->getFileDescriptor();
944 try {
945 // this is a static method that is why it is passing the
946 // file object and the file path
947
948 loadEntryMetadata<EventWorkspaceCollection_sptr>(m_filename, m_ws, m_top_entry_name);
949 } catch (std::runtime_error &e) {
950 // Missing metadata is not a fatal error. Log and go on with your life
951 g_log.error() << "Error loading metadata: " << e.what() << '\n';
952 }
953
954 m_ws->setNPeriods(static_cast<size_t>(nPeriods),
955 periodLog); // This is how many workspaces we are going to make.
956
957 // Make sure you have a non-NULL m_allBanksPulseTimes
958 if (m_allBanksPulseTimes == nullptr) {
959 std::vector<DateAndTime> temp;
960 m_allBanksPulseTimes = std::make_shared<BankPulseTimes>(temp);
961 }
962
963 if (!m_ws->getInstrument() || !m_instrument_loaded_correctly) {
964 // Load the instrument (if not loaded before)
965 prog->report("Loading instrument");
966 // Note that closing an re-opening the file is needed here for loading
967 // instruments directly from the nexus file containing the event data.
968 // This may not be needed in the future if both LoadEventNexus and
969 // LoadInstrument are made to use the same Nexus/HDF5 library
970 m_file->close();
971 m_instrument_loaded_correctly = loadInstrument(m_filename, m_ws, m_top_entry_name, this, &descriptor);
972
973 if (!m_instrument_loaded_correctly)
974 throw std::runtime_error("Instrument was not initialized correctly! Loading cannot continue.");
975 // reopen file
976 safeOpenFile(m_filename);
977 }
978
979 // top level file information
980 m_file->openAddress("/");
981 // Start with the base entry
982 m_file->openGroup(m_top_entry_name, "NXentry");
983
984 // Now we want to go through all the bankN_event entries
985 vector<string> bankNames;
986 vector<std::size_t> bankNumEvents;
987 std::string classType = monitors ? "NXmonitor" : "NXevent_data";
988 bool oldNeXusFileNames(false);
989 bool haveWeights = false;
990 auto firstPulseT = DateAndTime::maximum();
991 if (m_file->classTypeExists(classType)) {
992 std::map<std::string, std::string> entries = m_file->getEntries();
993 for (auto entry : entries) {
994 if (entry.second != classType) {
995 continue;
996 }
997
998 // skip entries with junk data
999 if (entry.first == "bank_error_events" || entry.first == "bank_unmapped_events")
1000 continue;
1001
1002 m_file->openGroup(entry.first, classType);
1003 // get the number of events
1004 const std::string prefix = "/" + m_top_entry_name + "/" + entry.first;
1005 bool hasTotalCounts = true;
1006 std::size_t num = numEvents(*m_file, hasTotalCounts, oldNeXusFileNames, prefix);
1007 bankNames.emplace_back(entry.first);
1008 bankNumEvents.emplace_back(num);
1009 if (takeTimesFromEvents && num > 0) {
1010 /* If we are here, we are loading logs, but have failed to establish
1011 * the run_start from the proton_charge log. We are going to get this
1012 * from our event_time_zero instead
1013 */
1014 auto localFirstLast = firstLastPulseTimes(*m_file, this->g_log);
1015 firstPulseT = std::min(firstPulseT, localFirstLast.first);
1016 }
1017 // Look for weights in simulated file
1018 const std::string absoluteEventWeightName = prefix + "/event_weight";
1019 haveWeights = m_file->hasAddress(absoluteEventWeightName);
1020 m_file->closeGroup();
1021 }
1022 }
1023 if (takeTimesFromEvents)
1024 run_start = firstPulseT;
1025
1026 loadSampleDataISIScompatibility(*m_file, *m_ws);
1027
1028 // Close the 'top entry' group (raw_data_1 for NexusProcessed, etc.)
1029 m_file->closeGroup();
1030
1031 // Delete the output workspace name if it existed
1032 std::string outName = getPropertyValue("OutputWorkspace");
1033 if (AnalysisDataService::Instance().doesExist(outName))
1034 AnalysisDataService::Instance().remove(outName);
1035
1036 // --------------------------- Time filtering
1037 // ------------------------------------
1038 double filter_time_start_sec, filter_time_stop_sec;
1039 filter_time_start_sec = getProperty("FilterByTimeStart");
1040 filter_time_stop_sec = getProperty("FilterByTimeStop");
1041
1042 // Default to ALL pulse times
1043 filter_time_start = Types::Core::DateAndTime::minimum();
1044 filter_time_stop = Types::Core::DateAndTime::maximum();
1045
1046 if (m_allBanksPulseTimes->numberOfPulses() > 0) {
1047 // If not specified, use the limits of doubles. Otherwise, convert from
1048 // seconds to absolute PulseTime
1049 if (filter_time_start_sec != EMPTY_DBL()) {
1050 filter_time_start = run_start + filter_time_start_sec;
1051 m_is_time_filtered = true;
1052 }
1053
1054 if (filter_time_stop_sec != EMPTY_DBL()) {
1055 filter_time_stop = run_start + filter_time_stop_sec;
1056 m_is_time_filtered = true;
1057 }
1058
1059 // Silly values?
1060 if (filter_time_stop < filter_time_start) {
1061 std::string msg = "Your ";
1062 if (monitors)
1063 msg += "monitor ";
1064 msg += "filter for time's Stop value is smaller than the Start value.";
1065 throw std::invalid_argument(msg);
1066 }
1067 }
1068
1069 // setup filter bad pulses
1070 filter_bad_pulses = !isDefault(PropertyNames::BAD_PULSES_CUTOFF);
1071
1072 if (filter_bad_pulses) {
1073 double min_pcharge, max_pcharge;
1074 std::tie(min_pcharge, max_pcharge, std::ignore) =
1075 m_ws->run().getBadPulseRange(LOG_CHARGE_NAME, getProperty(PropertyNames::BAD_PULSES_CUTOFF));
1076
1077 const auto *pcharge_log =
1078 dynamic_cast<Kernel::TimeSeriesProperty<double> *>(m_ws->run().getLogData(LOG_CHARGE_NAME));
1079 bad_pulses_timeroi = std::make_shared<TimeROI>(
1080 pcharge_log->makeFilterByValue(min_pcharge, max_pcharge, false, TimeInterval(0, 1), 0., true));
1081 }
1082
1083 if (metaDataOnly) {
1084 // Now, create a default X-vector for histogramming, with just 2 bins.
1085 auto axis = HistogramData::BinEdges{1, static_cast<double>(std::numeric_limits<uint32_t>::max()) * 0.1 - 1};
1086 // Set the binning axis using this.
1087 m_ws->setAllX(axis);
1088
1089 createSpectraMapping(m_filename, monitors, std::vector<std::string>());
1090 return;
1091 }
1092
1093 // --------- Loading only one bank ? ----------------------------------
1094 std::vector<std::string> someBanks = getProperty("BankName");
1095 const bool SingleBankPixelsOnly = getProperty("SingleBankPixelsOnly");
1096 if ((!someBanks.empty()) && (!monitors)) {
1097 std::vector<std::string> eventedBanks;
1098 eventedBanks.reserve(someBanks.size());
1099 std::transform(someBanks.cbegin(), someBanks.cend(), std::back_inserter(eventedBanks),
1100 [](const auto &bank) { return bank + "_events"; });
1101
1102 // check that all of the requested banks are in the file
1103 const auto invalidBank =
1104 std::find_if(eventedBanks.cbegin(), eventedBanks.cend(), [&bankNames](const auto &someBank) {
1105 return std::none_of(bankNames.cbegin(), bankNames.cend(),
1106 [&someBank](const auto &name) { return name == someBank; });
1107 });
1108 if (invalidBank != eventedBanks.cend()) {
1109 throw std::invalid_argument("No entry named '" + *invalidBank + "' was found in the .NXS file.");
1110 }
1111
1112 // change the number of banks to load
1113 bankNames.assign(eventedBanks.cbegin(), eventedBanks.cend());
1114
1115 // TODO this equally weights the banks
1116 bankNumEvents.assign(someBanks.size(), 1);
1117
1118 if (!SingleBankPixelsOnly)
1119 someBanks.clear(); // Marker to load all pixels
1120 }
1121
1122 prog->report("Initializing all pixels");
1123
1124 // Remove unused banks if parameter is set
1125 if (m_ws->getInstrument()->hasParameter("remove-unused-banks")) {
1126 std::vector<double> instrumentUnused = m_ws->getInstrument()->getNumberParameter("remove-unused-banks", true);
1127 if (!instrumentUnused.empty()) {
1128 const auto unused = static_cast<int>(instrumentUnused.front());
1129 if (unused == 1)
1130 deleteBanks(m_ws, bankNames);
1131 }
1132 }
1133 //----------------- Pad Empty Pixels -------------------------------
1134 createSpectraMapping(m_filename, monitors, someBanks);
1135
1136 // Set all (empty) event lists as sorted by pulse time. That way, calling
1137 // SortEvents will not try to sort these empty lists.
1138 for (size_t i = 0; i < m_ws->getNumberHistograms(); i++)
1139 m_ws->getSpectrum(i).setSortOrder(DataObjects::PULSETIME_SORT);
1140
1141 // Count the limits to time of flight
1142 shortest_tof = static_cast<double>(std::numeric_limits<uint32_t>::max()) * 0.1;
1143 longest_tof = 0.;
1144
1145 bool precount = getProperty("Precount");
1146 int chunk = getProperty("ChunkNumber");
1147 int totalChunks = getProperty("TotalChunks");
1148 const auto startTime = std::chrono::high_resolution_clock::now();
1149 DefaultEventLoader::load(this, *m_ws, haveWeights, event_id_is_spec, bankNames, periodLog->valuesAsVector(),
1150 classType, bankNumEvents, oldNeXusFileNames, precount, chunk, totalChunks);
1151 addTimer("loadEvents", startTime, std::chrono::high_resolution_clock::now());
1152
1153 // Info reporting
1154 const std::size_t eventsLoaded = m_ws->getNumberEvents();
1155 g_log.information() << "Read " << eventsLoaded << " events"
1156 << ". Shortest TOF: " << shortest_tof << " microsec; longest TOF: " << longest_tof
1157 << " microsec.\n";
1158
1159 if (shortest_tof < 0)
1160 g_log.warning() << "The shortest TOF was negative! At least 1 event has an "
1161 "invalid time-of-flight.\n";
1162 if (bad_tofs > 0)
1163 g_log.warning() << "Found " << bad_tofs
1164 << " events with TOF > 2e8. This "
1165 "may indicate errors in the raw "
1166 "TOF data.\n";
1167
1168 // Use T0 offset from TOPAZ Parameter file if it exists
1169 if (m_ws->getInstrument()->hasParameter("T0")) {
1170 std::vector<double> instrumentT0 = m_ws->getInstrument()->getNumberParameter("T0", true);
1171 if (!instrumentT0.empty()) {
1172 const double mT0 = instrumentT0.front();
1173 if (mT0 != 0.0) {
1174 auto numHistograms = static_cast<int64_t>(m_ws->getNumberHistograms());
1175 PARALLEL_FOR_IF(Kernel::threadSafe(*m_ws))
1176 for (int64_t i = 0; i < numHistograms; ++i) {
1177 PARALLEL_START_INTERRUPT_REGION
1178 // Do the offsetting
1179 m_ws->getSpectrum(i).addTof(mT0);
1180 PARALLEL_END_INTERRUPT_REGION
1181 }
1182 PARALLEL_CHECK_INTERRUPT_REGION
1183 // set T0 in the run parameters
1184 API::Run &run = m_ws->mutableRun();
1185 run.addProperty<double>("T0", mT0, true);
1186 }
1187 }
1188 }
1189 // Now, create a default X-vector for histogramming, with just 2 bins.
1190 if (eventsLoaded > 0) {
1191 int nBins = getProperty("NumberOfBins");
1192 auto binEdgesVec = std::vector<double>(nBins + 1);
1193 binEdgesVec[0] = shortest_tof; // left edge is inclusive
1194 binEdgesVec[nBins] = longest_tof + 1; // right edge is exclusive
1195 double binStep = (binEdgesVec[nBins] - binEdgesVec[0]) / nBins;
1196 for (int binIndex = 1; binIndex < nBins; binIndex++) {
1197 binEdgesVec[binIndex] = binEdgesVec[0] + (binStep * binIndex);
1198 }
1199 m_ws->setAllX(HistogramData::BinEdges{binEdgesVec});
1200 } else
1201 m_ws->setAllX(HistogramData::BinEdges{0.0, 1.0});
1202
1203 // if there is time_of_flight load it
1204 adjustTimeOfFlightISISLegacy(*m_file, m_ws, m_top_entry_name, classType, &descriptor);
1205
1206 if (m_is_time_filtered) {
1207 // events were filtered during read
1208 // filter the logs the same way FilterByTime does
1209 TimeROI timeroi(filter_time_start, filter_time_stop);
1210 if (filter_bad_pulses)
1211 timeroi.update_intersection(*bad_pulses_timeroi);
1212 m_ws->mutableRun().setTimeROI(timeroi);
1213 m_ws->mutableRun().removeDataOutsideTimeROI();
1214 } else if (filter_bad_pulses) {
1215 m_ws->mutableRun().setTimeROI(*bad_pulses_timeroi);
1216 m_ws->mutableRun().removeDataOutsideTimeROI();
1217 }
1218}

1297void LoadEventNexus::deleteBanks(const EventWorkspaceCollection_sptr &workspace,
1298 const std::vector<std::string> &bankNames) {
1299 Instrument_sptr inst = std::const_pointer_cast<Instrument>(workspace->getInstrument()->baseInstrument());
1300 // Build a list of Rectangular Detectors
1301 std::vector<std::shared_ptr<RectangularDetector>> detList;
1302 for (int i = 0; i < inst->nelements(); i++) {
1303 std::shared_ptr<RectangularDetector> det;
1304 std::shared_ptr<ICompAssembly> assem;
1305 std::shared_ptr<ICompAssembly> assem2;
1306
1307 det = std::dynamic_pointer_cast<RectangularDetector>((*inst)[i]);
1308 if (det) {
1309 detList.emplace_back(det);
1310 } else {
1311 // Also, look in the first sub-level for RectangularDetectors (e.g.
1312 // PG3). We are not doing a full recursive search since that will be
1313 // very long for lots of pixels.
1314 assem = std::dynamic_pointer_cast<ICompAssembly>((*inst)[i]);
1315 if (assem) {
1316 for (int j = 0; j < assem->nelements(); j++) {
1317 det = std::dynamic_pointer_cast<RectangularDetector>((*assem)[j]);
1318 if (det) {
1319 detList.emplace_back(det);
1320
1321 } else {
1322 // Also, look in the second sub-level for RectangularDetectors
1323 // (e.g. PG3). We are not doing a full recursive search since that
1324 // will be very long for lots of pixels.
1325 assem2 = std::dynamic_pointer_cast<ICompAssembly>((*assem)[j]);
1326 if (assem2) {
1327 for (int k = 0; k < assem2->nelements(); k++) {
1328 det = std::dynamic_pointer_cast<RectangularDetector>((*assem2)[k]);
1329 if (det) {
1330 detList.emplace_back(det);
1331 }
1332 }
1333 }
1334 }
1335 }
1336 }
1337 }
1338 }
1339 if (detList.empty())
1340 return;
1341 for (auto &det : detList) {
1342 bool keep = false;
1343 std::string det_name = det->getName();
1344 for (const auto &bankName : bankNames) {
1345 size_t pos = bankName.find("_events");
1346 if (det_name == bankName.substr(0, pos))
1347 keep = true;
1348 if (keep)
1349 break;
1350 }
1351 if (!keep) {
1352 std::shared_ptr<const IComponent> parent = inst->getComponentByName(det_name);
1353 std::vector<Geometry::IComponent_const_sptr> children;
1354 std::shared_ptr<const Geometry::ICompAssembly> asmb =
1355 std::dynamic_pointer_cast<const Geometry::ICompAssembly>(parent);
1356 asmb->getChildren(children, false);
1357 for (auto &col : children) {
1358 std::shared_ptr<const Geometry::ICompAssembly> asmb2 =
1359 std::dynamic_pointer_cast<const Geometry::ICompAssembly>(col);
1360 std::vector<Geometry::IComponent_const_sptr> grandchildren;
1361 asmb2->getChildren(grandchildren, false);
1362
1363 for (auto &row : grandchildren) {
1364 auto *d = dynamic_cast<Detector *>(const_cast<IComponent *>(row.get()));
1365 if (d)
1366 inst->removeDetector(d);
1367 }
1368 }
1369 auto *comp = dynamic_cast<IComponent *>(det.get());
1370 inst->remove(comp);
1371 }
1372 }

1470std::unique_ptr<std::pair<std::vector<int32_t>, std::vector<int32_t>>>
1471LoadEventNexus::loadISISVMSSpectraMapping(const std::string &entry_name) {
1472 const std::string vms_str = "/isis_vms_compat";
1473 try {
1474 g_log.debug() << "Attempting to load custom spectra mapping from '" << entry_name << vms_str << "'.\n";
1475 m_file->openAddress("/" + entry_name + vms_str);
1476 } catch (Nexus::Exception const &) {
1477 return nullptr; // Doesn't exist
1478 }
1479
1480 // The ISIS spectrum mapping is defined by 2 arrays in isis_vms_compat
1481 // block:
1482 // UDET - An array of detector IDs
1483 // SPEC - An array of spectrum numbers
1484 // There sizes must match. Hardware allows more than one detector ID to be
1485 // mapped to a single spectrum
1486 // and this is encoded in the SPEC/UDET arrays by repeating the spectrum
1487 // number in the array
1488 // for each mapped detector, e.g.
1489 //
1490 // 1 1001
1491 // 1 1002
1492 // 2 2001
1493 // 3 3001
1494 //
1495 // defines 3 spectra, where the first spectrum contains 2 detectors
1496
1497 // UDET
1498 m_file->openData("UDET");
1499 std::vector<int32_t> udet;
1500 m_file->getData(udet);
1501 m_file->closeData();
1502 // SPEC
1503 m_file->openData("SPEC");
1504 std::vector<int32_t> spec;
1505 m_file->getData(spec);
1506 m_file->closeData();
1507 // Go up/back. this assumes one level for entry name and a second level
1508 // for /isis_vms_compat, typically: /raw_data_1/isis_vms_compat
1509 m_file->closeGroup();
1510 m_file->closeGroup();
1511
1512 // The spec array will contain a spectrum number for each udet but the
1513 // spectrum number
1514 // may be the same for more that one detector
1515 const size_t ndets(udet.size());
1516 if (ndets != spec.size()) {
1517 std::ostringstream os;
1518 os << "UDET/SPEC list size mismatch. UDET=" << udet.size() << ", SPEC=" << spec.size() << "\n";
1519 throw std::runtime_error(os.str());
1520 }
1521 // If mapping loaded the event ID is the spectrum number and not det ID
1522 this->event_id_is_spec = true;
1523 return std::make_unique<std::pair<std::vector<int32_t>, std::vector<int32_t>>>(std::move(spec), std::move(udet));
1524}
