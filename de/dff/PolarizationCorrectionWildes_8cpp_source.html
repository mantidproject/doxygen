68void fourInputsCorrectedAndErrors(Eigen::Vector4d &corrected, Eigen::Vector4d &errors, const double ppy,
69 const double ppyE, const double pmy, const double pmyE, const double mpy,
70 const double mpyE, const double mmy, const double mmyE, const double f1,
71 const double f1E, const double f2, const double f2E, const double p1,
72 const double p1E, const double p2, const double p2E) {
73 using namespace boost::math;
74 // Note that f1 and f2 correspond to 1-F1 and 1-F2 in [Wildes, 1999].
75 // These are inverted forms of the efficiency matrices.
76 const auto diag1 = 1. / f1;
77 const auto off1 = (f1 - 1.) / f1;
78 Eigen::Matrix4d F1m;
79 F1m << 1., 0., 0., 0., 0., 1., 0., 0., off1, 0., diag1, 0., 0., off1, 0., diag1;
80
81 const auto diag2 = 1. / f2;
82 const auto off2 = (f2 - 1.) / f2;
83 Eigen::Matrix4d F2m;
84 F2m << 1., 0., 0., 0., off2, diag2, 0., 0., 0., 0., 1., 0., 0., 0., off2, diag2;
85 const auto diag3 = p1 / (2. * p1 - 1);
86 const auto off3 = (p1 - 1.) / (2. * p1 - 1.);
87 Eigen::Matrix4d P1m;
88 P1m << diag3, 0, off3, 0, 0, diag3, 0, off3, off3, 0, diag3, 0, 0, off3, 0, diag3;
89 const auto diag4 = p2 / (2. * p2 - 1.);
90 const auto off4 = (p2 - 1.) / (2. * p2 - 1.);
91 Eigen::Matrix4d P2m;
92 P2m << diag4, off4, 0., 0., off4, diag4, 0., 0., 0., 0., diag4, off4, 0., 0., off4, diag4;
93 const Eigen::Vector4d intensities(ppy, pmy, mpy, mmy);
94 const auto FProduct = F2m * F1m;
95 const auto PProduct = P2m * P1m;
96 const auto PFProduct = PProduct * FProduct;
97 corrected = PFProduct * intensities;
98 // The error matrices here are element-wise algebraic derivatives of
99 // the matrices above, multiplied by the error.
100 const auto elemE1 = -1. / pow<2>(f1) * f1E;
101 Eigen::Matrix4d F1Em;
102 F1Em << 0., 0., 0., 0., 0., 0., 0., 0., -elemE1, 0., elemE1, 0., 0., -elemE1, 0., elemE1;
103 const auto elemE2 = -1. / pow<2>(f2) * f2E;
104 Eigen::Matrix4d F2Em;
105 F2Em << 0., 0., 0., 0., -elemE2, elemE2, 0., 0., 0., 0., 0., 0., 0., 0., -elemE2, elemE2;
106 const auto elemE3 = 1. / pow<2>(2. * p1 - 1.) * p1E;
107 Eigen::Matrix4d P1Em;
108 P1Em << elemE3, 0., -elemE3, 0., 0., elemE3, 0., -elemE3, -elemE3, 0., elemE3, 0., 0., -elemE3, 0., elemE3;
109 const auto elemE4 = 1. / pow<2>(2. * p2 - 1.) * p2E;
110 Eigen::Matrix4d P2Em;
111 P2Em << elemE4, -elemE4, 0., 0., -elemE4, elemE4, 0., 0., 0., 0., elemE4, -elemE4, 0., 0., -elemE4, elemE4;
112 const Eigen::Vector4d yErrors(ppyE, pmyE, mpyE, mmyE);
113 const auto e1 = (P2Em * P1m * FProduct * intensities).array();
114 const auto e2 = (P2m * P1Em * FProduct * intensities).array();
115 const auto e3 = (PProduct * F2Em * F1m * intensities).array();
116 const auto e4 = (PProduct * F2m * F1Em * intensities).array();
117 const auto sqPFProduct = (PFProduct.array() * PFProduct.array()).matrix();
118 const auto sqErrors = (yErrors.array() * yErrors.array()).matrix();
119 const auto e5 = (sqPFProduct * sqErrors).array();
120 errors = (e1 * e1 + e2 * e2 + e3 * e3 + e4 * e4 + e5).sqrt();
121}

139double twoInputsErrorEstimate01(const double i00, const double e00, const double i11, const double e11, const double p1,
140 const double p1E, const double p2, const double p2E, const double f1, const double f1E,
141 const double f2, const double f2E) {
142 using namespace boost::math;
143 // Derivatives of the equation which solves the I01 intensities
144 // with respect to i00, i11, f1, etc.
145 const auto pmdi00 =
146 -((f1 * (-1. + 2. * p1) * (-f2 * pow<2>(1. - 2. * p2) + pow<2>(f2) * pow<2>(1. - 2. * p2) + (-1. + p2) * p2)) /
147 (f2 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) +
148 f1 * (-1. + 2. * p1) * ((1. - p2) * p2 + f2 * (-1. + p1 + p2) * (-1. + 2. * p2))));
149 const auto pmdi11 = (f2 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2)) /
150 (f2 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) +
151 f1 * (-1. + 2. * p1) * ((1. - p2) * p2 + f2 * (-1. + p1 + p2) * (-1. + 2. * p2)));
152 const auto pmdf1 =
153 -(((-1. + 2. * p1) * ((1. - p2) * p2 + f2 * (-1. + p1 + p2) * (-1. + 2. * p2)) *
154 (f2 * i11 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) -
155 f1 * i00 * (-1. + 2. * p1) *
156 (-f2 * pow<2>(1. - 2. * p2) + pow<2>(f2) * pow<2>(1. - 2. * p2) + (-1. + p2) * p2))) /
157 pow<2>(f2 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) +
158 f1 * (-1. + 2. * p1) * ((1. - p2) * p2 + f2 * (-1. + p1 + p2) * (-1. + 2. * p2)))) -
159 (i00 * (-1. + 2. * p1) * (-f2 * pow<2>(1. - 2. * p2) + pow<2>(f2) * pow<2>(1. - 2. * p2) + (-1. + p2) * p2)) /
160 (f2 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) +
161 f1 * (-1. + 2. * p1) * ((1. - p2) * p2 + f2 * (-1. + p1 + p2) * (-1. + 2. * p2)));
162 const auto pmdf2 =
163 -(((f1 * (-1. + 2. * p1) * (-1. + p1 + p2) * (-1 + 2 * p2) + p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2)) *
164 (f2 * i11 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) -
165 f1 * i00 * (-1. + 2. * p1) *
166 (-f2 * pow<2>(1. - 2. * p2) + pow<2>(f2) * pow<2>(1. - 2. * p2) + (-1. + p2) * p2))) /
167 pow<2>(f2 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) +
168 f1 * (-1. + 2. * p1) * ((1. - p2) * p2 + f2 * (-1. + p1 + p2) * (-1. + 2. * p2)))) +
169 (-f1 * i00 * (-1. + 2. * p1) * (-pow<2>(1. - 2. * p2) + 2 * f2 * pow<2>(1. - 2. * p2)) +
170 i11 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2)) /
171 (f2 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) +
172 f1 * (-1. + 2. * p1) * ((1. - p2) * p2 + f2 * (-1. + p1 + p2) * (-1. + 2. * p2)));
173 const auto pmdp1 =
174 -(((f2 * i11 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) -
175 f1 * i00 * (-1. + 2. * p1) *
176 (-f2 * pow<2>(1. - 2. * p2) + pow<2>(f2) * pow<2>(1. - 2. * p2) + (-1. + p2) * p2)) *
177 (f2 * p1 * (1. - 2. * p2) + f1 * f2 * (-1. + 2. * p1) * (-1. + 2. * p2) +
178 f2 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) +
179 2. * f1 * ((1. - p2) * p2 + f2 * (-1. + p1 + p2) * (-1. + 2. * p2)))) /
180 pow<2>(f2 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) +
181 f1 * (-1. + 2. * p1) * ((1. - p2) * p2 + f2 * (-1. + p1 + p2) * (-1. + 2. * p2)))) +
182 (f2 * i11 * p1 * (1. - 2. * p2) + f2 * i11 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) -
183 2. * f1 * i00 * (-f2 * pow<2>(1. - 2. * p2) + pow<2>(f2) * pow<2>(1. - 2. * p2) + (-1. + p2) * p2)) /
184 (f2 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) +
185 f1 * (-1. + 2. * p1) * ((1. - p2) * p2 + f2 * (-1. + p1 + p2) * (-1. + 2. * p2)));
186 const auto pmdp2 =
187 -(((f2 * i11 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) -
188 f1 * i00 * (-1. + 2. * p1) *
189 (-f2 * pow<2>(1. - 2. * p2) + pow<2>(f2) * pow<2>(1. - 2. * p2) + (-1. + p2) * p2)) *
190 (f2 * (2. - 2. * p1) * p1 +
191 f1 * (-1. + 2. * p1) * (1. - 2. * p2 + 2. * f2 * (-1. + p1 + p2) + f2 * (-1. + 2. * p2)))) /
192 pow<2>(f2 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) +
193 f1 * (-1. + 2. * p1) * ((1. - p2) * p2 + f2 * (-1. + p1 + p2) * (-1. + 2. * p2)))) +
194 (f2 * i11 * (2. - 2. * p1) * p1 -
195 f1 * i00 * (-1. + 2. * p1) * (-1. + 4. * f2 * (1. - 2. * p2) - 4. * pow<2>(f2) * (1. - 2. * p2) + 2. * p2)) /
196 (f2 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) +
197 f1 * (-1. + 2. * p1) * ((1. - p2) * p2 + f2 * (-1. + p1 + p2) * (-1. + 2. * p2)));
198 // Estimate the error components using linearized extrapolation,
199 // sum in squares.
200 const auto e01_I00 = pow<2>(pmdi00 * e00);
201 const auto e01_I11 = pow<2>(pmdi11 * e11);
202 const auto e01_F1 = pow<2>(pmdf1 * f1E);
203 const auto e01_F2 = pow<2>(pmdf2 * f2E);
204 const auto e01_P1 = pow<2>(pmdp1 * p1E);
205 const auto e01_P2 = pow<2>(pmdp2 * p2E);
206 return std::sqrt(e01_I00 + e01_I11 + e01_F1 + e01_F2 + e01_P1 + e01_P2);
207}

225double twoInputsErrorEstimate10(const double i00, const double e00, const double i11, const double e11, const double p1,
226 const double p1E, const double p2, const double p2E, const double f1, const double f1E,
227 const double f2, const double f2E) {
228 using namespace boost::math;
229 // Derivatives of the equation which solves the I10 intensities
230 // with respect to i00, i11, f1, etc.
231 const auto a = -1. + p1 + 2. * p2 - 2. * p1 * p2;
232 const auto b = -1. + 2. * p1;
233 const auto c = -1. + 2. * p2;
234 const auto d = -1. + p2;
235 const auto mpdi00 = (-pow<2>(f1) * f2 * pow<2>(b) * c + f1 * f2 * pow<2>(b) * c + f2 * p1 * a) /
236 (f2 * p1 * a + f1 * b * (-d * p2 + f2 * (p1 + d) * c));
237 const auto mpdi11 = -((f1 * b * d * p2) / (f2 * p1 * a + f1 * b * (-d * p2 + f2 * (p1 + d) * c)));
238 const auto mpdf1 =
239 -(((-1. + 2. * p1) * ((1. - p2) * p2 + f2 * (-1. + p1 + p2) * (-1. + 2. * p2)) *
240 (-pow<2>(f1) * f2 * i00 * pow<2>(1. - 2. * p1) * (-1. + 2. * p2) +
241 f2 * i00 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) +
242 f1 * (-1. + 2. * p1) * (-i11 * (-1. + p2) * p2 + f2 * i00 * (-1. + 2. * p1) * (-1. + 2. * p2)))) /
243 pow<2>(f2 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) +
244 f1 * (-1. + 2. * p1) * ((1. - p2) * p2 + f2 * (-1. + p1 + p2) * (-1. + 2. * p2)))) +
245 (-2. * f1 * f2 * i00 * pow<2>(1. - 2. * p1) * (-1. + 2. * p2) +
246 (-1. + 2. * p1) * (-i11 * (-1. + p2) * p2 + f2 * i00 * (-1. + 2. * p1) * (-1. + 2. * p2))) /
247 (f2 * p1 * (-1. + p1 + 2. * p2 - 2. * p1 * p2) +
248 f1 * (-1. + 2. * p1) * ((1. - p2) * p2 + f2 * (-1. + p1 + p2) * (-1. + 2. * p2)));
249 const auto mpdf2 = -(((f1 * b * (p1 + d) * c + p1 * a) * (-pow<2>(f1) * f2 * i00 * pow<2>(b) * c + f2 * i00 * p1 * a +
250 f1 * b * (-i11 * d * p2 + f2 * i00 * b * c))) /
251 pow<2>(f2 * p1 * a + f1 * b * (-d * p2 + f2 * (p1 + d) * c))) +
252 (-pow<2>(f1) * i00 * pow<2>(b) * c + f1 * i00 * pow<2>(b) * c + i00 * p1 * a) /
253 (f2 * p1 * a + f1 * b * (-d * p2 + f2 * (p1 + d) * c));
254 const auto mpdp1 =
255 -(((-pow<2>(f1) * f2 * i00 * pow<2>(b) * c + f2 * i00 * p1 * a + f1 * b * (-i11 * d * p2 + f2 * i00 * b * c)) *
256 (f2 * p1 * -c + f1 * f2 * b * c + f2 * a + 2. * f1 * (-d * p2 + f2 * (p1 + d) * c))) /
257 pow<2>(f2 * p1 * a + f1 * b * (-d * p2 + f2 * (p1 + d) * c))) +
258 (f2 * i00 * p1 * -c + 4. * pow<2>(f1) * f2 * i00 * -b * c + 2. * f1 * f2 * i00 * b * c + f2 * i00 * a +
259 2. * f1 * (-i11 * d * p2 + f2 * i00 * b * c)) /
260 (f2 * p1 * a + f1 * b * (-d * p2 + f2 * (p1 + d) * c));
261 const auto mpdp2 =
262 -(((f2 * (2. - 2. * p1) * p1 + f1 * b * (1. - 2. * p2 + 2. * f2 * (p1 + d) + f2 * c)) *
263 (-pow<2>(f1) * f2 * i00 * pow<2>(b) * c + f2 * i00 * p1 * a + f1 * b * (-i11 * d * p2 + f2 * i00 * b * c))) /
264 pow<2>(f2 * p1 * a + f1 * b * (-d * p2 + f2 * (p1 + d) * c))) +
265 (-2. * pow<2>(f1) * f2 * i00 * pow<2>(b) + f2 * i00 * (2. - 2. * p1) * p1 +
266 f1 * b * (2. * f2 * i00 * b - i11 * d - i11 * p2)) /
267 (f2 * p1 * a + f1 * b * (-d * p2 + f2 * (p1 + d) * c));
268 // Estimate the error components using linearized extrapolation,
269 // sum in squares.
270 const auto e10_I00 = pow<2>(mpdi00 * e00);
271 const auto e10_I11 = pow<2>(mpdi11 * e11);
272 const auto e10_F1 = pow<2>(mpdf1 * f1E);
273 const auto e10_F2 = pow<2>(mpdf2 * f2E);
274 const auto e10_P1 = pow<2>(mpdp1 * p1E);
275 const auto e10_P2 = pow<2>(mpdp2 * p2E);
276 return std::sqrt(e10_I00 + e10_I11 + e10_F1 + e10_F2 + e10_P1 + e10_P2);
277}

354void PolarizationCorrectionWildes::exec() {
355 const std::string flipperProperty = getProperty(Prop::FLIPPERS);
356 const auto flippers = Kernel::SpinStateHelpers::splitSpinStateString(flipperProperty);
357 // Check if the input flipper configuration includes an analyser
358 const bool hasAnalyser = flippers.front().size() > 1;
359 const auto inputs = mapInputsToDirections(flippers);
360 checkConsistentNumberHistograms(inputs);
361 const EfficiencyMap efficiencies = efficiencyFactors();
362 checkConsistentX(inputs, efficiencies);
363 WorkspaceMap outputs;
364 switch (inputs.size()) {
365 case 1:
366 outputs = directBeamCorrections(inputs, efficiencies);
367 break;
368 case 2:
369 if (hasAnalyser) {
370 outputs = twoInputCorrections(inputs, efficiencies);
371 } else {
372 outputs = analyzerlessCorrections(inputs, efficiencies);
373 }
374 break;
375 case 3:
376 outputs = threeInputCorrections(inputs, efficiencies);
377 break;
378 case 4:
379 outputs = fullCorrections(inputs, efficiencies);
380 }
381 setProperty(Prop::OUTPUT_WS, groupOutput(outputs, hasAnalyser));
382}

388std::map<std::string, std::string> PolarizationCorrectionWildes::validateInputs() {
389 std::map<std::string, std::string> issues;
390 API::MatrixWorkspace_const_sptr factorWS = getProperty(Prop::EFFICIENCIES);
391 if (factorWS) {
392 const auto &factorAxis = factorWS->getAxis(1);
393 if (!factorAxis) {
394 issues[Prop::EFFICIENCIES] = "The workspace is missing a vertical axis.";
395 } else if (!factorAxis->isText()) {
396 issues[Prop::EFFICIENCIES] = "The vertical axis in the workspace is not text axis.";
397 } else if (factorWS->getNumberHistograms() < 4) {
398 issues[Prop::EFFICIENCIES] = "The workspace should contain at least 4 histograms.";
399 } else {
400 std::vector<std::string> tags{{"P1", "P2", "F1", "F2"}};
401 for (size_t i = 0; i != factorAxis->length(); ++i) {
402 const auto label = factorAxis->label(i);
403 auto found = std::find(tags.begin(), tags.end(), label);
404 if (found != tags.cend()) {
405 std::swap(tags.back(), *found);
406 tags.pop_back();
407 }
408 }
409 if (!tags.empty()) {
410 issues[Prop::EFFICIENCIES] = "A histogram labeled " + tags.front() + " is missing from the workspace.";
411 }
412 }
413 }
414 const std::vector<std::string> inputs = getProperty(Prop::INPUT_WS);
415 const auto flipperConfig = Kernel::SpinStateHelpers::splitSpinStateString(getPropertyValue(Prop::FLIPPERS));
416 const auto flipperCount = flipperConfig.size();
417 if (inputs.size() != flipperCount) {
418 issues[Prop::FLIPPERS] = "The number of flipper configurations (" + std::to_string(flipperCount) +
419 ") does not match the number of input workspaces (" + std::to_string(inputs.size()) + ")";
420 }
421 // SpinStates checks.
422 const auto spinStates = Kernel::SpinStateHelpers::splitSpinStateString(getPropertyValue(Prop::SPIN_STATES));
423 if (inputs.size() == 1 && !spinStates.empty()) {
424 issues[Prop::SPIN_STATES] = "Output workspace order cannot be set for direct beam calculations.";
425 } else if (!spinStates.empty()) {

653PolarizationCorrectionWildes::WorkspaceMap
654PolarizationCorrectionWildes::directBeamCorrections(const WorkspaceMap &inputs, const EfficiencyMap &efficiencies) {
655 using namespace boost::math;
656 checkInputExists(inputs.ppWS, FlipperConfigurations::OFF);
657 WorkspaceMap outputs;
658 outputs.ppWS = createWorkspaceWithHistory(inputs.ppWS);
659 const size_t nHisto = inputs.ppWS->getNumberHistograms();
660 // cppcheck-suppress unreadVariable
661 const bool threadSafe = Kernel::threadSafe(*inputs.ppWS, *outputs.ppWS);
662 PARALLEL_FOR_IF(threadSafe)
663 for (int wsIndex = 0; wsIndex < static_cast<int>(nHisto); ++wsIndex) {
664 PARALLEL_START_INTERRUPT_REGION
665 const auto &ppY = inputs.ppWS->y(wsIndex);
666 const auto &ppE = inputs.ppWS->e(wsIndex);
667 auto &ppYOut = outputs.ppWS->mutableY(wsIndex);
668 auto &ppEOut = outputs.ppWS->mutableE(wsIndex);
669 PARALLEL_FOR_IF(threadSafe)
670 for (int binIndex = 0; binIndex < static_cast<int>(ppY.size()); ++binIndex) {
671 const auto P1 = efficiencies.P1->y()[binIndex];
672 const auto P2 = efficiencies.P2->y()[binIndex];
673 const double f = 1. - P1 - P2 + 2. * P1 * P2;
674 ppYOut[binIndex] = ppY[binIndex] / f;
675 const auto P1E = efficiencies.P1->e()[binIndex];
676 const auto P2E = efficiencies.P2->e()[binIndex];
677 const auto e1 = pow<2>(P1E * (2. * P1 - 1.) / pow<2>(f) * ppY[binIndex]);
678 const auto e2 = pow<2>(P2E * (2. * P2 - 1.) / pow<2>(f) * ppY[binIndex]);
679 const auto e3 = pow<2>(ppE[binIndex] / f);
680 const auto errorSum = std::sqrt(e1 + e2 + e3);
681 ppEOut[binIndex] = errorSum;
682 }
683 PARALLEL_END_INTERRUPT_REGION
684 }
685 PARALLEL_CHECK_INTERRUPT_REGION
686 return outputs;
687}

697PolarizationCorrectionWildes::WorkspaceMap
698PolarizationCorrectionWildes::analyzerlessCorrections(const WorkspaceMap &inputs, const EfficiencyMap &efficiencies) {
699 using namespace boost::math;
700 checkInputExists(inputs.mmWS, FlipperConfigurations::ON);
701 checkInputExists(inputs.ppWS, FlipperConfigurations::OFF);
702 WorkspaceMap outputs;
703 outputs.mmWS = createWorkspaceWithHistory(inputs.mmWS);
704 outputs.ppWS = createWorkspaceWithHistory(inputs.ppWS);
705 const size_t nHisto = inputs.mmWS->getNumberHistograms();
706 // cppcheck-suppress unreadVariable
707 const bool threadSafe = Kernel::threadSafe(*inputs.mmWS, *inputs.ppWS, *outputs.mmWS, *outputs.ppWS);
708 PARALLEL_FOR_IF(threadSafe)
709 for (int wsIndex = 0; wsIndex < static_cast<int>(nHisto); ++wsIndex) {
710 PARALLEL_START_INTERRUPT_REGION
711 const auto &mmY = inputs.mmWS->y(wsIndex);
712 const auto &mmE = inputs.mmWS->e(wsIndex);
713 const auto &ppY = inputs.ppWS->y(wsIndex);
714 const auto &ppE = inputs.ppWS->e(wsIndex);
715 auto &mmYOut = outputs.mmWS->mutableY(wsIndex);
716 auto &mmEOut = outputs.mmWS->mutableE(wsIndex);
717 auto &ppYOut = outputs.ppWS->mutableY(wsIndex);
718 auto &ppEOut = outputs.ppWS->mutableE(wsIndex);
719 PARALLEL_FOR_IF(threadSafe)
720 for (int binIndex = 0; binIndex < static_cast<int>(mmY.size()); ++binIndex) {
721 const auto F1 = efficiencies.F1->y()[binIndex];
722 const auto P1 = efficiencies.P1->y()[binIndex];
723 Eigen::Matrix2d F1m;
724 F1m << 1., 0., (F1 - 1.) / F1, 1. / F1;
725 const double divisor = (2. * P1 - 1.);
726 const double off = (P1 - 1.) / divisor;
727 const double diag = P1 / divisor;
728 Eigen::Matrix2d P1m;
729 P1m << diag, off, off, diag;
730 const Eigen::Vector2d intensities(ppY[binIndex], mmY[binIndex]);
731 const auto PFProduct = P1m * F1m;
732 const auto corrected = PFProduct * intensities;
733 ppYOut[binIndex] = corrected[0];
734 mmYOut[binIndex] = corrected[1];
735 const auto F1E = efficiencies.F1->e()[binIndex];
736 const auto P1E = efficiencies.P1->e()[binIndex];
737 const auto elemE1 = -1. / pow<2>(F1) * F1E;
738 Eigen::Matrix2d F1Em;
739 F1Em << 0., 0., -elemE1, elemE1;
740 const auto elemE2 = 1. / pow<2>(divisor) * P1E;
741 Eigen::Matrix2d P1Em;
742 P1Em << elemE2, -elemE2, -elemE2, elemE2;
743 const Eigen::Vector2d errors(ppE[binIndex], mmE[binIndex]);
744 const auto e1 = (P1Em * F1m * intensities).array();
745 const auto e2 = (P1m * F1Em * intensities).array();
746 const auto sqPFProduct = (PFProduct.array() * PFProduct.array()).matrix();
747 const auto sqErrors = (errors.array() * errors.array()).matrix();
748 const auto e3 = (sqPFProduct * sqErrors).array();
749 const auto errorSum = (e1 * e1 + e2 * e2 + e3).sqrt();
750 ppEOut[binIndex] = errorSum[0];
751 mmEOut[binIndex] = errorSum[1];
752 }
753 PARALLEL_END_INTERRUPT_REGION
754 }
755 PARALLEL_CHECK_INTERRUPT_REGION
756 return outputs;
757}

1011 PARALLEL_FOR_IF(threadSafe)

1013 PARALLEL_START_INTERRUPT_REGION
1014 const auto &I00 = inputs.ppWS->y(wsIndex);
1015 const auto &E00 = inputs.ppWS->e(wsIndex);
1016 const auto &I11 = inputs.mmWS->y(wsIndex);
1017 const auto &E11 = inputs.mmWS->e(wsIndex);
1018 auto &I01 = fullInputs.pmWS->mutableY(wsIndex);
1019 auto &E01 = fullInputs.pmWS->mutableE(wsIndex);
1020 auto &I10 = fullInputs.mpWS->mutableY(wsIndex);

1022 PARALLEL_FOR_IF(threadSafe)
1023 for (int binIndex = 0; binIndex < static_cast<int>(I00.size()); ++binIndex) {
1024 const auto i00 = I00[binIndex];
1025 const auto i11 = I11[binIndex];
