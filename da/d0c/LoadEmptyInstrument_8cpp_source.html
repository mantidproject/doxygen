44// This method attempts to retrieve a valid, i.e. enumerated, instrument file name extension.
45// Possible double and single extensions are considered. If no valid file name extension could
46// be retrieved, an std::runtime_error will be thrown. For example, a file name might have a double
47// extension such as ".nxs.h5", which is valid. It may also have a double extension such as .lite.nxs,
48// which is invalid. In the latter case a single extension, .nxs, which is valid, should be retrieved.
49std::string LoadEmptyInstrument::retrieveValidInstrumentFilenameExtension(const std::string &filename) {
50 std::string ext{std::filesystem::path(filename).extension().string()};
51 std::string stem{std::filesystem::path(filename).stem().string()};
52 std::string pre_ext{std::filesystem::path(stem).extension().string()};
53
54 // Test a possible double extension
55 if (!pre_ext.empty()) {
56 try {
57 const std::string double_ext{pre_ext + ext};
58 FilenameExtension fne(double_ext);
59 return fne.c_str();
60 } catch (const std::runtime_error &) {
61 }
62 }
63
64 // Test a possible single extension
65 try {
66 FilenameExtension fne(ext);
67 return fne.c_str();
68 } catch (std::runtime_error &) {
69 }
70
71 std::ostringstream os;
72 os << "Instrument file name \"" << filename << "\" has an invalid extension.";
73 throw std::runtime_error(os.str());
74}

87int LoadEmptyInstrument::confidence(Kernel::FileDescriptor &descriptor) const {
88 const std::string &filePath = descriptor.filename();
89
90 int confidence(0);
91 if (descriptor.isAscii()) // Only consider an Ascii file
92 {
93 // Filename must contain "Definition"
94 std::string::size_type stripPath = filePath.find_last_of("\\/");
95 if (stripPath == std::string::npos)
96 stripPath = 0;
97 if (filePath.find("Definition", stripPath) != std::string::npos) {
98 // We have some confidence and it depends on the filetype.
99 if (descriptor.extension() == "xml") {
100 confidence = 80;
101 } else {
102 confidence = 20;
103 }
104 } // Has "Definition"
105 } // Ascii file
106 return confidence;
107}

145void LoadEmptyInstrument::exec() {
146 Progress prog(this, 0.0, 1.0, 10);
147
148 // load the instrument into this workspace
149 MatrixWorkspace_sptr ws;
150 const std::string instrumentName = getPropertyValue("InstrumentName");
151 const std::string filename = getPropertyValue("Filename");
152 if (!instrumentName.empty())
153 ws = this->runLoadInstrument(filename, instrumentName);
154 else {
155 FilenameExtension enFilenameExtension{retrieveValidInstrumentFilenameExtension(filename)};
156 switch (enFilenameExtension) {
157 case FilenameExtensionEnum::XML:
158 case FilenameExtensionEnum::HDF5:
159 ws = this->runLoadInstrument(filename, instrumentName);
160 break;
161 case FilenameExtensionEnum::NXS:
162 case FilenameExtensionEnum::NXS_H5:
163 ws = this->runLoadIDFFromNexus(filename);
164 break;
165 default:
166 std::ostringstream os;
167 os << "Instrument file name has an invalid extension: "
168 << "\"" << enFilenameExtension.c_str() << "\"";
169 throw std::runtime_error(os.str());
170 }
171 }
172
173 auto timerStart = std::chrono::high_resolution_clock::now();
174 Instrument_const_sptr instrument = ws->getInstrument();
175 addTimer("getInstrument", timerStart, std::chrono::high_resolution_clock::now());
176
177 // Get number of detectors stored in instrument
178 timerStart = std::chrono::high_resolution_clock::now();
179 const size_t number_spectra = instrument->getNumberDetectors();
180 addTimer("getNumberDetectors", timerStart, std::chrono::high_resolution_clock::now());
181
182 // Check that we have some spectra for the workspace
183 if (number_spectra == 0) {
184 g_log.error("Instrument has no detectors, unable to create workspace for it");
185 throw Kernel::Exception::InstrumentDefinitionError("No detectors found in instrument");
186 }
187
188 timerStart = std::chrono::high_resolution_clock::now();
189 Indexing::IndexInfo indexInfo(number_spectra);
190 addTimer("createIndexInfo", timerStart, std::chrono::high_resolution_clock::now());
191 bool MakeEventWorkspace = getProperty("MakeEventWorkspace");
192 prog.reportIncrement(5, "Creating Data");
193 if (MakeEventWorkspace) {
194 setProperty("OutputWorkspace",
195 create<EventWorkspace>(instrument, indexInfo, BinEdges{0.0, std::numeric_limits<double>::min()}));
196 } else {
197 timerStart = std::chrono::high_resolution_clock::now();
198 const double detector_value = getProperty("DetectorValue");
199 const double monitor_value = getProperty("MonitorValue");
200 auto ws2D = create<Workspace2D>(
201 instrument, indexInfo,
202 Histogram(BinEdges{0.0, 1.0}, Counts(1, detector_value), CountStandardDeviations(1, detector_value)));
203 addTimer("makeWorkspace2D", timerStart, std::chrono::high_resolution_clock::now());
204
205 if (monitor_value != detector_value) {
206 timerStart = std::chrono::high_resolution_clock::now();
207 Counts v_monitor_y(1, monitor_value);
208 CountStandardDeviations v_monitor_e(1, monitor_value);
209
210 const auto &spectrumInfo = ws2D->spectrumInfo();
211 const auto size = static_cast<int64_t>(spectrumInfo.size());
212#pragma omp parallel for
213 for (int64_t i = 0; i < size; i++) {
214 if (spectrumInfo.isMonitor(i)) {
215 ws2D->setCounts(i, v_monitor_y);
216 ws2D->setCountStandardDeviations(i, v_monitor_e);
217 }
218 }
219 addTimer("setMonitors", timerStart, std::chrono::high_resolution_clock::now());
220 }
221 setProperty("OutputWorkspace", std::move(ws2D));
222 }
223}

225// Call LoadInstrument as a child algorithm
226API::MatrixWorkspace_sptr LoadEmptyInstrument::runLoadInstrument(const std::string &filename,
227 const std::string &instrumentname) {
228 auto ws = WorkspaceFactory::Instance().create("Workspace2D", 1, 2, 1);
229
230 auto loadInst = createChildAlgorithm("LoadInstrument", 0, 0.5);
231 loadInst->setPropertyValue("Filename", filename);
232 loadInst->setPropertyValue("InstrumentName", instrumentname);
233 loadInst->setProperty("RewriteSpectraMap", OptionalBool(true));
234 loadInst->setProperty<MatrixWorkspace_sptr>("Workspace", ws);
235
236 loadInst->execute();
237
238 return ws;
239}

242API::MatrixWorkspace_sptr LoadEmptyInstrument::runLoadIDFFromNexus(const std::string &filename) {
243 const std::string instrumentEntryName{"/instrument/instrument_xml"};
244 // There are two possibilities for the parent of the instrument IDF entry
245 const std::string instrumentParentEntryName_1{"mantid_workspace_1"};
246 const std::string instrumentParentEntryName_2{"raw_data_1"};
247 std::string instrumentParentEntryName{instrumentParentEntryName_1};
248
249 // Test if instrument XML definition exists in the file
250 bool foundIDF{false};
251 try {
252 Nexus::File nxsfile(filename);
253 nxsfile.openAddress("/" + instrumentParentEntryName + instrumentEntryName);
254 foundIDF = true;
255 } catch (Nexus::Exception const &) {
256 }
257
258 if (!foundIDF) {
259 instrumentParentEntryName = instrumentParentEntryName_2;
260 try {
261 Nexus::File nxsfile(filename);
262 nxsfile.openAddress("/" + instrumentParentEntryName + instrumentEntryName);
263 foundIDF = true;
264 } catch (Nexus::Exception const &) {
265 }
266 }
267
268 if (!foundIDF) {
269 throw std::runtime_error("No instrument XML definition found in " + filename + " at " +
270 instrumentParentEntryName_1 + instrumentEntryName + " or at " +
271 instrumentParentEntryName_2 + instrumentEntryName);
272 }
273
274 auto loadInst = createChildAlgorithm("LoadIDFFromNexus");
275
276 // Execute the child algorithm. Catch and log any error.
277 auto ws = WorkspaceFactory::Instance().create("Workspace2D", 1, 2, 1);
278 try {
279 loadInst->setPropertyValue("Filename", filename);
280 loadInst->setProperty<Mantid::API::MatrixWorkspace_sptr>("Workspace", ws);
281 loadInst->setPropertyValue("InstrumentParentPath", instrumentParentEntryName);
282 loadInst->execute();
283 } catch (std::invalid_argument &) {
284 getLogger().error("Invalid argument to LoadIDFFromNexus Child Algorithm ");
285 } catch (std::runtime_error &) {
286 getLogger().debug("No instrument definition found by LoadIDFFromNexus in " + filename + " at " +
287 instrumentParentEntryName + instrumentEntryName);
288 }
289
290 if (!loadInst->isExecuted())
291 getLogger().information("No IDF loaded from the Nexus file " + filename);
292
293 return ws;
294}
