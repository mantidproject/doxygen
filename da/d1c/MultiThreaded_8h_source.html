67template <typename T, typename BinaryOp> void AtomicOp(std::atomic<T> &f, T d, BinaryOp op) {
68 T old = f.load();
69 T desired;
70 do {
71 desired = op(old, d);
72 } while (!f.compare_exchange_weak(old, desired));
73}
