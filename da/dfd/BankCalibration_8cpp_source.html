16void copy_values_from_map_to_offset_vector(const std::map<detid_t, double> &map_values, const detid_t idmin,
17 const detid_t idmax, std::vector<double> &vector_values) {
18 // allocate memory and set the default value to 1
19 vector_values.assign(static_cast<size_t>(idmax - idmin + 1), 1.);
20
21 // set up iterators for copying data - assumes ordered map
22 auto iter = std::find_if(map_values.cbegin(), map_values.cend(),
23 [idmin](const auto &itervalue) { return itervalue.first >= idmin; });
24 if (iter == map_values.end())
25 throw std::runtime_error("Failed to find idmin=" + std::to_string(idmin) + " in map");
26 auto iter_end =
27 std::find_if(iter, map_values.cend(), [idmax](const auto itervalue) { return itervalue.first > idmax; });
28 if (iter_end != map_values.end())
29 ++iter_end;
30
31 // copy over values that matter
32 for (; iter != iter_end; ++iter) {
33 const auto index = static_cast<size_t>(iter->first - idmin);
34 vector_values[index] = iter->second;
35 }
36}

52BankCalibration::BankCalibration(const double time_conversion, const std::vector<detid_t> &det_in_group,
53 const std::map<detid_t, double> &calibration_map,
54 const std::map<detid_t, double> &scale_at_sample, const std::set<detid_t> &mask) {
55 // determine the range
56 auto [idmin_found, idmax_found] = getDetidRange(det_in_group, calibration_map);
57
58 // all the outputs are vectors that are offset by the minimum detid in the bank
59 m_detid_offset = idmin_found;
60
61 // get the values copied over for calibration
62 copy_values_from_map_to_offset_vector(calibration_map, idmin_found, idmax_found, m_calibration);
63 // apply time conversion here so it is effectively applied for each detector once rather than on each event
64 if (time_conversion != 1.) {
65 std::transform(m_calibration.begin(), m_calibration.end(), m_calibration.begin(),
66 [time_conversion](const auto &value) { return std::move(time_conversion * value); });
67 }
68
69 // get the values copied over for scale_at_sample
70 if (!scale_at_sample.empty())
71 copy_values_from_map_to_offset_vector(scale_at_sample, idmin_found, idmax_found, m_scale_at_sample);
72
73 // mask values that are in the mask or not in the detector group
74 if (det_in_group.empty()) {
75 for (const auto &detid : mask) {
76 if (this->detidInRange(detid)) {
77 m_calibration[detid - m_detid_offset] = IGNORE_PIXEL;
78 }
79 }
80 } else {
81 // mask anything that isn't int the group this assumes both are sorted
82 // find the first and last detector id that is in the range being used
83 auto detid_first = std::lower_bound(det_in_group.cbegin(), det_in_group.cend(), idmin_found);
84 auto detid_last = det_in_group.cend();
85 for (size_t i = 0; i < m_calibration.size(); ++i) {
86 if (m_calibration[i] != IGNORE_PIXEL) {
87 const detid_t detid = static_cast<detid_t>(i) + m_detid_offset;
88 auto detid_found = std::find(detid_first, detid_last, detid);
89 if (detid_found == detid_last) {
90 m_calibration[i] = IGNORE_PIXEL; // not in the list
91 } else {
92 if (mask.contains(detid))
93 m_calibration[i] = IGNORE_PIXEL; // shouldn't use it
94 // next search can start from here
95 detid_first = detid_found;
96 }
97 }
98 }
99 }

102const std::pair<detid_t, detid_t> BankCalibration::getDetidRange(const std::vector<detid_t> &det_in_group,
103 const std::map<detid_t, double> &calibration_map) {
104 if (det_in_group.empty()) {
105 detid_t idminE = calibration_map.begin()->first;
106 detid_t idmaxE = calibration_map.rbegin()->first;
107 return std::make_pair<detid_t, detid_t>(std::move(idminE), std::move(idmaxE));
108 } else {
109 detid_t idminG = det_in_group.front();
110 detid_t idmaxG = det_in_group.back();
111 return std::make_pair<detid_t, detid_t>(std::move(idminG), std::move(idmaxG));
112 }
