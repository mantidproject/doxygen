27void calculateScatteringLengths(NeutronAtom &atom) {
28 // 1 barn = 100 fm
29 atom.tot_scatt_length = 10. * std::sqrt(atom.tot_scatt_xs * INV_FOUR_PI);
30
31 if (atom.coh_scatt_length_img == 0.)
32 atom.coh_scatt_length = std::abs(atom.coh_scatt_length_real);
33 else
34 atom.coh_scatt_length = 10. * std::sqrt(atom.coh_scatt_xs * INV_FOUR_PI);
35
36 if (atom.inc_scatt_length_img == 0.)
37 atom.inc_scatt_length = std::abs(atom.inc_scatt_length_real);
38 else
39 atom.inc_scatt_length = 10. * std::sqrt(atom.inc_scatt_xs * INV_FOUR_PI);
40}
