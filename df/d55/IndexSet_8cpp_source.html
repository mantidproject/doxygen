20IndexSet::IndexSet(int64_t min, int64_t max, size_t fullRange) {

33IndexSet::IndexSet(const std::vector<size_t> &indices, size_t fullRange) : m_isRange(false) {
34 // We use a set to create unique and ordered indices.
