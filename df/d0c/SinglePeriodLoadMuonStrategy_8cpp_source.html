33void SinglePeriodLoadMuonStrategy::loadMuonLogData() {

49void SinglePeriodLoadMuonStrategy::loadGoodFrames() {
50 // Overwrite existing log entry
51 std::string goodframeProp{"goodfrm"};
52 auto &run = m_workspace.mutableRun();
53 run.removeProperty(goodframeProp);
54 NXInt goodframes = m_nexusLoader.loadGoodFramesDataFromNexus(m_isFileMultiPeriod);
55
56 if (m_isFileMultiPeriod) {
57 run.addProperty(goodframeProp, goodframes[m_entryNumber - 1]);
58 } else {
59 run.addProperty(goodframeProp, goodframes[0]);
60 }
61}

69Workspace_sptr SinglePeriodLoadMuonStrategy::loadDetectorGrouping() const {
70
71 auto grouping = m_nexusLoader.loadDetectorGroupingFromNexus(m_detectors, m_isFileMultiPeriod, m_entryNumber);
72 TableWorkspace_sptr table = createDetectorGroupingTable(m_detectors, grouping);
73
74 if (table->rowCount() != 0) {
75 return table;
76 } else {
77 m_logger.notice("Loading grouping information from IDF");
78 return loadDefaultDetectorGrouping(m_workspace);
79 }

101void SinglePeriodLoadMuonStrategy::applyTimeZeroCorrection() {
102 double timeZero = m_nexusLoader.loadTimeZeroFromNexusFile();
103 auto numHistograms = m_workspace.getNumberHistograms();
104 for (size_t i = 0; i < numHistograms; ++i) {
105 auto &timeAxis = m_workspace.mutableX(i);
106 timeAxis = timeAxis - timeZero;
107 }
