50void SinglePeriodLoadMuonStrategy::loadGoodFrames() {
51 // Overwrite existing log entry
52 std::string goodframeProp{"goodfrm"};
53 auto &run = m_workspace.mutableRun();
54 run.removeProperty(goodframeProp);
55 NXInt goodframes = m_nexusLoader.loadGoodFramesDataFromNexus(m_isFileMultiPeriod);
56
57 if (m_isFileMultiPeriod) {
58 run.addProperty(goodframeProp, goodframes[m_entryNumber - 1]);
59 } else {
60 run.addProperty(goodframeProp, goodframes[0]);
61 }

70Workspace_sptr SinglePeriodLoadMuonStrategy::loadDetectorGrouping() const {
71 auto const grouping = m_nexusLoader.loadDetectorGroupingFromNexus(m_detectors, m_isFileMultiPeriod, m_entryNumber);
72 auto const table = createDetectorGroupingTable(m_detectors, grouping);
73 if (table && (*table)->rowCount() != 0) {
74 return *table;
75 }
76 return loadDefaultDetectorGrouping(m_workspace);

98void SinglePeriodLoadMuonStrategy::applyTimeZeroCorrection() {
99 double timeZero = m_nexusLoader.loadTimeZeroFromNexusFile();
100 auto numHistograms = m_workspace.getNumberHistograms();
101 for (size_t i = 0; i < numHistograms; ++i) {
102 auto &timeAxis = m_workspace.mutableX(i);
103 timeAxis = timeAxis - timeZero;
104 }
